בֵּיצָה שֶׁנּוֹלְדָה בְיוֹם טוֹב. [בְּיוֹם טוֹב] שֶׁלְאַחַר שַׁבָּת עָסְקִינַן. וְטַעֲמַיְהוּ דְּבֵית הִלֵּל דְּאָמְרֵי לֹא תֵאָכֵל, מִשּׁוּם דְּכָל בֵּיצָה דְּמִתְיַלְדָא הָאִידְנָא, מֵאֶתְמוֹל גָּמְרָה לַהּ, וְנִמְצֵאת שַׁבָּת מְכִינָה לְיוֹם טוֹב, וְהַתּוֹרָה אָמְרָה (שמות טז) וְהָיָה בַּיּוֹם הַשִּׁשִּׁי וְהֵכִינוּ, וּסְתָם שִׁשִּׁי חֹל הוּא, חֹל מֵכִין לְשַׁבָּת, וְחֹל מֵכִין לְיוֹם טוֹב, דְּיוֹם טוֹב נַמִּי אִקְרֵי שַׁבָּת, וְאֵין יוֹם טוֹב מֵכִין לְשַׁבָּת וְלֹא שַׁבָּת מְכִינָה לְיוֹם טוֹב. וַהֲכָנָה כִּי הַךְ דְּבֵיצָה אַף עַל גַּב דְּבִידֵי שָׁמָיִם הִיא, מִקַּרְיָא הֲכָנָה. וּמִיהוּ שַׁבָּת וְיוֹם טוֹב דַּחֲשִׁיבֵי, בָּעֲיָא סְעוּדָתָן הַזְמָנָה בְּחֹל, וְאֵין אֶחָד מֵהֶן מֵכִין לַחֲבֵרוֹ אֲפִלּוּ בַּהֲכָנָה דְּבִידֵי שָׁמָיִם, אֲבָל סְעוּדַת חֹל לֹא חֲשִׁיבָא וְלֹא בָּעֲיָא הַזְמָנָה, הִלְכָּךְ בְּאֶחָד בְּשַׁבָּת בְּעָלְמָא שֶׁאֵינוֹ יוֹם טוֹב לֵית לָן לְמֵיסַר בֵּיצָה שֶׁנּוֹלְדָה בּוֹ לְפִי שֶׁנִּגְמְרָה בְּשַׁבָּת, דִּסְעוּדַת חֹל לֹא אַצְרְכָהּ רַחֲמָנָא זִמּוּן מִבְּעוֹד יוֹם. וְאָסְרוּ בֵּיצָה שֶׁנּוֹלְדָה בְּכָל יוֹם טוֹב, וְאַף עַל פִּי שֶׁאֵינוֹ אַחַר שַׁבָּת, גְּזֵרָה מִשּׁוּם יוֹם טוֹב שֶׁאַחַר שַׁבָּת. וְכֵן אָסְרוּ בֵּיצָה שֶׁנּוֹלְדָה בְּכָל שַׁבָּת לְגָמְעָהּ חַיָּה, גְּזֵרָה מִשּׁוּם שַׁבָּת שֶׁאַחַר יוֹם טוֹב. וְשַׁבָּת וְיוֹם טוֹב הַסְּמוּכִים זֶה לָזֶה, בֵּיצָה שֶׁנּוֹלְדָה בָּזֶה אֲסוּרָה בָּזֶה. וְכֵן שְׁנֵי יָמִים טוֹבִים שֶׁל רֹאשׁ הַשָּׁנָה. אֲבָל שְׁנֵי יָמִים טוֹבִים שֶׁל גָּלֻיּוֹת, דְּמִמַּה נַּפְשָׁךְ אֶחָד מֵהֶם חֹל, בֵּיצָה שֶׁנּוֹלְדָה בָּזֶה מֻתֶּרֶת בָּזֶה:
שְׂאֹר בְּכַזַּיִת. לְעִנְיַן אֲכִילָה כֻּלֵּי עָלְמָא לֹא פְלִיגֵי דְּזֶה וָזֶה בְּכַזַּיִת, מִדְּפָתַח הַכָּתוּב בִּשְׂאֹר וְסִיֵּם בְּחָמֵץ, שְׂאֹר לֹא יִמָּצֵא בְּבָתֵּיכֶם כִּי כָּל אוֹכֵל מַחֲמֶצֶת (שמות יב), לוֹמַר לְךָ זֶהוּ שְׂאֹר וְזֶהוּ חָמֵץ, כְּזֶה כֵּן זֶה. כִּי פְלִיגֵי לְעִנְיַן בִּעוּר, בֵּית שַׁמַּאי סָבְרֵי מִדְּכָתַב רַחֲמָנָא לְתַרְוַיְהוּ, שְׂאֹר וְחָמֵץ, וְלִכְתּוּב חָמֵץ שֶׁאֵין חִמּוּצוֹ קָשֶׁה וַאֲפִלּוּ הָכִי חַיָּב לְבַעֵר וְקַל וָחֹמֶר לַשְּׂאֹר שֶׁחִמּוּצוֹ קָשֶׁה, אֶלָּא שְׁמַע מִנָּהּ שִׁעוּרוֹ שֶׁל זֶה לֹא כְשִׁעוּרוֹ שֶׁל זֶה, וְלֹא יָלְפִינַן בִּעוּר מֵאֲכִילָה. וּבֵית הִלֵּל סָבְרֵי זֶה וָזֶה בְּכַזַּיִת, דְּיָלְפִינַן בִּעוּר מֵאֲכִילָה. וּמִשּׁוּם דְּהָנָךְ תְּלַת מִלֵּי מִקֻלֵּי בֵּית שַׁמַּאי וּמֵחֻמְרֵי בֵּית הִלֵּל גַּבֵּי יוֹם טוֹב, תָּנָא לְהוּ הָכָא גַּבֵּי הֲדָדֵי:
בְּכַכּוֹתֶבֶת. תְּמָרָה:

=== 2 | chapter 2 ===
יוֹם טוֹב שֶׁחָל לִהְיוֹת עֶרֶב שַׁבָּת לֹא יְבַשֵּׁל בַּתְּחִלָּה. כְּלוֹמַר לֹא יְהֵא תְּחִלַּת בִּשּׁוּלוֹ וְעִקָּרוֹ לְשׁוּם שַׁבָּת, אֶלָּא לְשׁוּם יוֹם טוֹב יְהֵא תְּחִלַּת בִּשּׁוּלוֹ, וּמַה שֶּׁמּוֹתִיר יְהֵא לְשַׁבָּת.
וְעוֹשֶׂה תַבְשִׁיל מֵעֶרֶב יוֹם טוֹב. לְשֵׁם עֵרוּבֵי תַּבְשִׁילִין. אִית דְּאָמְרֵי טַעֲמָא דְּהַאי תַּבְשִׁיל מֵעֶרֶב יוֹם טוֹב, מִשּׁוּם כְּבוֹד שַׁבָּת הוּא, שֶׁמִּתּוֹךְ שֶׁמֵּעֶרֶב יוֹם טוֹב זוֹכֵר הַשַּׁבָּת, יִבְרֹר מָנָה יָפָה לְשַׁבָּת וְלֹא יִשְׁכַּח שַׁבָּת מִשּׁוּם טִרְדַּת יוֹם טוֹב. וְאִית דְּאָמְרֵי מִשּׁוּם כְּבוֹד יוֹם טוֹב הוּא, כְּדֵי שֶׁיִּרְאוּ שֶׁאֵין אוֹפִין מִיּוֹם טוֹב לְשַׁבָּת אֶלָּא אִם כֵּן הִתְחִיל מִבְּעוֹד יוֹם, דִּבְיוֹם טוֹב אֵינוֹ אֶלָּא גּוֹמֵר, קַל וָחֹמֶר מִיּוֹם טוֹב לְחֹל דִּלְגַמְרֵי אֵין אוֹפִין וְלֹא מְבַשְּׁלִין. וּמְבָרְכִים עַל עֵרוּבֵי תַּבְשִׁילִין כְּדֶרֶךְ שֶׁמְּבָרְכִים עַל עֵרוּבֵי חֲצֵרוֹת. וְצָרִיךְ לְזַכּוֹת עַל יְדֵי אַחֵר לְכָל מִי שֶׁרוֹצֶה אַחַר כָּךְ לִסְמֹךְ עַל עֵרוּבוֹ, וַאֲפִלּוּ שֶׁלֹּא מִדַּעְתּוֹ, דְּזָכִין לָאָדָם שֶׁלֹּא בְּפָנָיו:
אֲכָלוֹ אוֹ שֶׁאָבַד. אוֹתוֹ תַּבְשִׁיל שֶׁבִּשֵּׁל, מֵעֶרֶב שַׁבָּת:
סוֹמֵךְ עָלָיו. לְבַשֵּׁל בִּשּׁוּלוֹ בְּיוֹם טוֹב לְצֹרֶךְ שַׁבָּת. אֲבָל מִתְּחִלָּתוֹ אֵין עֵרוּבֵי תַּבְשִׁילִין פָּחוֹת מִכַּזַּיִת, בֵּין לְאֶחָד בֵּין לְמֵאָה. וְאִם הִתְחִיל בְּעִסָּתוֹ וְנֶאֱבַד עֵרוּבוֹ, מַה שֶּׁהִתְחִיל גּוֹמֵר:

אֵין צָדִין דָּגִים. אַף עַל גַּב דִּשְׁחִיטָה וַאֲפִיָּה וּבִשּׁוּל מֵאֲבוֹת מְלָאכוֹת הֵן וְהֻתְּרוּ לְצֹרֶךְ יוֹם טוֹב, צִידָה דָּמְיָא לִקְצִירָה, וּקְצִירָה לֹא הֻתְּרָה בְּיוֹם טוֹב. בֵּיבָרִין שֶׁל דָּגִים, הֵן בְּרֵכוֹת שֶׁל מַיִם שֶׁדָּגִים מִתְגַּדְּלִים שָׁם. בֵּיבָרִים שֶׁל חַיָּה, קַרְפִּיפוֹת מֻקָּפִין גָּדֵר סָבִיב וּמְבִיאִים שָׁם חַיּוֹת וְיוֹלְדוֹת וּמִתְגַּדְּלוֹת שָׁם:
וְאֵין נוֹתְנִין. לִפְנֵי הַדָּגִים מְזוֹנוֹת, דְּאֶפְשָׁר לָהֶם בְּלֹא מְזוֹנוֹת וְאֵין מְזוֹנוֹתָן עָלֶיךָ:
אֲבָל צָדִין חַיָּה וָעוֹף. הַמּוּכָנִים מֵאֶתְמוֹל:
מִן הַבֵּיבָרִין. הַקְּטַנִּים, שֶׁאֵין מְחֻסָּרִים צִידָה. וְרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל לָאו לְאַפְלוּגֵי אַתַּנָּא קַמָּא אָתָא, אֶלָּא לִפְרוּשֵׁי:
כָּל הַמְּחֻסָּר צִידָה. שֶׁצָּרִיךְ לְבַקֵּשׁ תַּחְבּוּלוֹת כְּדֵי לְתָפְסוֹ. וְכָל דְּרָהֵיט בַּתְרֵיהּ וּמָטֵי לֵיהּ בְּחַד שְׁחִיָּה לֹא הָוֵי מְחֻסַּר צִידָה, וְאִי לֹא, הָוֵי מְחֻסַּר צִידָה:

הַמֵּבִיא. מִמָּקוֹם לְמָקוֹם. בְּתוֹךְ הַתְּחוּם:
לֹא יְבִיאֵם וְכוּ'. לָתֵת שָׁלֹשׁ וְאַרְבַּע כַּדּוֹת בְּתוֹךְ סַל אוֹ קֻפָּה וְיִשָּׂאֵם, מִשּׁוּם דְּנִרְאֶה כְּמַעֲשֵׂה חֹל לָשֵׂאת מַשָּׂאוֹת. וְאִם אִי אֶפְשָׁר לוֹ לְשַׁנּוֹת מֻתָּר:

אוֹ לְפָנָיו. בְּיָדוֹ:
וְכֵן הַמּוֹלִיךְ אֶת הַתֶּבֶן. לְהֶסֵּק אוֹ לִבְהֵמָה:

וּמַתְחִילִין בַּעֲרֵמַת הַתֶּבֶן. וְאַף עַל פִּי שֶׁלֹּא זִמְּנָהּ מִבְּעוֹד יוֹם וְלֹא הָיָה רָגִיל לְהַסִּיק מִמֶּנָּה, דְּלֵית לֵיהּ לְהַאי תַּנָּא מֻקְצֶה:
אֲבָל לֹא בָעֵצִים [שֶׁבַּמֻּקְצֶה]. רְחָבָה שֶׁאֲחוֹרֵי הַבָּתִּים קְרוּיָה מֻקְצֶה, עַל שֵׁם שֶׁהִיא מֻקְצָה לְאָחוֹר וְאֵין נִכְנָסִים וְיוֹצְאִים לָהּ תָּדִיר. וְהָנָךְ עֵצִים דְּאָיְרֵי בְּהוּ הָכָא הֵן קוֹרוֹת גְּדוֹלוֹת שֶׁל אֲרָזִים הָעוֹמְדוֹת לְבִנְיָן, דְּמֻקְצֶה מֵחֲמַת חֶסְרוֹן כִּיס נִינְהוּ, שֶׁדְּמֵיהֶן יְקָרִים, וּבְהָא אֲפִלּוּ רַבִּי שִׁמְעוֹן דְּלֵית לֵיהּ מֻקְצֶה מוֹדֶה:

מַשִּׁילִין פֵּרוֹת דֶּרֶךְ אֲרֻבָּה. מִי שֶׁיֵּשׁ לוֹ פֵּרוֹת אוֹ תְּבוּאָה שָׁטוּחַ עַל גַּגּוֹ לְהִתְיַבֵּשׁ וְרָאָה גְּשָׁמִים מְמַשְׁמְשִׁים וּבָאִים, הִתִּירוּ לוֹ לִטְרֹחַ וּלְהַשְׁלִיךְ דֶּרֶךְ אֲרֻבָּה שֶׁבַּגַּג וְהֵם נוֹפְלִים לָאָרֶץ, דְּלֵיכָּא טִרְחָא יְתֵירָא. וְדַוְקָא אֲרֻבָּה, דְּכָל אֲרֻבָּה מִלְּמַעְלָה לְמַטָּה [בְּתִקְרַת הַגַּג], אֲבָל חַלּוֹן כְּגוֹן גַּג שֶׁמֻּקָּף מְחִצּוֹת וְחַלּוֹן בַּכֹּתֶל וְצָרִיךְ לְהַגְבִּיהּוֹ עַד הַחַלּוֹן וּלְהַשְׁלִיכוֹ, כֻּלֵּי הַאי לֹא שָׁרוּ רַבָּנָן לְמִטְרָח:
וּמְכַסִּין אֶת הַפֵּרוֹת. וְלֹא אָמְרִינַן טֹרַח שֶׁלֹּא לְצֹרֶךְ יוֹם טוֹב הוּא, דְּמִשּׁוּם הֶפְסֵד מָמוֹן שָׁרוּ לֵיהּ רַבָּנָן:
דֶּלֶף. גְּשָׁמִים הַנּוֹטְפִים מִן הַגַּג:
וְכֵן כַּדֵּי יַיִן וְכַדֵּי שָׁמֶן. מְכַסִּין אוֹתָן מִפְּנֵי הַדֶּלֶף:
תַּחַת הַדֶּלֶף. לְקַבֵּל אֶת הַמַּיִם שֶׁלֹּא יְטַנְּפוּ הַבַּיִת. וְאִם נִתְמַלֵּא הַכְּלִי שׁוֹפֵךְ וְשׁוֹנֶה וְאֵינוֹ נִמְנָע:
