(א) מֵי רְבִיעִית. וְנֻסַּח אַחֵר מֵרְבִיעִית, כְּלִי הַמַּחֲזִיק רְבִיעִית. רַשִׁ"י:
(ב) לַיָּדַיִם. דִּתְנַן (חֲגִיגָה פ"ב מ"ה) נוֹטְלִין לַיָּדַיִם לַחֻלִּין וְלַמַּעֲשֵׂר וְלַתְּרוּמָה, וּסְתָם יָדַיִם אֵינָן אֶלָּא שְׁנִיּוֹת וּפוֹסְלִין תְּרוּמָה וְלֹא חֻלִּין, אֲבָל מִשּׁוּם סְרַךְ תְּרוּמָה תִּקְּנוּ אַף לַחֻלִּין, כְּדֵי שֶׁיְּהֵא רָגִיל בִּנְטִילַת יָדַיִם וְיִטְּלֵן לַתְּרוּמָה. גְּמָרָא:
(ג) וְכֵן הָא דְּקָאָמַר לְאַרְבָּעָה, דְּמַשְׁמַע וְלֹא לַחֲמִשָּׁה, שֶׁהֲרֵי לֹא נָתְנוּ חֲכָמִים שִׁעוּר רַק לַמַּיִם שֶׁבַּכְּלִי, וְכִי אִיכָּא שִׁעוּר לֹא חָשׁוּ כַּמָּה יַגִּיעַ לְכָל אֶחָד רַק שֶׁיַּגִּיעַ עַד הַפֶּרֶק. הָרַ"שׁ:
(ד) וּלְמֵאָה. גֻּזְמָא הִיא, כְּלוֹמַר כָּל זְמַן שֶׁמַּסְפִּיקִין לִשְׁפֹּךְ כָּל אֶחָד שָׁלֹשׁ פְּעָמִים. טוּר:
(ה) רַבִּי יוֹסֵי כוּ'. פָּלֵיג אַף בְּרֵישָׁא דְּלִשְׁנַיִם. הָכִי מוּכָח בַּגְּמָרָא:
(ו) וּכְשֶׁנְּטָלָן בְּפָחוֹת מֵרְבִיעִית דַּאֲתוּ מִשְּׁיָרֵי טָהֳרָה. דְּאִלּוּ נְטָלָן מֵרְבִיעִית, הַמַּיִם עַצְמָן טְהוֹרִים, כְּתַנָּא קַמָּא דְּרֵישׁ פֶּרֶק ב', וְעַיֵּן שָׁם בְּהָרַ"ב:
(ז) דְּכֵיוָן שֶׁלֹּא טִהֲרוּ אֶת הַיָּדַיִם כָּל מַה שֶּׁמּוֹסִיף עֲלֵיהֶם הֲרֵי הוּא מוֹסִיף טֻמְאָה. הָרַ"שׁ:

(א) אַחַת. נִתְבָּאֵר בַּחֲגִיגָה כְּשֶׁנִּטְמְאָה יָדוֹ אַחַת, לֹא תִּטַּמֵּא הַשְּׁנִיָּה, וְכֵן כְּשֶׁנִּטְמְאוּ שְׁנֵיהֶם מְטַהֲרָן אַחַת אַחַת. וְכָל זֶה לַתְּרוּמָה וְלֹא לַקֹּדֶשׁ. הָרַמְבַּ"ם:
(ב) וְהוּא שֶׁאוֹתָהּ הַשְּׁטִיפָה מְרֻבָּה כִּשְׁתַּיִם. תּוֹסָפוֹת:
(ג) לְשִׁיטָתֵיהּ בְּרֵישׁ פֶּרֶק קַמָּא, דְּנוֹתְנִים לַיָּדַיִם, אַמַּיִם שְׁנִיִּים קָאֵי:
(ד) הַיְנוּ לַפֵּרוּשׁ הָרִאשׁוֹן. אֲבָל לְהָרַמְבַּ"ם, מְפֹרָשׁ בְּנֻסְחַת אֶרֶץ יִשְׂרָאֵל דְּקָאֵי אַמַּיִם שְׁנִיִּים, שֶׁאִם נָפַל כִּכָּר שֶׁל תְּרוּמָה בְּאֵלּוּ מַיִם שְׁנִיִּים, טָהוֹר לְתַנָּא קַמָּא כְּמוֹ בְּמַיִם שְׁנִיִּים דְּעָלְמָא. וְרַבִּי יוֹסֵי אוֹמֵר שֶׁמְּטַמְּאִין כִּכָּר שֶׁל תְּרוּמָה כְּמוֹ מַיִם רִאשׁוֹנִים שֶׁהֵם תְּחִלָּה, הוֹאִיל וּמַיִם רִאשׁוֹנִים הָיוּ שְׁטִיפָה אַחַת לִשְׁתֵּי הַיָּדַיִם:
