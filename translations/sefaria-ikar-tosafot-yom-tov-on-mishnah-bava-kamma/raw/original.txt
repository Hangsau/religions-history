(א) אַרְבָּעָה. מִנְיָנָא לְמִעוּטֵי דְּרַבִּי אוֹשַׁעְיָא וּדְרַבִּי חִיָּא [דְּמָנוּ נַמִּי לְנִזְקֵי גּוּפוֹ]. דְּתַנָּא דִּידָן בְּנִזְקֵי מָמוֹנוֹ קָמַיְרֵי, בְּנִזְקֵי גּוּפוֹ לֹא קָמַיְרֵי. גְּמָרָא:
(ב) לְהָנָךְ דִּכְתִיבֵי בִּקְרָא בְּהֶדְיָא. רַשִׁ"י:
(ג) נְזִיקִין. שֶׁעוֹשִׂין נֶזֶק. וּנְזָקִין שֵׁם הַהֶזֵּק. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ד) כִּיחוֹ רֹק הַפֶּה. נִיעוֹ לַחוּת הַחֹטֶם. רַשִׁ"י וְעַיֵּן עוֹד. תּוֹלָדָה דְּבוֹר לְקַמָּן מִשְׁנָה ג', וּפֶרֶק י'‏ דְּבָבָא מְצִיעָא מִשְׁנָה ד':
(ה) וְהָא דְּלֹא נָקַט שֵׁן, דַּהֲוָה אֲמֵינָא אֲפִלּוּ הֵיכָא דְּאֵין הֲנָאָה לְהֶזֵּקָה, וּלְפִיכָךְ נָקַט לִישְׁנָא דְּמַבְעֶה, דְּמַשְׁמַע לִישְׁנָא דְּגִלּוּיֵי כוּ'‏ דְּהַיְנוּ לַהֲנָאָתָהּ. שֶׁפְּעָמִים נִגְלִין פְּעָמִים נִכְסִין, דְּהַיְנוּ בִּשְׁעַת אֲכִילָה. נִמּוּקֵי יוֹסֵף:
(ו) וְהַהֶבְעֵר. יֵשׁ לְהַקְשׁוֹת אַמַּאי לֹא תָּנֵי אֵשׁ. וְיֵשׁ לוֹמַר דְּאֵשׁ יֵשׁ לְהָבִין מִסְתָּמָא שֶׁהוּא בְּעִנְיָן שֶׁדּוֹלֵק הָאֵשׁ בְּשֶׁל חֲבֵרוֹ, וּקְרָא קָאָמַר כִּי תֵּצֵא אֵשׁ, אֲפִלּוּ מֵעַצְמָהּ, לְכָךְ קָאָמַר הַהֶבְעֵר, וְהוּא לִישְׁנָא דִּקְרָא לֹא תְבַעֲרוּ וְכוּ', וּמַיְרֵי מִסְתָּמָא דִּבְשֶׁלּוֹ הוּא מַדְלִיק, הָכָא נַמִּי דִּכְוָתֵיהּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ז) לֹא הֲרֵי כוּ'. פֵּרוּשׁ אֵין קֻלָּתוֹ שֶׁל שׁוֹר כְּקֻלָּתוֹ שֶׁל מַבְעֶה, דַּהֲוָה אֲמֵינָא רֶגֶל כוּ', כְּמוֹ שֶׁכָּתַב הָרַ"ב, וּלְפִיכָךְ אִי כָּתַב רַחְמָנָא שׁוֹר לֹא אָתֵי מַבְעֶה מִנֵּיהּ, שֶׁהוּא קִיל מִנֵּיהּ. וְהָא דְּלֹא שָׁנְאָה מִשְׁנָתֵנוּ הַקַּל וָחֹמֶר כְּמוֹ שֶׁאָמַר לֹא זֶה וָזֶה [שֶׁיֵּשׁ בָּהֵם רוּחַ חַיִּים], מִשּׁוּם דְּחַד מֵחַד קַל לִמְצֹא חֹמֶר בְּאֶחָד מִבַּחֲבֵרוֹ. תּוֹסָפוֹת. וְעַיֵּן עוֹד:
(ח) הָאֵשׁ. וְהָא דְּלֹא הֲדַר וְתָנֵי לֹא הֲרֵי הָאֵשׁ כַּהֲרֵי שְׁנֵיהֶם כִּדְקָא הֲדַר בָּרֵישָׁא, מִשּׁוּם דְּטִרְחָא לֵיהּ מִלְּתָא. נִמּוּקֵי יוֹסֵף. סְבִירָא לֵיהּ דְּיֵשׁ בָּאֵשׁ מַה שֶּׁאֵין בִּשְׁנֵיהֶם לְפִי שֶׁכֹּחַ אַחֵר מְעֹרָב בּוֹ וְאֵין הוֹלֵךְ לְדַעְתּוֹ כְּמוֹ שׁוֹרוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ט) אֲבָל מִבּוֹר לְחוּד לֹא נָפֵיק, דְּאִיכָּא לְמִפְרַךְ מַה לְּבוֹר שֶׁתְּחִלַּת עֲשִׂיָּתוֹ לְנֶזֶק כוּ'. וְהַתַּנָּא הֶאֱרִיךְ, לְהַגְדִּיל תּוֹרָה וְיַאֲדִיר. תּוֹסָפוֹת:
(י) דִּכְתִיב וּבִעֵר בִּשְׂדֵה אַחֵר. רַשִׁ"י:
(יא) כְּדִי נָסֵיב לַהּ, דְּהָא בּוֹר אִצְטְרִיכָא לֵיהּ לְמִכְתַּב, דְּלֹא אָתֵי מִכּוּלְּהוּ, כְּדִתְנַן בְּמַתְנִיתִין. רַשִׁ"י וְתּוֹסָפוֹת:
(יב) וּלְרַבִּי יְהוּדָה דְּמִשְׁנָה ה'‏ פֶּרֶק ו'‏ דִּמְחַיֵּב טָמוּן, יֵשׁ לוֹמַר כְּמוֹ שֶׁכָּתְבוּ הַתּוֹסָפוֹת דַּהֲוָה מָצֵי לְמֵימַר דְּכֻלְּהוּ אִצְטְרִיךְ לְמִכְתַּב, דְּאִי הֲוָה נָפְקָא מִבּוֹר וּמְחַד מֵהָנָךְ הֲוָה פָּטְרֵי בְּהוּ כֵּלִים. וּלְרַבִּי יְהוּדָה אִין הָכִי נַמִּי דְּאִצְטְרִיךְ מִשּׁוּם הָכִי:
(יג) אַף עַל גַּב דְּלֹא דָּמֵי לְחַד מֵהָנֵי אָבוֹת, אִי יְכוֹלִין לַאֲתוֹיֵי מִתַּרְתֵּי אוֹ מִתְּלַת אוֹ מִטְּפֵי בְּהַצַּד הַשָּׁוֶה לְחִיּוּבָא, מְחַיְּבִינַן לֵיהּ. הָרַ"ן:
(יד) וּשְׁמִירָתָן כוּ'. שֶׁנִּתְחַיַּבְתָּ בִּשְׁמִירָתָן מֵחֲמַת שֶׁהֵם מָמוֹנְךָ, אוֹ שֶׁנִּמְסְרוּ לְךָ לְשָׁמְרָן, אוֹ שֶׁעָשִׂיתָ בָּהֶם מַעֲשֶׂה שֶׁתִּתְחַיֵּב בִּשְׁמִירָתָן כוּ'. לַאֲפוֹקֵי הֵיכָא דְּמַסְרִינְהוּ לַאֲחֵרִים לְנָטְרִינְהוּ כְּדַחֲזִי, וּפָשַׁע בָּהֶן הַשּׁוֹמֵר, דְּהַשֵּׁנִי חַיָּב וְהָרִאשׁוֹן פָּטוּר. וּלְאַפּוֹקֵי נַמִּי הֵיכָא דְּנָטְרִינְהוּ כְּאוֹרְחַיְהוּ וְאַזִּיקוּ, דְּפָטוּר. נִמּוּקֵי יוֹסֵף:
(טו) חָב. הַאי תָּנָא יְרוּשַׁלְמִי הוּא, דְּתָנֵי לִישָׁנָא קַלִּילָא, וְלֹא תָּנֵי חַיָּב. גְּמָרָא:
(טז) תַּשְׁלוּמֵי. לְהָכִי נָקַט לְשׁוֹן הַשְׁלָמָה, דִּפְחַת נְבֵלָה מִשְּׁעַת מִיתָה עַד שְׁעַת הַעֲמָדָה בַּדִּין, דְּנִזָּק הָוֵי. וְהַמּוֹתָר מְשַׁלֵּם לוֹ מַזִּיק. בַּגְּמָרָא דַּף י. וְהוּא הַדִּין דִּשְׁבָחָהּ דְּבֵין מִיתָה לַהַעֲמָדָה בַּדִּין, דְּנִזָּק הָוֵי. נִמּוּקֵי יוֹסֵף. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(יז) דְּהָא מֵיטַב שָׂדֵהוּ דְּהַאִיךְ דְּקָא מְשַׁלֵּם מַשְׁמַע. גְּמָרָא:
(יח) וְיָלְפִינַן מִבּוֹר דִּכְתִיב בֵּיהּ יָשִׁיב, לְרַבּוֹת שָׁוֵה כֶּסֶף. נִמּוּקֵי יוֹסֵף:

=== 2 | chapter 2 ===
(א) וְאַף עַל גַּב דְּאָמְרָן לַהּ בַּפֶּרֶק דִּלְעֵיל, אַיְדֵי דְּבָעֵי לְמִתְנֵי תּוֹלָדָה דְּהַיְנוּ הַבְּהֵמָה מוּעֶדֶת [וְכוּ'], תָּנָא אָב בְּשֶׁשָּׁבְרָה בְּרַגְלֶיהָ מַמָּשׁ. נִמּוּקֵי יוֹסֵף:
(ב) מְבַעֶטֶת. אַגַּב דְּשָׁוֶה לִצְרוֹרוֹת נָקַט לַהּ הָכָא. תּוֹסָפוֹת:
(ג) וְתוּ נָפְקָא מִנַּהּ דַּהֲוֵי תּוֹלָדָה דְּרֶגֶל, הַיְנוּ לְשַׁלֵּם מִן הָעֲלִיָּה, כֵּיוָן דְּמוּעָד מִתְּחִלָּתוֹ, אֶלָּא דַּהֲלָכָה אַפְּקֵיהּ לַחֲצִי נֶזֶק. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ד) דָּרְסָה כוּ'. וְאַף עַל גַּב דִּכְבָר שָׁמַעְנוּ, מִכָּל מָקוֹם אִצְטְרִיךְ לְאַשְׁמְעִינַן, דְּסָלְקָא דַּעְתָּךְ אֲמֵינָא מְשֻׁנָּה הִיא בְּהַךְ דְּרִיסָה הוֹאִיל וְדָרְסָה בְּחֹזֶק כָּל כָּךְ שֶׁנִּתַּז הַשֶּׁבֶר עַל כְּלִי אַחֵר וּשְׁבָרוֹ, מִסְתָּמָא לְהַזִּיק נִתְכַּוְּנָה, קָא מַשְׁמַע לָן דְּלֹא. אִי נַמִּי קָא מַשְׁמַע לָן דְּמַשְׁכַּחַת לֵיהּ נֶזֶק וַחֲצִי נֶזֶק בְּחַד מַעֲשֶׂה בַּהֲדָדֵי כְּהַאי גַּוְנָא. נִמּוּקֵי יוֹסֵף:
(ה) לְהַלֵּךְ. תָּנָא רֶגֶל וְהוּא הַדִּין שֵׁן. נִמּוּקֵי יוֹסֵף. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

=== 3 | chapter 3 ===
(א) מִשּׁוּם דְּאָדָם בַּעַל מַחְשָׁבוֹת הוּא וְאַגַּב טִרְדָא דְּלִבֵּיהּ לָאו אוֹרְחָא לְעַיּוֹנֵי וּמֵיזָל. נִמּוּקֵי יוֹסֵף. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ב) הֶחָבִית. פָּתַח בְּכַד וְסִיֵּם בֶּחָבִית, לוֹמַר לְךָ הַיְנוּ כַּד הַיְנוּ חָבִית. לְמַאי נָפְקָא מִנַּהּ, לְמִקָּח וּמִמְכָּר. עַיֵּן בַּגְּמָרָא:
(ג) נִשְׁבְּרָה. שֶׁלֹּא הִנִּיחַ אֶלָּא נִתְקַל וְנִשְׁבְּרָה. וְהַיְנוּ דְּהָרַ"ב קָרֵי לֵיהּ נִתְקַל לְמַזִּיק בְּכַדּוֹ שֶׁנִּשְׁבַּר:

(ד) וַאֲפִלּוּ כוּ'. דִּכְוִיָּה, בֵּין אִית בָּהּ חַבּוּרָה בֵּין אֵין בָּהּ חַבּוּרָה מַשְׁמַע. וְכִי תֵּימַא חַבּוּרָה לָמָּה לִי, לְדָמֵי יְתֵרֵי. גְּמָרָא. וּפֵרְשׁוּ הַתּוֹסָפוֹת שֶׁמְּשַׁלֵּם יוֹתֵר לְפִי מַה שֶּׁמִּתְגַּנֶּה מֵחֲמַת הָרֹשֶׁם. וַאֲנִי תָּמֵהַּ עַל זֶה, דְּזֶה בִּכְלַל בֹּשֶׁת הוּא. וְהַנִּמּוּקֵי יוֹסֵף כָּתַב בְּשֵׁם הָרְמָ"ה דִּלְדָמֵי יְתֵרֵי הָכִי פֵּרוּשׁוֹ, שֶׁאִם חָבַל זֶה בָּזֶה וְשֶׁל זֶה מְרֻבֶּה מִשֶּׁל זֶה, דְּשָׁמִין, וְלֹא אָמְרִינַן יֵצֵא זֶה בָּזֶה, וְהָכִי מַשְׁמַע קְרָא, חַבּוּרָה תַּחַת חַבּוּרָה, וְלֹא כְּוִיָּה תַּחַת חַבּוּרָה, וְאַף עַל גַּב דְּתַרְוַיְהוּ מִקְרֵי כְּוִיָּה כוּ'. וְהוּא הַדִּין לִשְׁתֵּי חַבּוּרוֹת שֶׁשָּׁמִין וּמְשַׁלֵּם הַמּוֹתָר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) לִטֹּל. אֵין שָׁמִין כַּמָּה אָדָם רוֹצֶה לִטֹּל וּלְהִצְטַעֵר, דְּכָל אֲשֶׁר לָאָדָם יִתֵּן בְּעַד נַפְשׁוֹ, וְאֵין עֶרֶךְ לְשׁוּמָא זוֹ. אֶלָּא שָׁמִין מִי שֶׁנִּתְחַיֵּב לַמַּלְכוּת לִכְוֹּתוֹ בְּשַׁפּוּד עַל צִפָּרְנוֹ כַּמָּה הָיָה נוֹתֵן שֶׁלֹּא יֵעָשֶׂה לוֹ צַעַר זֶה. וְשׁוּמָא זוֹ קַלָּה לְהוֹצִיא מָמוֹן לְהִנָּצֵל מִצַּעַר מִלִּקַּח מָמוֹן וּלְקַבֵּל צַעַר. וְהַיְנוּ כְּשׁוּמָא שֶׁכָּתַב הָרַ"ב בְּצַעַר דְּבִמְקוֹם נֶזֶק כוּ'. וּבַגְּמָרָא פָּרֵיךְ, הַאי לִטֹּל, לִתֵּן מִבָּעֵי לֵיהּ. וּמְשַׁנִּינַן לִטֹּל זֶה מִזֶּה מַה שֶּׁנָּתַן זֶה:
(ו) דְּמַיְרֵי בִּסְתָם בְּנֵי אָדָם שֶׁאֵינָם בְּנֵי אֻמָּנוּת, אֲבָל אִם הוּא בֶּן אֻמָּנוּת וְיָכוֹל לַעֲסֹק בִּמְלַאכְתּוֹ אַחַר חָלְיוֹ מְשַׁעֲרִין כְּפִי הַמְּלָאכָה שֶׁיָּכוֹל לַעֲשׂוֹת אַחַר שֶׁיִּתְרַפֵּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ז) מְסַיֵּם בַּגְּמָרָא שֶׁאֵינָן חוֹזְרִין. כְּלוֹמַר וְהָוֵי כְּעֵין מִיתָה, שֶׁאוֹתוֹ אֵבָר מֵת:
(ח) הַיְנוּ אִם עָלוּ בּוֹ צְמָחִים מֵחֲמַת מַכָּה דִּתְנַן בְּמַתְנִיתִין. וְיָלְפִינַן לַהּ מִכֶּפֶל לָשׁוֹן, וְרַפֹּא יְרַפֵּא:
(ט) הַיְנוּ נַמִּי שֶׁלֹּא מֵחֲמַת מַכָּה דִּתְנַן. וַהֲוָה אֲמֵינָא דְּעָבַר עַל דִּבְרֵי רוֹפֵא לָאו שֶׁלֹּא מֵחֲמַת מַכָּה הִיא, דְּהָא מִכָּל מָקוֹם עַל יְדֵי הַמַּכָּה בָּא לוֹ זֶה, שֶׁאִלּוּ לֹא הֻכָּה לֹא הָיָה צָרִיךְ לְהִזָּהֵר בְּדִבְרֵי הָרוֹפֵא, קָא מַשְׁמַע לָן דְּלֹא. וְיָלֵיף לֵיהּ מִדִּכְתִיב רַק:
(י) וּבַסִּפְרִי יָלֵיף גְּזֵרָה שָׁוָה דְּתָחוֹס תָּחוֹס. וּפֵרֵשׁ רַשִׁ"י נֶאֱמַר כָּאן לֹא תָחוֹס. וְנֶאֱמַר בְּעֵדִים זוֹמְמִין לֹא תָחוֹס, מַה לְּהַלָּן מָמוֹן אַף כָּאן מָמוֹן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(יא) בַּגְּמָרָא, לֹא אָמְרוּ אֶלָּא הוֹדָאוֹת וְהַלְוָאוֹת. וּפֵרֵשׁ רַשִׁ"י, הוֹדָאוֹת, הַבָּא לָדוּן בְּעֵדֵי הוֹדָאָה שֶׁאוֹמְרִים בְּפָנֵינוּ הוֹדָה לוֹ. וְהַלְוָאוֹת, שֶׁבָּא לָדוּן בְּעֵדֵי הַלְוָאָה שֶׁאוֹמְרִים בְּפָנֵינוּ הִלְוָה לוֹ. עַד כָּאן. וּכְפִירוֹת, הַיְנוּ מוֹדֶה בְּמִקְצָת וְכוֹפֵר הַכֹּל. וְהַקְנָאוֹת, הַיְנוּ שֶׁהִקְנָה לוֹ בְּמַתָּנָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(יב) דְּאִי בְּקֶרֶן, קָיְמָא לָן פַּלְגָּא נִזְקָא קְנָסָא, וְאֵין גּוֹבִין קְנָסָא בְּחוּץ לָאָרֶץ. וְכֵיוָן דְּמִנְּגִיחוֹת דְּתַמּוּת לֹא מַיְתִינַן לֵיהּ לְבֵי דִּינָא, הֵיכִי מְיַעֵד לֵיהּ. מַאי אָמַרְתְּ, דְּאִיַּעוּד הָתָם וְאַיְתֵיהּ לְהָכָא, דָּבָר שֶׁאֵינוֹ מָצוּי הוּא, כִּדְאִיתָא בַּגְּמָרָא:
(יג) הֶעָרֹם. אִם רָקַק בּוֹ אוֹ סְטָרוֹ. דְּכָךְ מִתְבַּיֵּשׁ בִּרְקִיקָה וּסְטִירָה כְּשֶׁהוּא עָרֹם כְּמוֹ כְּשֶׁהוּא לָבוּשׁ. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(יד) הַיָּשֵׁן. וּכְשֶׁהֵקִיץ וְהִרְגִּישׁ בָּזֶה שֶׁבִּיְּשׁוֹ כוּ'. אֲבָל אִי לֹא הִרְגִּישׁ אַחַר שֶׁהֵקִיץ, שֶׁנִּסְתַּלְּקָה הַבּוּשָׁה קֹדֶם שֶׁהֵקִיץ, הוּא בִּכְלַל בַּעְיָא בַּגְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(טו) וְשָׁלְחָה כוּ'. מִוְּשָׁלְחָה יָדָהּ דִּיֵּק. וְהָרַמְבַּ"ם לֹא הֶעְתִּיק וְהֶחֱזִיקָה בִּמְבֻשָׁיו. וְתַנָּא שְׁנָאוֹ לוֹמַר דִּקְרָא בְּבֹשֶׁת אַיְרֵי:

(א) עֵצִים כוּ'. מַסְּקִינַן בַּגְּמָרָא דְּדַוְקָא בְּשִׁנּוּי דְּלֹא הֲדַר לִבְרִיָּתוֹ כְּבַתְּחִלָּה. וּכְגוֹן שֶׁגָּזַל עֵצִים שֶׁאֵינָם מְשֻׁפִּין וַעֲשָׂאָן כֵּלִים בּוּכָאנֵי, דְּהַיְנוּ שִׁפָּן, דְּלֹא הֲדַר לִבְרִיָּתוֹ, וְנִשְׁתַּנָּה שְׁמוֹ שֶׁנִּקְרָא בּוּכָאנֵי. וְהַיְנוּ דִּתְנַן וַעֲשָׂאָן כֵּלִים:
(ב) צֶמֶר כוּ'. הָכִי נַמִּי שֶׁגָּזַל צֶמֶר שֶׁאֵינוֹ טָווּי וְעָשָׂה מֵהֶן לְבָדִים, שֶׁאֵינוֹ חוֹזֵר לִבְרִיָּתוֹ לִהְיוֹת כָּל נִימָא וְנִימָא בִּפְנֵי עַצְמָהּ, וְנִשְׁתַּנָּה שְׁמוֹ שֶׁנִּקְרָא לֶבֶד. וְהַיְנוּ דִּתְנַן וַעֲשָׂאָן בְּגָדִים:
(ג) כִּדְיָלֵיף לֵיהּ בַּגְּמָרָא דַּף ס"ז מִקְּרָא [עַיֵּן לְקַמָּן בְּאוֹת ל"ו]. רַשִׁ"י:
(ה) רְצוֹנוֹ לוֹמַר דְּדַוְקָא הַוָּלָד וְהַגִּזָּה קוֹנֶה בְּשִׁנּוּי וּמְשַׁלֵּם מַה שֶּׁהָיוּ דְּמֵיהֶם, וְהָעֹדֶף שֶׁשָּׁוֶה עַכְשָׁיו יוֹתֵר הוּא שֶׁלּוֹ. אֲבָל פָּרָה עַצְמָהּ חוֹזֶרֶת בְּעֵינָהּ. דְּלֵידָה וְגִזָּה לָאו שִׁנּוּי הוּא בְּגוּף הַפָּרָה כוּ'. נִמּוּקֵי יוֹסֵף. אֲבָל לְהָרֹא"שׁ, דְּמֵי פָּרָה מַמָּשׁ, דַּהֲוֵי שִׁנּוּי אַף לַפָּרָה עַצְמָהּ וּקְנָאָהּ בּוֹ:
(ד) הָעוֹמֶדֶת כוּ'. כְּלוֹמַר לַאֲפוֹקֵי שֶׁיָּשׁוּמוּ כְּאִלּוּ יָלְדָה, דְּהָא עַל יְדֵי הַלֵּידָה קְנָאָהּ. וְזֶה לְשׁוֹן הַתּוֹסָפוֹת, וְכָל שֶׁבַח שֶׁמִּשְּׁעַת גְּזֵלָה עַד שֶׁתֵּלֵד, לַגַּזְלָן. עַד כָּאן. וְטַעֲמָא, דִּלְעוֹלָם מְשַׁלֵּם כִּשְׁעַת הַגְּזֵלָה, כִּדְכָיֵל בַּסֵּיפָא, וְקָאֵי אַכּוּלְּהוּ בָּבֵי דְּמַתְנִיתִין, וְלֹא אָזְלִינַן בָּתַר שְׁעַת קְנִיָּה, אֶלָּא מִשֶּׁקְּנָאָהּ חָזְרָה הַשּׁוּמָא לִשְׁעַת הַגְּזֵלָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ו) וְיָלְדָה. הוּא הַדִּין אֲפִלּוּ לֹא יָלְדָה, וְאַיְדֵי דְּנָסֵיב רֵישָׁא וְיָלְדָה, נָסֵיב נַמִּי סֵיפָא וְיָלְדָה. גְּמָרָא. וְאַף עַל גַּב דִּבְרֵישָׁא נַמִּי מְשַׁלֵּם כִּשְׁעַת הַגְּזֵלָה, סֵיפָא הָא קָא מַשְׁמַע לָן דְּעִבּוּר וּטְעִינָה הָוֵי שִׁנּוּי כְּמוֹ לֵידָה. וְנִרְאֶה לִי דְּהָכִי נַמִּי דִּינָא דְּמַחֲזִיר הַפָּרָה לְפֵרוּשׁ הַנִּמּוּקֵי יוֹסֵף וּכְדִלְעֵיל. דְּמַאי שְׁנָא שִׁנּוּי הַמִּלּוּי אוֹ שִׁנּוּי הָרֵקוּת. וְקַשְׁיָא לִי, דְּלִתְנֵי נַמִּי בָּרֵישָׁא מְשַׁלֵּם כִּשְׁעַת הַגְּזֵלָה. וְצָרִיךְ עִיּוּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ז) וְלֹא אָמְרִינַן דְּעַל כָּל פָּנִים יְשַׁלֵּם אַרְבָּעָה וַחֲמִשָּׁה כִּשְׁעַת הַגְּזֵלָה, לְפִי דְּשֶׁלּוֹ הוּא טוֹבֵחַ כוּ'. וּמִמֵּילָא דְּמִכָּל שֶׁכֵּן שֶׁאֵינוֹ מְשַׁלֵּם הַכֶּפֶל אֶלָּא כִּשְׁעַת הַגְּזֵלָה:

(א) דִּרְשׁוּת יוֹרֵשׁ לָאו כִּרְשׁוּת לוֹקֵחַ דָּמֵי, וְיֵאוּשׁ כְּדִי לֹא קָנֵי. רַשִׁ"י. וְהָא דְּדִיֵּק הָרַ"ב לִכְתֹּב וְהִיא בְּעֵינָהּ, מִשּׁוּם וְחַיָּבִים לְהַחֲזִיר דְּנָקַט, כְּלוֹמַר דְּחַיָּבִים לְהַחֲזִיר הַגְּזֵלָה עַצְמָהּ. אֲבָל כְּשֶׁאֵינָהּ בְּעֵינָהּ אֶלָּא נִשְׁתַּנֵּית בְּשִׁנּוּי דְּאָמְרָן בְּרֵישׁ פִּרְקִין דִּלְעֵיל, אֵין חַיָּבִים אֶלָּא דְּמֵי הַגְּזֵלָה, וַהֲרֵי דִּין הַבָּנִים כְּדִין אֲבִיהֶם:

(ה) דְּאִלּוּ סְלָעִים בְּסֶלַע דְּלֹא מִתְהַנֵּי בַּחִלּוּף לֵיכָּא אִסּוּרָא. וְהָא דְּפֵרֵשׁ דְּאִם חַיָּב אָדָם פְּרוּטוֹת כוּ'‏ דְּשָׁרֵי מִשּׁוּם שֶׁהוּא כְּמַצִּיל מִיָּדוֹ, הַיְנוּ לוֹמַר הֵיכָא דְּמִתְכַּוֵּן לָתֵת סֶלַע כְּדֵי לְקַבֵּל פְּרוּטוֹת, דְּמִתְהַנֵּי. שִׁלְטֵי גִּבּוֹרִים:
