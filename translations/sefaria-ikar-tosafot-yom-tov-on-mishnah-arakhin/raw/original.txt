(א) וְנִרְאֶה שֶׁצָּרִיךְ לוֹמַר בֶּן שְׁתֵּים עֶשְׂרֵה וְיוֹם אֶחָד. וְכֵן הוּא בַּתּוֹסָפוֹת. וּכְדִתְנַן בְּמִשְׁנָה ו'‏ פֶּרֶק ה'‏ דְּנִדָּה. וְהָתָם בַּגְּמָרָא דָרְשִׁינַן לַהּ בְּמֻפְלָא הַסָּמוּךְ לְאִישׁ:
(ב) נְפָשׁוֹת כָּל דְּהוּ. גְּמָרָא:
(ג) וְתָנֵי וַהֲדַר מְפָרֵשׁ. גְּמָרָא. וְהוּא הַדִּין נַמִּי דְּמָצֵי לְמִנְקַט טֻמְטוּם כוּ'‏ חֵרֵשׁ שׁוֹטֶה וְקָטָן וְנָכְרִי, אֶלָּא חֲדָא מִתַּרְתֵּי וּתְלַת נָקַט כוּ'. רַשִׁ"י. וְעַיֵּן עוֹד:
(ד) מַעֲרִיכִין כוּ'. וְהָעֲרָכִין כֻּלָּן וְהַדָּמִים, הַכֹּל סְתָמָן לְבֶדֶק הַבַּיִת, וְיִפְּלוּ הַכֹּל לַלִּשְׁכָּה שֶׁהָיְתָה בַמִּקְדָּשׁ מוּכֶנֶת לְקָדְשֵׁי בֶּדֶק הַבַּיִת. הָרַמְבַּ"ם:
(ה) הַאי קְרָא בְּמַעֲרִיךְ כְּתִיב, וּלְפִיכָךְ הָיָה נִרְאֶה לִי לְהַגִּיהַּ לְאַשְׁמְעִינַן דְּמַעֲרִיכִין. וְיֵשׁ לוֹמַר דְּלִישְׁנָא דְמִשְׁנָה ג'‏ דִּלְקַמָּן נָקַט. וְעַיֵּן שָׁם:
(ו) אִי נַמִּי כְגוֹן שֶׁנָּתְנוּ לָהֶם עַל מְנָת שֶׁאֵין לָרַב אוֹ לַבַּעַל רְשׁוּת בָּהֶן. תּוֹסָפוֹת:
(ז) וַעֲבָדִים. אַף עַל גַּב דְּגַבֵּי מִצְוֹת יֵשׁ לוֹ דִּין אִשָּׁה, לְפִי עֶרֶךְ זָכָר הוּא נֶעֱרָךְ, דְּלֹא אִשְׁתְּמִיט תַּנָּא דְלֵימָא אִם הָיָה נֶעֱרָךְ כְּאִשָּׁה. תּוֹסָפוֹת:

=== 2 | chapter 2 ===
(א) וְכֵן לְקַמָּן כָּתַב עָנִי בֶּן עֶשְׂרִים שֶׁהֶעֱרִיךְ עַצְמוֹ כוּ'. וְלָאו דַּוְקָא, דְּהוּא הַדִּין הֶעֱרִיךְ לְאַחֵר, אֶלָּא חֲדָא מִתַּרְתֵּי נָקַט:
(ב) פָּחוֹת כוּ'. לָאו דַּוְקָא פָּחוֹת מִסֶּלַע, אֶלָּא אֲפִלּוּ נָתַן כָּל הָעֵרֶךְ חוּץ מִסֶּלַע אֶחָד מְגַלְגְּלִין עָלָיו אֶת הַכֹּל. תּוֹסָפוֹת. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ג) אֵין כוּ'. הָא תוּ לָמָּה לִי, הָא קָא מַשְׁמַע לָן פָּחוּת מִסֶּלַע הוּא דְלֵיכָּא, הָא יָתֵר עַל סֶלַע אִיכָּא [אַף עַל גַּב דְּלֵיכָּא חֲמִשִּׁים סְלָעִים, דְּאִם הָיוּ בְיָדוֹ חֲמִשָּׁה סְלָעִים נוֹתֵן כֻּלָּם, דְּהַיְנוּ יָתֵר עַל סֶלַע וּפָחוּת מֵחֲמִשִּׁים]. יָתֵר עַל חֲמִשִּׁים הוּא דְלֵיכָּא, הָא פָּחוּת מֵחֲמִשִּׁים אִיכָּא [כְּדַאֲמַרַן]. וּסְתָמָא כְּרַבָּנָן. גְּמָרָא:
(ד) [וּמְסַיֵּם] וּצְרִיכָה שִׁבְעָה נְקִיִּים וְקָרְבָּן, פֵּרוּשׁ עוֹלָה בָּאָה עַל תְּנַאי, וְהַחַטָּאת אֵינָהּ נֶאֱכֶלֶת. וְזֶה לְשׁוֹן הָרַמְבַּ"ם, וְאִם אָמְרָה שְׁנֵי יָמִים רָאִיתִי, נֹאמַר כָּאן פִּתְחָהּ שִׁבְעָה עָשָׂר, לְפִי שֶׁאָנוּ אוֹמְרִים אֶפְשָׁר שֶׁאֵלּוּ הַשְּׁנֵי יָמִים יוֹם אֶחָד מֵהֶן קֹדֶם עֵת נִדָּתָהּ וְיוֹם אֶחָד מִתְּחִלַּת עֵת נִדָּתָהּ וְנִשְׁאֲרוּ מִן הַתְּקוּפָה [שֶׁיֵּשׁ בֵּין תְּחִלַּת נִדָּה לִתְחִלַּת נִדָּה] שִׁבְעָה עָשָׂר יָמִים. וְכֵן אִם אָמְרָה שְׁלֹשָׁה יָמִים רָאִיתִי, אֶפְשָׁר שֶׁשְּׁנֵי יָמִים קֹדֶם כוּ'‏ וְיוֹם אֶחָד מִתְּחִלַּת כוּ'‏ נִשְׁאֲרוּ מִן הַתְּקוּפָה שִׁבְעָה עָשָׂר יוֹם, הֲרֵי הִיא גַּם כֵּן פִּתְחָהּ שִׁבְעָה עָשָׂר, אֶלָּא שֶׁהִיא סְפֵק זָבָה, לְפִי שֶׁאֶפְשָׁר שֶׁשְּׁלֹשָׁה יָמִים אֵלּוּ שֶׁרָאֲתָה עַל הַסֵּדֶר בְּלֹא עֵת נִדָּתָהּ הֵם. וְאִם אָמְרָה אַרְבָּעָה יָמִים, גַּם כֵּן הִיא סָפֵק כוּ', לְפִי שֶׁאֶפְשָׁר שֶׁיִּהְיוּ אֵלּוּ הָאַרְבָּעָה יָמִים שְׁנַיִם קֹדֶם נִדָּתָהּ וּשְׁנַיִם מִימֵי נִדָּתָהּ וְלֹא תְהֵא זָבָה. וּכְשֶׁאָנוּ מְשַׁעֲרִין הַסָּפֵק הַזֶּה, יְהֵא פִתְחָה שִׁשָּׁה עָשָׂר יוֹם, לְפִי שֶׁאִם נֹאמַר שֶׁשְּׁנֵי יָמִים מֵאֵלּוּ הָאַרְבָּעָה, מִתְּחִלַּת עֵת נִדָּתָהּ, יִשָּׁאֲרוּ עִם הַאֶחָד עָשָׂר יָמִים שֶׁהִיא תְקוּפָה קְטַנָּה חֲמִשָּׁה יָמִים שֶׁהֵם עִם הָאַחַד עָשָׂר שִׁשָּׁה עָשָׂר יוֹם. וְאִם אָמְרָה חֲמִשָּׁה יָמִים רָאִיתִי נֹאמַר שֶׁשְּׁלֹשָׁה מֵאֵלּוּ הַחֲמִשָּׁה מִתְּחִלַּת עֵת נִדָּתָהּ, וְיִשָּׁאֲרוּ מִן הַתְּקוּפָה חֲמִשָּׁה עָשָׂר כוּ'. וְכֵן אִם אָמְרָה הָאַחַד עָשָׂר יוֹם הִיא גַּם כֵּן סָפֵק זָבָה, לְפִי שֶׁנֹּאמַר, שְׁנֵי יָמִים הֵן מִסּוֹף הָאַחַד עָשָׂר, וְשִׁבְעָה יְמֵי נִדָּה, וּשְׁנֵי יָמִים שְׁאֵרִית הָאַחַד עָשָׂר שֶׁרָאֲתָה, וּפִתְחָהּ תִּשְׁעָה. וְאִם אָמְרָה שְׁנֵים עָשָׂר יָמִים רָאִיתִי, פִּתְחָהּ שְׁמֹנָה, וְהִיא זָבָה וַדָּאִית. לְפִי שֶׁאִם תֹּאמַר שֶׁשְּׁנֵי יָמִים מִזְּמַן רְאִיַּת הַדָּם קֹדֶם עֵת נִדָּתָהּ, וְשִׁבְעָה שֶׁל נִדָּה, וּשְׁלֹשָׁה אַחַר הַנִּדָּה, הֲרֵי הִיא זָבָה, לְפִי שֶׁכְּבָר רָאֲתָה שְׁלֹשָׁה יָמִים רְצוּפִין אַחַר זְמַן הַנִּדּוּת. וְאִם נֹאמַר שֶׁאֶפְשָׁר שֶׁיְּהֵא מִכְּלָלָם שְׁלֹשָׁה יָמִים קֹדֶם עֵת נִדָּתָהּ, שִׁבְעָה שֶׁל נִדָּה, וּשְׁנַיִם אַחַר הַנִּדָּה, הֲרֵי הִיא גַּם כֵּן זָבָה, לְפִי שֶׁכְּבָר קָדְמָה זִיבוּת לְנִדּוּת. וְכֵן אִם רָאֲתָה שְׁלֹשָׁה עָשָׂר יוֹם, פִּתְחָהּ לְפִי הַהֶקֵּשׁ הַזֶּה בְּעַצְמוֹ כוּ'. וְזֶה שֶׁאָמַרְנוּ פִתְחָהּ שִׁבְעָה עָשָׂר אוֹ פָּחוֹת, אֵין הַכַּוָּנָה שֶׁתְּהֵא כָל הַזְּמַן הַזֶּה אֲסוּרָה עַל בַּעֲלָהּ, אֶלָּא הָעִנְיָן שֶׁזָּכַרְנוּ שֶׁאַחַר אוֹתוֹ הַזְּמַן תִּקְבַּע לָהּ וֶסֶת לְנִדָּתָהּ כוּ'. וְאֵין הֶפְרֵשׁ בֵּין שְׁלֹשָׁה עָשָׂר יוֹם וְיוֹתֵר מִזֶּה, שֶׁאֵין לָהּ פֶּתַח פָּחוּת מִשִּׁבְעָה, לְפִי שֶׁהִיא זָבָה וַדָּאִית:

(א) שֶׁבְּיִשְׂרָאֵל. אֲפִלּוּ תֵימָא רַבִּי מֵאִיר דְּמִשְׁנָה ב'‏ פֶּרֶק קַמָּא דְאָמַר הַנָּכְרִי נֶעֱרָךְ. וְהָא דְאִתְנֵי שֶׁבְּיִשְׂרָאֵל, מִלְּתָא אַגַּב אוֹרְחָא קָא מַשְׁמַע לָן כוּ'‏ דְּאָסוּר לָאָדָם לוֹמַר כַּמָּה נָאֶה נָכְרִי זֶה כוּ'. גְּמָרָא:

(א) וְנִקְרָא בִּלְשׁוֹן אַשְׁכְּנַז עלינבוגי"ן:
(ב) וְאַף עַל פִּי שֶׁאֵין רְאָיָה לַדָּבָר זֵכֶר לַדָּבָר, שֶׁנֶּאֱמַר [יְחֶזְקֵאל כג] אֲשֶׁר בְּשַׂר חֲמוֹרִים בְּשָׂרָם. בַּגְּמָרָא:
(ג) פֵּרוּשׁ לְפֵרוּשׁוֹ, שֶׁבַּתְּחִלָּה אוֹמְדִין לִקַּח כְּשִׁעוּר הַיָּד כוּ', וּמְמַלְּאִין בּוֹ הֶחָבִית דְּשֶׁמָּא לֹא כִוְּנוּ הָאֹמֶד וְיִצְטָרֵךְ לְהוֹסִיף אוֹ לִגְרֹעַ, וּלְפִיכָךְ שׁוֹקְלִין אַחַר כָּךְ כְּשֶׁכְּבָר נִתְמַלֵּא הֶחָבִית מִמַּה שֶּׁהָיָה חָסֵר עַל יְדֵי הַיָּד, שֶׁעַכְשָׁיו אָנוּ יוֹדְעִים שֶׁלָּקַחְנוּ הַשִּׁעוּר בְּשָׁוֶה. וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם, שֶׁיִּטֹּל כוּ'‏ וְיִתֵּן לְתוֹךְ הֶחָבִית כוּ'‏ וְאַחַר כָּךְ שׁוֹקֵל. וְיִתָּכֵן שֶׁמְּפָרֵשׁ וְשׁוֹקֵל דְּמַתְנִיתִין מִלְּשׁוֹן שְׁקַל קָלָא פְּתַק בֵּיהּ שֶׁעִנְיָנוֹ נְטִילָה. אֲבָל הַמִּשְׁקָל לֹא תְנֵינָא לֵיהּ בְּמַתְנִיתִין אֶלָּא דְמִמֵּילָא הָכִי עָבְדִינַן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
