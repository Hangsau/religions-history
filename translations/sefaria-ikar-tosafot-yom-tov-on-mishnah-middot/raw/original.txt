(א) הָרַמְבַּ"ם. וְאֵין נִרְאִין דִּבְרֵי הַמְפָרֵשׁ דְּמַסֶּכֶת תָּמִיד שֶׁכָּתַב שֶׁיִּהְיוּ שׁוֹמְרִים יוֹם וְלַיְלָה:

(א) רֻבּוֹ כוּ'. שֶׁכָּל רֹחַב הָעֲזָרָה הוּא מֵאָה שְׁלֹשִׁים וְחָמֵשׁ [אַמָּה] מִצָּפוֹן לְדָרוֹם, כְּדִתְנַן רֵישׁ פֶּרֶק ה', וְנִמְצָא שֶׁהַנּוֹתָר מֵהַר הַבַּיִת בִּשְׁתֵּי רוּחוֹת הַלָּלוּ הוּא שְׁלֹשׁ מֵאוֹת שִׁשִּׁים וְחָמֵשׁ אַמָּה. [וְכָל אֹרֶךְ הָעֲזָרָה מִן הַמִּזְרָח לַמַּעֲרָב מֵאָה שְׁמוֹנִים וְשֶׁבַע], שֶׁבָּעֲזָרָה הַפְּנִימִית אֲנַן קַיָּמִין וְאֵין עֶזְרַת נָשִׁים בִּכְלָל, שֶׁעֶזְרַת נָשִׁים בִּכְלַל הַר הַבַּיִת, נִמְצָא שֶׁבְּאֵלּוּ שְׁתֵּי הָרוּחוֹת הַנּוֹתָר מֵהַר הַבַּיִת הוּא שְׁלֹשׁ מֵאוֹת וּשְׁלֹשׁ עֶשְׂרֵה כוּ'. הִנֵּה רֻבּוֹ לַדָּרוֹם, מָאתַיִם וַחֲמִשִּׁים. שֵׁנִי לוֹ לַמִּזְרָח, מָאתַיִם וּשְׁלֹשׁ עֶשְׂרֵה. לַצָּפוֹן, מֵאָה וַחֲמִשָּׁה עָשָׂר. לַמַּעֲרָב, מֵאָה:
(ב) רֹב כוּ'. שֶׁהוּא עַל לְשָׁכוֹת הַרְבֵּה וּבָתִּים שֶׁהָיוּ עֲשׂוּיִים בֶּאֱמֶת וְיָשָׁר לִמְשַׁמְּשֵׁי הַבַּיִת נוֹסָף עַל הַלְּשָׁכוֹת שֶׁנֶּאֶמְרוּ בַּמִּשְׁנָה, כִּי אוֹתָן לֹא יַסְפִּיקוּ, וְהֵם שְׁלֹשִׁים [וְסִיַּעְתָּא מִיחֶזְקֵאל מ']. וְאוּלַי הָיוּ עוֹד חֲמֵשׁ עֶשְׂרֵה לְהַמְּמֻנִּים שֶׁבְּפֶרֶק ה'‏ דִּשְׁקָלִים. וְטַעְמָא שֶׁהָיָה רֻבּוֹ בַּדָּרוֹם, לְפִי שֶׁשָּׁם הָיָה כְּנִיסַת כָּל אָדָם מֵהָעִיר שֶׁהִיא בַּנֶּגֶב כַּכָּתוּב בִּיחֶזְקֵאל כוּ'. שֵׁנִי לוֹ מִן הַמִּזְרָח, שֶׁכֵּן מַקִּיף דֶּרֶךְ יָמִין:

(א) שְׁלֹשִׁים וּשְׁתַּיִם כוּ'. לְמַאי דִּמְפָרֵשׁ הָרַ"ב בְּסוֹף הַמִּשְׁנָה שֶׁלֹּא הָיָה קֶרֶן בָּנוּי כְּלָל בְּמִזְרָח וְדָרוֹם כִּי אִם בְּאַמָּה מִכָּאן וְאַמָּה מִכָּאן, מְפָרֵשׁ בַּגְּמָרָא דְּכִי תְּנַן הָכָא שְׁלֹשִׁים וּשְׁתַּיִם עַל שְׁלֹשִׁים וּשְׁתַּיִם הַיְנוּ מִן הַצַּד, פֵּרֵשׁ רַשִׁ"י, מִצַּד צָפוֹן וּמַעֲרָב הָיוּ שְׁלֹשִׁים וּשְׁתַּיִם, וְלֹא בְּאַרְבַּעַת רְבָעָיו, דִּבְמִזְרָח וְדָרוֹם לֹא הָיָה אֶלָּא שְׁלֹשִׁים וְאַחַת עַל שְׁלֹשִׁים וְאַחַת:
(ב) גְּמָרָא. וּבַמְּכִילְתָּא דָּרִישׁ, וְאִם מִזְבַּח אֲבָנִים, אֵינוֹ רְשׁוּת אֶלָּא חוֹבָה. וְדִכְתִיב מִזְבַּח אֲדָמָה, שֶׁיְּהֵא מְחֻבָּר בָּאֲדָמָה:
(ג) שְׁלֹשִׁים כוּ'. בְּמַסֶּכֶת כֵּלִים פֶּרֶק י"ז מִשְׁנָה י'‏ תְּנַן דְּכָל אַמּוֹת בַּת שִׁשָּׁה טְפָחִים חוּץ מֵהַקֶּרֶן וְהַיְסוֹד וְהַסּוֹבֵב. וְיֵשׁ שְׁנֵי פֵּרוּשִׁים, לְרַשִׁ"י גֹּבַהּ הַיְסוֹד בְּבַת חֲמִשָּׁה, וְלֹא הַכְּנִיסָה. וּלְהָרַמְבַּ"ם אַף כְּנִיסַת הַיְסוֹד בְּבַת חֲמִשָּׁה, וּלְכָךְ כָּתַב שְׁלֹשִׁים אַמָּה וּשְׁנֵי טְפָחִים עַל כוּ'. וּמַתְנִיתִין לָא חֲשִׁיב לֵיהּ כֵּיוָן דְּלֹא הָוֵי אַמָּה:
(ד) עֶשְׂרִים וּשְׁמֹנֶה כוּ'. הָכָא לְכֻלֵּי עָלְמָא לָאו דַּוְקָא, דְּהָא כְּנִיסַת הַסּוֹבֵב לְכֻלֵּי עָלְמָא בַּת חֲמִשָּׁה טְפָחִים. [וְכֵן נַמִּי גֹּבַהּ הַקֶּרֶן] וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) כְּלוֹמַר, וְגֹבַהּ כָּל קֶרֶן אַמָּה, הֲרֵי גֹּבַהּ הַמִּזְבֵּחַ עֶשֶׂר אַמּוֹת. כֶּסֶף מִשְׁנֶה בְּשֵׁם סְמַ"ג. וְהָרר"שׁ לֹא גָּרַס לֵיהּ וְכָתַב מְקוֹם הַקְּרָנוֹת שֶׁלֹּא הָיָה שָׁם כְּנִיסָה לֹא קָחֲשִׁיב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ו) עֶשְׂרִים וְאַרְבַּע. הָרַמְבַּ"ם כָּתַב, עֶשְׂרִים וְאַרְבַּע אַמָּה וְאַרְבָּעָה טְפָחִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ז) הוֹסִיפוּ. וְהָכְתִיב הַכֹּל בִּכְתָב מִיַּד ה'‏ עָלַי הִשְׂכִּיל [דִּבְרֵי הַיָּמִים א כח]. אָמַר רַב יוֹסֵף קְרָא אַשְׁכַּח וְדָרוּשׁ, וַיֹאמֶר דָּוִד זֶה הוּא בֵּית ה'‏ הָאֱלֹהִים וְזֶה מִזְבֵּחַ לְעוֹלָה לְיִשְׂרָאֵל. כִּי בַּיִת, מַה בַּיִת שִׁשִּׁים אַמָּה [כְּדִכְתִיב [מְלָכִים א ו] שִׁשִּׁים אַמָּה אָרְכּוֹ. רַשִׁ"י] אַף מִזְבֵּחַ שִׁשִּׁים אַמָּה. פֵּרֵשׁ רַשִׁ"י, וּשְׁלֹמֹה לֹא הֵבִין לְדָרְשׁוֹ וְהֵם דְּרָשׁוּהוּ וְהוֹסִיפוּ לְפִי הַצֹּרֶךְ לָהֶם, וּלְשִׁשִּׁים אַמָּה לֹא הֻצְרְכוּ. וְהָכִי דָּרַשׁ, עַד שִׁשִּׁים אַמָּה רָאוּי לוֹ וּמְקֻדָּשׁ הַמָּקוֹם אִם יִרְצוּ:
(ח) הַדָּרוֹם. וְאִם תֹּאמַר מִסְּתָמָא לֹא הָיָה בְּבַיִת רִאשׁוֹן יְסוֹד לְקֶרֶן דְּרוֹמִית מִזְרָחִית לְפִי שֶׁלֹּא הָיָה בְּחֶלְקוֹ שֶׁל טוֹרֵף. וְאִם בְּבַיִת שֵׁנִי הוֹסִיפוּ בַּדָּרוֹם אִם כֵּן בָּנוּ בְּחֶלְקוֹ שֶׁל יְהוּדָה. תֵּרְצוּ הַתּוֹסָפוֹת, דִּבְבַיִת רִאשׁוֹן הָיָה לָהֶן יְסוֹד לַדָּרוֹם שֶׁהָיָה מָשׁוּךְ בְּחֶלְקוֹ שֶׁל טוֹרֵף, חוּץ מֵרוּחַ מִזְרָחִית שֶׁלֹּא נִשְׁתַּנָּה שֶׁהָיָה בְּלֹא יְסוֹד עַל חֶלְקוֹ שֶׁל יְהוּדָה כְּמוֹ בְּבַיִת שֵׁנִי. וּקְרָאֵי דִּמְמַעֲטֵי בְּסוֹף פֶּרֶק אֵיזֶהוּ מְקוֹמָן קֶרֶן שֶׁאֵין לָהּ יְסוֹד, מִשּׁוּם בַּיִת שֵׁנִי נֶאֶמְרוּ. אִי נַמִּי כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ט) וּרְצוֹנוֹ לוֹמַר בִּכְתָב משיט"א שֶׁאֵין לָהּ אֶלָּא שְׁנֵי צְדָדִים. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(י) רוֹצֶה לוֹמַר בִּתְחִלָּה בִּשְׁעַת הַבִּנְיָן עָשׂוּ כֵן כוּ', וּכְמוֹ שֶׁפֵּרֵשׁ לְקַמָּן בַּיְסוֹד דְּמִזְרָח וְדָרוֹם. דְּאִם לֹא כֵן הֲוָה פְּגִימָה וּפוֹסֶלֶת:
(יא) וְחוּט כוּ'. שֶׁנֶּאֱמַר וְהָיְתָה הָרֶשֶׁת עַד חֲצִי הַמִּזְבֵּחַ, הַתּוֹרָה נָתְנָה מְחִצָּה לְהַבְדִּיל בֵּין דָּמִים הָעֶלְיוֹנִים לַדָּמִים הַתַּחְתּוֹנִים. גְּמָרָא:

=== 4 | chapter 4 ===
(א) הֵיכָל. יְרוּשַׁלְמִי. שֶׁמִּשָּׁם כַּלְכָּלָה יוֹצֵא לָעוֹלָם, בִּתְפִלָּה שֶׁל כֹּהֵן גָּדוֹל בְּיוֹם הַכִּפּוּרִים. עוֹד שָׁם, דְּבִיר, עַל שֵׁם וּדְבַר ה'‏ מִירוּשָׁלַיִם. וְאִיכָּא דְּאָמְרֵי, לְמַעַן אַחַי וְרֵעַי אֲדַבְּרָה נָא שָׁלוֹם בָּךְ [תְּהִלִּים קכב]. רַאֲבִיָ"ה:
(ב) לַהֵיכָל וְלַקֹּדֶשׁ. לַהֵיכָל כְּלַפֵּי חוּץ, וְלַקֹּדֶשׁ כְּלַפֵּי פְּנִים. הָרַ"שׁ:
(ג) מוּסַבּוֹת כוּ'. שֶׁנִּקְפָּלוֹת כְּמוֹ שֶׁנִּקְפָּלוֹת הַדְּלָתוֹת הַמְחֻלָּקוֹת. הָרַמְבַּ"ם:
