=== 1 | chapter 1 ===
בְּאֶחָד בַּאֲדָר מַשְׁמִיעִין עַל הַשְּׁקָלִים. בֵּית דִּין שׁוֹלְחִים בְּכָל עָרֵי יִשְׂרָאֵל וּמַכְרִיזִין שֶׁיָּבִיאוּ שִׁקְלֵיהֶן. לְפִי שֶׁבְּאֶחָד בְּנִיסָן צָרִיךְ לְהָבִיא קָרְבְּנוֹת צִבּוּר מִתְּרוּמָה חֲדָשָׁה, דִּכְתִיב (במדבר כח) זֹאת עוֹלַת חֹדֶשׁ בְּחָדְשׁוֹ לְחָדְשֵׁי הַשָּׁנָה, חַדֵּשׁ וְהָבֵא קָרְבָּן מִתְּרוּמָה חֲדָשָׁה, וְיָלְפִינַן חָדְשֵׁי הַשָּׁנָה דְּהָכָא, מֵרִאשׁוֹן הוּא לָכֶם לְחָדְשֵׁי הַשָּׁנָה (שמות יב), הִלְכָּךְ מַקְדִּימִין שְׁלֹשִׁים יוֹם, דְּהַיְנוּ מֵאֶחָד בַּאֲדָר, לְהַכְרִיז שֶׁיָּבִיאוּ שִׁקְלֵיהֶם:
וְעַל הַכִּלְאָיִם. מַכְרִיזִין שֶׁיְּמַעֲטוּ הַזֶּרַע הָאַחֵר עַד שֶׁלֹּא יִשָּׁאֵר בּוֹ רֹבַע קַב לִסְאָה, כְּדִתְנַן בְּפֶרֶק ב'‏ דְּכִלְאַיִם כָּל סְאָה שֶׁיֵּשׁ בּוֹ רֹבַע קַב מִמִּין אַחֵר יְמַעֵט. וְרַבּוֹתֵינוּ פֵּרְשׁוּ דִּלְאַחַר שֶׁגָּדְלוּ הַזְּרָעִים אֲפִלּוּ אֶחָד בְּאֶלֶף צָרִיךְ לַעֲקֹר הַכֹּל, דְּכָל תְּרֵי מִילֵי דְּכָל חַד לְחוּדֵיהּ שָׁרֵי וְנֶאֶסְרוּ עַל יְדֵי תַּעֲרֹבֶת לֹא שַׁיָּךְ בְּהוּ בִטּוּל, דְּדַוְקָא כְּשֶׁמִּין אַחֵר מְעֹרָב בִּשְׁעַת הַזֶּרַע אֵין צָרִיךְ כִּי אִם לְמַעֵט, דְּמִדְּאוֹרַיְתָא חַד בִּתְרֵי בָּטִיל וְלֹא מִקְרֵי זוֹרֵעַ כִּלְאַיִם, אֶלָּא מִפְּנֵי מַרְאִית הָעַיִן בְּעַלְמָא צָרִיךְ לְמַעֵט. אֲבָל לְאַחַר שֶׁגָּדְלוּ לֹא שַׁיָּךְ בְּהוּ בִּטּוּל וְצָרִיךְ לַעֲקֹר אֶת הַכֹּל וְלֹא יְשַׁיֵּר אֶלָּא אֶחָד מִן הַמִּינִים:
בַּחֲמִשָּׁה עָשָׂר בּוֹ קוֹרִין אֶת הַמְגִלָּה בַּכְּרַכִּין. הַמֻּקָּפִין חוֹמָה מִימוֹת יְהוֹשֻׁעַ בִּן נוּן. אַיְידֵי דְּבָעֵי לְמִתְנֵי דְּבַחֲמִשָּׁה עָשָׂר בַּאֲדָר שֻׁלְחָנוֹת יוֹשְׁבִין בַּמְּדִינָה, תָּנֵי נַמִּי כָּל מַה שֶּׁהָיוּ עוֹשִׂין בּוֹ בַּיּוֹם:
וּמְתַקְּנִים אֶת הַדְּרָכִים וְאֶת הָרְחוֹבוֹת. הַשְּׁוָקִים שֶׁנִּתְקַלְקְלוּ בַּגְּשָׁמִים בִּימוֹת הַחֹרֶף, מְתַקְּנִים אוֹתָם בִּשְׁבִיל עוֹלֵי רְגָלִים. וְאִית דִּמְפָרְשֵׁי בִּשְׁבִיל הָרוֹצְחִים הַמַּכִּים נֶפֶשׁ בִּשְׁגָגָה, שֶׁיּוּכְלוּ לָנוּס מִפְּנֵי גּוֹאֵל הַדָּם, כְּדִכְתִיב (דברים יט) תָּכִין לְךָ הַדֶּרֶךְ:
וְאֶת מִקְוְאוֹת הַמַּיִם. אִם נִתְרַבָּה בָּהֶן טִיט מְנַקִּין אוֹתָן, וְאִם נֶחֱסַר שִׁעוּרָן מַמְשִׁיכִין לָהֶם מַיִם שְׁאוּבִים וּמַשְׁלִימִין אוֹתָן לִכְשִׁעוּר אִם הָיָה שָׁם רֹב אַרְבָּעִים סְאָה מִן הַכָּשֵׁר:
וְעוֹשִׂים כָּל צָרְכֵי רַבִּים. כְּגוֹן דִּינֵי מָמוֹנוֹת וְדִינֵי נְפָשׁוֹת, וְדִינֵי מַכּוֹת, וּפְדִיַּת עֲרָכִים וַחֲרָמִים וְהַקְדָּשׁוֹת, וְהַשְׁקָאַת סוֹטָה, וּשְׂרֵפַת פָּרָה, וּרְצִיעַת עֶבֶד עִבְרִי, וְטָהֳרַת מְצֹרָע, וְשׁוֹלְחִין לִפְתֹּחַ בּוֹרוֹת שֶׁל מַיִם מְכֻנָּסִים כְּדֵי שֶׁיִּהְיוּ מְצוּיִין לָעָם לִשְׁתּוֹת מֵהֶן בִּימוֹת הַחַמָּה. דְּכָל הָנָךְ צָרְכֵי רַבִּים נִינְהוּ:
וּמְצַיְּנִין אֶת הַקְּבָרוֹת. שֶׁלֹּא יַאֲהִילוּ עֲלֵיהֶם הַכֹּהֲנִים וְעוֹשֵׂי טָהֳרוֹת. וְהַצִּיּוּן הוּא שֶׁמְּמַחִים סִיד וְשֹׁפְכִים אוֹתוֹ סְבִיב הַקֶּבֶר, וּבִימוֹת הַגְּשָׁמִים הָיָה הַסִּיד נִמּוֹחַ וְצָרִיךְ לַחְזֹר וּלְצַיְּנוֹ:
וְיוֹצְאִים. שְׁלוּחֵי בֵּית דִּין:
אַף עַל הַכִּלְאָיִם. וְאַף עַל פִּי שֶׁכְּבָר הִכְרִיזוּ עָלֵיהֶם בְּאֶחָד בַּאֲדָר, לֹא הָיוּ סוֹמְכִים עַל הַהַכְרָזָה, דְּשֶׁמָּא לֹא עֲקָרוּם בְּעָלִים. וְהָיוּ הֵם יוֹצְאִים וְעוֹקְרִים אוֹתָם:

=== 2 | chapter 2 ===
מְצָרְפִין שְׁקָלִים לְדַרְכּוֹנוֹת. בְּנֵי הָעִיר שֶׁקִּבְּצוּ שִׁקְלֵיהֶם יְכוֹלִים לְהַחְלִיפָם בַּדָּרְבוֹנוֹת. וְהוּא מַטְבֵּעַ שֶׁל זָהָב כְּדִכְתִיב בְּעֶזְרָא דַּרְכְּמוֹנֵי זָהָב, לְהָקֵל מֵעֲלֵיהֶם מַשְּׂאוֹי הַדֶּרֶךְ:
שׁוֹפָרוֹת. תֵּבוֹת שֶׁפִּיהֶן צָר לְמַעְלָה, כְּעֵין שׁוֹפָר שֶׁפִּיו צַר מִלְּמַעְלָה וְהוֹלֵךְ וּמַרְחִיב, כְּדֵי שֶׁלֹּא יוּכְלוּ לִטֹּל מֵהֶן כְּלוּם. כְּהַהִיא דִּכְתִיב בִּיהוֹיָדָע (מלכים ב יב) וַיַּעֲשׂוּ אֲרוֹן אֶחָד וַיִּתְּנוּהוּ בְּהֵיכַל וְכוּ'‏ וַיִּקֹּב חֹר בְּדַלְתּוֹ. וְהָיוּ עוֹמְדִים בַּעֲזָרָה, וְכָל אֶחָד מֵבִיא שִׁקְלוֹ וְנוֹתֵן בּוֹ:
כָּךְ הָיוּ בַּמְּדִינָה. בִּירוּשָׁלַיִם. וּלְדִבְרֵי רַמְבַּ"ם בִּשְׁאָר עָרֵי יִשְׂרָאֵל:
בְּנֵי הָעִיר שֶׁשָּׁלְחוּ שִׁקְלֵיהֶם. בְּיַד שָׁלִיחַ לְהוֹלִיכָם לַלִּשְׁכָּה:
אִם נִתְרְמָה הַתְּרוּמָה. שֶׁהָיוּ רְגִילִים לִתְרֹם מֵהַקֻּפּוֹת לְצֹרֶךְ הַקָּרְבָּנוֹת. וְהָיוּ תּוֹרְמִים עַל הַגָּבוּי וְעַל הֶעָתִיד לִגְבּוֹת, כְּדֵי שֶׁיִּהְיֶה חֵלֶק בַּקָּרְבָּנוֹת אַף לְאוֹתָם שֶׁעֲדַיִן לֹא שָׁקְלוּ:
נִשְׁבָּעִין. הַשְּׁלוּחִים:
לַגִּזְבָּרִים. דְּהוֹאִיל וְנִתְרְמָה תְּרוּמָה עַל הַמָּעוֹת הַלָּלוּ קֹדֶם שֶׁנֶּאֶבְדוּ הֲרֵי נַעֲשׂוּ כְּאִלּוּ הָיוּ בִּרְשׁוּת הַגִּזְבָּרִים מִשָּׁעָה שֶׁנִּתְרְמָה תְּרוּמָה עֲלֵיהֶן, וּכְשֶׁנִּגְנְבוּ אוֹ אָבְדוּ בִּרְשׁוּת הַגִּזְבָּרִים נִגְנְבוּ אוֹ אָבְדוּ, הִלְכָּךְ נִשְׁבָּעִים הַשְּׁלוּחִים לַגִּזְבָּרִים וְנִפְטָרִים. וְאַף עַל גַּב דְּאֵין נִשְׁבָּעִין עַל הַהַקְדָּשׁוֹת, שְׁבוּעָה זוּ תַּקָּנַת חֲכָמִים הִיא כְּדֵי שֶׁלֹּא יְזַלְזְלוּ בַּהַקְדָּשׁוֹת:
וְאִם לָאו. שֶׁבַּשָּׁעָה שֶׁאָבְדוּ עֲדַיִן לֹא נִתְרְמָה תְּרוּמָה וְלֹא הוֹצִיאוּ מָעוֹת מִן הַקֻּפּוֹת עַל הֶעָתִיד לִגְבּוֹת, בִּרְשׁוּת הַבְּעָלִים אָבְדוּ, הִלְכָּךְ נִשְׁבָּעִים הַשְּׁלוּחִים לְאַנְשֵׁי הָעִיר וְנִפְטָרִים. וְאַנְשֵׁי הָעִיר חוֹזְרִין וְשׁוֹקְלִים שְׁקָלִים אֲחֵרִים תַּחְתֵּיהֶן, שֶׁהַשְּׁקָלִים הָרִאשׁוֹנִים שֶׁאָבְדוּ לֹא עָלוּ לָהֶן:

=== 3 | chapter 3 ===
בִּשְׁלֹשָׁה פְרָקִים בַּשָּׁנָה תּוֹרְמִין אֶת הַלִּשְׁכָּה. כָּל הַשְּׁקָלִים הָיוּ נוֹתְנִים לְתוֹךְ לִשְׁכָּה אַחַת שֶׁבַּמִּקְדָּשׁ, וְשָׁלֹשׁ פְּעָמִים בַּשָּׁנָה נוֹטְלִין מִמֶּנָּה, וְנוֹתְנִים לְתוֹךְ שָׁלֹשׁ קֻפּוֹת, שֶׁכָּל קֻפָּה הָיְתָה מִשָּׁלֹשׁ סְאִין, כְּדֵי לִקְנוֹת מֵהֶן קָרְבְּנוֹת צִבּוּר. וְלֹא הָיוּ נוֹטְלִים הַכֹּל בְּפַעַם אַחַת לְצֹרֶךְ כָּל הַשָּׁנָה, לְפִי שֶׁהָרְחוֹקִים לֹא הֵבִיאוּ עֲדַיִן כָּל שִׁקְלֵיהֶם:
בִּפְרֹס הַפֶּסַח. חֲמִשָּׁה עָשָׂר יוֹם קֹדֶם הַמּוֹעֵד קָרֵי פְּרֹס הַמּוֹעֵד, לְפִי שֶׁשְּׁלֹשִׁים יוֹם קֹדֶם הָרֶגֶל שׁוֹאֲלִים וְדוֹרְשִׁים בְּהִלְכוֹת הָרֶגֶל, וּפְרֹס הוּא לְשׁוֹן פְּרוּסָה דְּהַיְנוּ חֵצִי:

מֶחֱצָה עַל מֶחֱצָה יִפְּלוּ לְגוֹזְלֵי עוֹלָה. בַּיְרוּשַׁלְמִי פָּרֵיךְ, בִּשְׁלָמָא קָרוֹב לְגוֹזְלֵי עוֹלָה, כֵּיוָן דְּמִדְּאוֹרַיְתָא אָזְלִינַן בָּתַר קוּרְבָה הֲוֵי כְּאִלּוּ וַדַּאי הֵן מִגּוֹזְלֵי עוֹלָה, אֶלָּא מֶחֱצָה עַל מֶחֱצָה, דְּמִסְּפֵיקָא שָׁדִינַן לְהוּ לְגוֹזְלֵי עוֹלָה לְחֻמְרָא, הָאִשָּׁה שֶׁהֵבִיאָה הַמָּעוֹת הַלָּלוּ בְּמַה תִּתְכַּפֵּר, שֶׁמָּא מִקִּנֵּי חוֹבָה נָפְלוּ, וּכְשֶׁאַתָּה מַקְרִיב מֵהֶן גּוֹזְלֵי עוֹלָה בְּמַה מִּתְכַּפֶּרֶת הָאִשָּׁה. וּמְשָׁנֵי, בֵּית דִּין הַמְּמֻנִּים עַל הַקִנִּין לוֹקְחִים מִשֶּׁל צִבּוּר כְּפִי אוֹתָן מָעוֹת שֶׁנִּמְצְאוּ, וּמַקְנִין אוֹתָן לְבַעֲלֵי הַמָּעוֹת שֶׁנִּמְצְאוּ, וּמַקְרִיבִין מֵהֶן קִנִּין עַל הַסָּפֵק, וְהַחַטָּאת לֹא תְהֵא נֶאֱכֶלֶת:
