מֵי רְבִיעִית. מַיִם שֶׁיֵּשׁ בָּהֶן שִׁעוּר רְבִיעִית הַלֹּג, שֶׁהוּא בֵּיצָה וּמֶחֱצָה:
נוֹתְנִים לַיָּדַיִם לְאֶחָד אַף לִשְׁנַיִם. לְחַד בָּעִינַן מִתְּחִלָּה רְבִיעִית. אַף לִשְׁנַיִם, אִם הָיָה בּוֹ מִתְּחִלָּה רְבִיעִית וְנָטַל הָרִאשׁוֹן וְאַחַר כָּךְ הַשֵּׁנִי, אַף עַל גַּב דְּלֹא הָיָה רְבִיעִית גַּבֵּי הַשֵּׁנִי, שַׁפִּיר דָּמֵי, הוֹאִיל וּמִשְּׁיָרֵי טָהֳרָה אֲתוּ. וְרַמְבַּ"ם פֵּרֵשׁ, דְּדַוְקָא מַיִם שְׁנִיִּים הוּא דְּלֹא בָּעִינַן רְבִיעִית לְכָל חַד וְחַד, הוֹאִיל וּכְבָר נִטְהֲרוּ יָדָיו מִשְּׁטִיפָה רִאשׁוֹנָה, אֲבָל בְּמַיִם רִאשׁוֹנִים, אֵין פָּחוֹת מֵרְבִיעִית לְכָל אֶחָד. וְרַבּוֹתַי לֹא פֵּרְשׁוּ כֵּן:
מַחֲצִית לֹג לִשְׁלֹשָׁה אוֹ לְאַרְבָּעָה. לְפוּם שִׁעוּרָא דִּרְבִיעִית לִשְׁנַיִם, הָיָה דַּי בִּרְבִיעִית וּמֶחֱצָה לִשְׁלֹשָׁה, אֶלָּא כֵּיוָן דִּנְפִישֵׁי גַּבְרֵי חָיְשִׁינַן שֶׁמָּא לֹא יְדַקְדְּקוּ יָפֶה לִטֹּל עַד הַפֶּרֶק, שֶׁכָּל אֶחָד מְצַמְצֵם בַּמַּיִם מִפְּנֵי חֲבֵרוֹ. וּמִיהוּ אִי אִיכָּא רְבִיעִית וּמֶחֱצָה, וְנָטַל הָרִאשׁוֹן מֵהֶם חֲצִי רְבִיעִית, הָא וַדַּאי דְּיוֹעִיל רְבִיעִית הַנִּשְׁאָר לַשְּׁנַיִם הָאַחֲרוֹנִים, דַּאֲפִלּוּ בִּתְחִלָּה סַגִּי לִשְׁנַיִם בִּרְבִיעִית, כָּל שֶׁכֵּן הַשְׁתָּא דַּאֲתוּ מִשְּׁיָרֵי טָהֳרָה. וְכִי לֹא מַהֲנֵי הָכָא, הָנֵי מִלֵּי הֵיכָא דְּלֹא נִשְׁאַר לַשְּׁנַיִם הָאַחֲרוֹנִים רְבִיעִית:
וּבִלְבַד שֶׁלֹּא יִפְחֹת לָאַחֲרוֹן שֶׁבָּהֶן מֵרְבִיעִית. שֶׁיִּהְיֶה בַּכְּלִי בְּשָׁעָה שֶׁהָאַחֲרוֹן נוֹטֵל אֶת יָדָיו, רְבִיעִית. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי:
מוֹסִיפִין עַל הַשְּׁנִיִּים וְאֵין מוֹסִיפִין עַל הָרִאשׁוֹנִים. דִּין נְטִילַת יָדַיִם שֶׁצָּרִיךְ לִטֹּל יָדָיו עַד הַפֶּרֶק, וּלְאַחַר שֶׁנָּטַל יָדָיו בִּשְׁטִיפָה אַחַת אַף עַל גַּב דְּיָדָיו טְהוֹרוֹת, אִם קִנַּח אוֹתָם בְּמַפָּה, מִכָּל מָקוֹם אִם חוֹזֵר וְנוֹגֵעַ בַּמַּיִם שֶׁעַל הַמַּפָּה, יָדָיו טְמֵאוֹת, וְאִם נָגַע בָּהֶן כִּכָּר שֶׁל תְּרוּמָה, נִטְמָא, וּלְכָךְ אוֹרְחֵיהּ דַּעֲבִיד שְׁטִיפָה אַחֶרֶת בְּמַיִם שְׁנִיִּים לְטַהֵר הַמַּיִם הָרִאשׁוֹנִים שֶׁעַל גַּבֵּי יָדָיו. וְהַשְׁתָּא קָאָמַר דְּאִם נָתַן הַמַּיִם שְׁנִיִּים וְלֹא הִגִּיעוּ לַפֶּרֶק, יָכוֹל לְהוֹסִיף עֲלֵיהֶם וְלִשְׁפֹּךְ מַיִם בַּמָּקוֹם שֶׁלֹּא שָׁפַךְ שָׁם תְּחִלָּה, וְיִטַּהֲרוּ הַמַּיִם הָרִאשׁוֹנִים שֶׁעַל גַּבֵּי יָדָיו. אֲבָל עַל הָרִאשׁוֹנִים אִם לֹא הִגִּיעוּ לַפֶּרֶק, אֵינוֹ יָכוֹל לְהוֹסִיף, אֶלָּא צָרִיךְ לְנַגֵּב יָדָיו וְלַחֲזֹר וְלִשְׁפֹּךְ בְּבַת אַחַת מַיִם שֶׁיַּגִּיעוּ לַפֶּרֶק:

נָטַל יָדוֹ אַחַת מִשְּׁטִיפָה אַחַת. קָא קָרֵי שְׁטִיפָה אַחַת כָּל זְמַן שֶׁלֹּא נָטַל כִּי אִם פַּעַם אַחַת עַד הַפֶּרֶק. וּבְיָדוֹ אַחַת הוּא דְּסַגִּי בְּפַעַם אַחַת, אַף עַל גַּב דְּאֵין בַּכְּלִי רְבִיעִית אֶלָּא דַּאֲתוּ מִשְּׁיָרֵי טָהֳרָה. אֲבָל לִשְׁתֵּי יָדָיו, אַף עַל גַּב דַּאֲתוּ מִשְּׁיָרֵי טָהֳרָה, לֹא מַהֲנֵי עַד שֶׁיִּטֹּל שְׁתֵּי פְּעָמִים. רִאשׁוֹנִים עַד הַפֶּרֶק, וּשְׁנִיִּים עַד הַפֶּרֶק. וְרַמְבַּ"ם מְפָרֵשׁ, מִשְּׁטִיפָה אַחַת, מִשְּׁפִיכָה אַחַת. וּכְשֶׁנּוֹטֵל שְׁתֵּי יָדָיו בִּשְׁפִיכָה אַחַת, הַמַּיִם מִטַּמְּאִים בְּיָדוֹ אַחַת, וְנִמְצָא רוֹחֵץ הַשְּׁנִיָּה בַּמַּיִם שֶׁנִּטְמָא, וְלֹא טָהֲרוּ יָדָיו עַד שֶׁיִּטֹּל מַיִם שְׁנִיִּים מֵרְבִיעִית. אֲבָל כְּשֶׁנּוֹטֵל יָדוֹ אַחַת בִּלְבַד, בִּשְׁפִיכָה אַחַת, נִטְהֲרָה יָדוֹ. וּכְשֶׁנּוֹטֵל מַיִם שְׁנִיִּים לְטַהֵר הַמַּיִם שֶׁעַל יָדוֹ, סַגִּי בְּפָחוֹת מֵרְבִיעִית. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר:
נָפַל כִּכָּר שֶׁל תְּרוּמָה. הֵיכָא דְּנָטַל אֶת הָרִאשׁוֹנִים מֵרְבִיעִית, וְנָפַל עֲלֵיהֶם כִּכָּר שֶׁל תְּרוּמָה, אוֹ נָגַע בְּכִכָּר שֶׁל תְּרוּמָה עִם הַמַּיִם שֶׁבְּיָדוֹ קֹדֶם שֶׁיְּנַגְּבֵם, אוֹ קֹדֶם שֶׁיִּשְׁפֹּךְ עֲלֵיהֶן מַיִם שְׁנִיִּים:
טָהוֹר. דְּכִי הֵיכִי דְּמַהֲנוּ לְטַהֵר יָדָיו, מַהֲנוּ נַמִּי לְטַהֵר עַצְמָן:
רַבִּי יוֹסֵי מְטַמֵּא. דְּסָבַר לְטַהֵר יָדָיו מַהֲנוּ, אֲבָל אִינְהוּ גּוּפַיְהוּ טְמֵאִים:

=== 3 | chapter 3 ===
הַמַּכְנִיס יָדָיו לְבַיִת הַמְנֻגָּע יָדָיו תְּחִלּוֹת. פּוֹסֵל אֶחָד בְּחֻלִּין, דְּרִאשׁוֹן עוֹשֶׂה שֵׁנִי בְּחֻלִּין. וּמְטַמֵּא אֶחָד וּפוֹסֵל אֶחָד בַּתְּרוּמָה. וּמְטַמֵּא שְׁנַיִם וּפוֹסֵל אֶחָד בַּקֹּדֶשׁ:
כֹּל הַמְטַמֵּא בְגָדִים בִּשְׁעַת מַגָּעוֹ. כְּגוֹן זוֹבוֹ שֶׁל זָב וְרֻקּוֹ, וְכָל הָנֵי דִּתְנַן בְּפֶרֶק בַּתְרָא דְּזָבִים, שֶׁהַנּוֹגֵעַ בָּהֶן מְטַמֵּא בְּגָדִים בִּשְׁעַת מַגָּעוֹ:
וְכִי הֵיאַךְ אֶפְשָׁר לָהֶם לִהְיוֹת תְּחִלָּה. שֶׁאֵין רָאוּי לִהְיוֹת תְּחִלָּה אֶלָּא הַנּוֹגֵעַ בְּאַב הַטֻּמְאָה. וְאִם נָגַע בְּיָדָיו בְּאַב הַטֻּמְאָה, נִטְמָא כָּל גּוּפוֹ. הִלְכָּךְ אֵין בְּיָדִי לִמְצֹא [לָכֶם] יָדַיִם תְּחִלָּה, חוּץ מֵהַמַּכְנִיס יָדָיו לְבַיִת הַמְנֻגָּע, שֶׁאִם הִכְנִיס יָדָיו לֹא נִטְמָא כֻּלּוֹ. וְהַמְטַמֵּא בְּגָדִים בִּשְׁעַת מַגָּעוֹ, שֶׁאַף עַל פִּי שֶׁהוּא חָשׁוּב אַב הַטֻּמְאָה לְטַמֵּא בְּגָדִים וְלַעֲשׂוֹת רִאשׁוֹן וְשֵׁנִי, אֵינוֹ חָשׁוּב אַב הַטֻּמְאָה לְטַמֵּא אָדָם, אֲבָל מְטַמֵּא אֶת הַיָּדַיִם שֶׁנָּגְעוּ בוֹ לִהְיוֹת תְּחִלּוֹת. וְאֵין הֲלָכָה כְּרַבִּי עֲקִיבָא:
אֶת שֶׁנִּטְמָא בְאַב הַטֻּמְאָה. כְּלוֹמַר, אֳכָלִים וְכֵלִים שֶׁנִּטְמְאוּ בְּאַב הַטֻּמְאָה:
מְטַמְּאִין אֶת הַיָּדַיִם. וְלֹא אֳכָלִים וְכֵלִים שֶׁנִּטְמְאוּ בְּמַשְׁקִין. דְּאֵין שֵׁנִי עוֹשֶׂה שֵׁנִי. וַהֲלָכָה כַּחֲכָמִים:
בַּמֶּה הָיְתָה הַטֻּמְאָה. שֶׁל תַּנּוּר, בְּאַב הַטֻּמְאָה אוֹ בְמַשְׁקֶה:

=== 4 | chapter 4 ===
בּוֹ בַיּוֹם. אַפִּרְקִין דִּלְעֵיל קָאֵי, אַמִּלְּתֵיהּ דְּרַבִּי שִׁמְעוֹן בֶּן עַזַּאי, דְּמַיְרֵי בַּיּוֹם שֶׁהוֹשִׁיבוּ אֶת רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה בַּיְשִׁיבָה. וְכָל מָקוֹם שֶׁנֶּאֱמַר בַּמִּשְׁנָה בּוֹ בַּיּוֹם, עַל אוֹתוֹ יוֹם נֶאֱמַר:
נִמְנוּ. עָמְדוּ לְמִנְיָן, לָדַעַת מִנְיַן הַמְטַמְּאִים וּמִנְיַן הַמְטַהֲרִים, וְגָמְרוּ הַדִּין עַל פִּי הָרֹב:
עֲרֵבַת הָרַגְלַיִם. עֲרֵבָה עֲשׂוּיָה לִרְחִיצַת הָרַגְלַיִם:
שֶׁנִּסְדְּקָה. סָמוּךְ לְשׁוּלֶיהָ, וְאֵינָהּ מַחֲזֶקֶת מַיִם כְּדֵי רְחִיצַת רַגְלוֹ אַחַת:
טְמֵאָה מִדְרָס. דַּעֲרֵבָה קֹדֶם שֶׁנִּסְדְּקָה נַמִּי הָיוּ רְגִילִים לֵישֵׁב בָּהּ, וְהָיְתָה מְשַׁמֶּשֶׁת יְשִׁיבָה עִם מְלַאכְתָּהּ:
שֶׁרַבִּי עֲקִיבָא. הָיָה חוֹלֵק וְאוֹמֵר עֲרֵבַת הָרַגְלַיִם כִּשְׁמָהּ, כְּלוֹמַר כִּשְׁמָהּ כֵּן הִיא, שֶׁאֵין שֵׁם מִדְרָס עָלֶיהָ. וְאֵין הֲלָכָה כְּרַבִּי עֲקִיבָא:
