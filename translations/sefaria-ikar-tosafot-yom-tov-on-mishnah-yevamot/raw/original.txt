(א) חֲמֵשׁ עֶשְׂרֵה כוּ'. מִנְיָנָא דְּהָכָא וְכֵן דְּסֵיפָא הֲרֵי אֵלּוּ כוּ', לְמִעוּטֵי צָרַת אַיְלוֹנִית שֶׁהִכִּיר בָּהּ, אוֹ צָרַת קְטַנָּה מְמָאֶנֶת בְּיָבָם, גְּזֵרָה מִשּׁוּם צָרַת בִּתּוֹ מְמָאֶנֶת אוֹ צָרַת מַחֲזִיר גְּרוּשָׁתוֹ. גְּמָרָא:
(ב) פּוֹטְרוֹת. לֹא תָּנֵי אוֹסְרוֹת, אַף עַל גַּב דְּוַדַּאי אִסּוּרָא אִיכָּא לְיַבֵּם, וְלַחְלֹץ נַמִּי שַׁיָּךְ לְמֵיסַר, שֶׁאִם אַתָּה אוֹמֵר חוֹלֶצֶת אָתֵי לְמֵימַר דְּתִתְיַבֵּם, אֶלָּא מִשּׁוּם שֶׁכֵּיוָן דְּבִמְקוֹם מִצְוָה הִיא דַּאֲסוּרָה וְשֶׁלֹּא בִּמְקוֹם מִצְוָה כְּגוֹן נָכְרִי שֶׁנַּעֲשָׂה חֲתָנוֹ וְלוֹ אִשָּׁה אַחֶרֶת וּמֵת מֻתֶּרֶת אִשְׁתּוֹ לְחָמִיו כְּדִתְנַן בְּמִשְׁנָה ג', מִשּׁוּם הָכִי תָּנֵי פּוֹטְרוֹת, דְּאִי תָּנֵי אוֹסְרוֹת מַשְׁמַע שֶׁהַבַּת אוֹסַרְתָּהּ עָלָיו לִקַּח, וְהוּא הַדִּין שֶׁלֹּא בִּמְקוֹם מִצְוָה. גְּמָרָא:
(ג) פּוֹטְרוֹת צָרוֹתֵיהֶן. דְּאִלּוּ הֵן עַצְמָן דִּפְטוּרוֹת מִלְּתָא דִּפְשִׁיטָא הִיא וְלֹא אִיצְטְרִיךְ לְמִתְנֵי, דְּוַדַּאי דְּלֹא אָתֵי עֲשֵׂה דְּיִבּוּם וְדוֹחֶה לֹא תַעֲשֶׂה שֶׁיֵּשׁ בּוֹ כָּרֵת. וְהָא דִּכְתִיב עָלֶיהָ כְּדִמְפָרֵשׁ הָרַ"ב מִשּׁוּם דְּרָשָׁא דְּלִצוֹר דְּאִיצְטְרִיךְ לֵיהּ. גְּמָרָא:
(ד) דְּאָמַר קְרָא כִּי כָּל אֲשֶׁר יַעֲשֶׂה מִכָּל הַתּוֹעֵבוֹת הָאֵלֶּה וְנִכְרְתָה, הֻקְשׁוּ כָּל הָעֲרָיוֹת כֻּלָּן לַהֲדָדֵי. גְּמָרָא:
(ה) מִן הַחֲלִיצָה. שֶׁנֶּאֱמַר וְעָלְתָה יְבִמְתּוֹ הַשַּׁעְרָה וְגוֹ'‏ מֵאֵן יַבְּמִי וְגוֹ'‏ וְחָלְצָה נַעֲלוֹ, כָּל הָעוֹלֶה לְיִבּוּם עוֹלֶה לַחֲלִיצָה, וְכָל שֶׁאֵינוֹ עוֹלֶה לְיִבּוּם אֵינוֹ עוֹלֶה לַחֲלִיצָה אֶלָּא תִפָּטֵר:
(ו) וּמִן הַיִּבּוּם. לְהָכִי לֹא סַגִּי לֵיהּ לְמִתְנֵי מִן הַחֲלִיצָה לְחֻדֵּיהּ, לְאַשְׁמְעִינַן דְּמִצְוַת חֲלִיצָה קוֹדֶמֶת. גְּמָרָא:
(ז) בְּקַל וָחֹמֶר כוּ'. וְאַף דְּאֵין עוֹנְשִׁין מִן הַדִּין, הָכָא גִּלּוּי מִלְּתָא הִיא. אִי נַמִּי אַתְיָא הֵנָּה הֵנָּה וְאַתְיָא זִמָּה זִמָּה, וְטַעַם הַסֵּדֶר דְּלֹא כַּכָּתוּב עַיֵּן בַּגְּמָרָא:
(ח) חֲמוֹתוֹ. שֶׁמֵּת בַּעֲלָהּ וּנְשָׂאָהּ אָחִיו וּמֵת בְּלֹא בָּנִים. רַשִׁ"י:
(ט) דִּכְתִיב הָכָא כִּי יֵּשְׁבוּ אַחִים, וְהָתָם כְּתִיב שְׁנֵים עָשָׂר עֲבָדֶיךָ אַחִים. וְלֹא יָלְפִינַן מִלּוֹט דִּכְתִיב כִּי אֲנָשִׁים אַחִים אֲנַחְנוּ, דִּלְהָכִי כְּתִיב יַחְדָּו (הַמְּיֻחָדִים בְּנַחֲלָה). וְאִי כְּתִיב יַחְדָּו לְחוּד הֲוָה אֲמִינָא עַד דִּמְיַחֲדֵי בְּאַבָּא וּבְאִמָּא. גְּמָרָא:
(י) וְלֹא נָקְטָה בְּדִיּוּקָא, אֶלָּא דְּבָעֵי לְפֵרוּשֵׁי לְדִבְרֵי הַכֹּל. מִיהוּ אַלִּיבָּא דְּהִלְכָתָא קַיְמָא לָן כְּרַבָּנָן דְּפֶרֶק ב'‏ מִשְׁנָה ב'‏ דַּאֲסוּרָה עָלָיו בֵּין שֶׁנּוֹלַד וְאַחַר כָּךְ יִבֵּם וּבֵין שֶׁיִּבֵּם וְאַחַר כָּךְ נוֹלַד. נִמּוּקֵי יוֹסֵף:
(יא) מִיֵּשְׁבוּ יָלְפִינַן, דְּיַחְדָּו הָא אִיצְטְרִיךְ לְעֵיל. וְאֵין לְהַקְשׁוֹת דְּלָמָּה אִיצְטְרִיךְ דְּרָשָׁא דְּיֵשְׁבוּ, הָא גַּבֵּי בְּנֵי יַעֲקֹב לֹא מָצִינוּ אַחִים אֶלָּא כְּשֶׁהָיוּ בָּעוֹלָם כְּאֶחָד, דִּפְשִׁיטָא דַּאֲפִלּוּ הָכִי אַחִין מִקְרוּ. תּוֹסָפוֹת:
(יב) מִשּׁוּם דְּלִישְׁנָא דְּנִמְצְאוּ הָכִי דַיִּק, וְכָךְ מְפָרֵשׁ רַב אַסִי בַּגְּמָרָא דַּף י"ב. אֲבָל רָבָא פָּסַק הִלְכָתָא אֲפִלּוּ הִכִּיר בָּהּ אָחִיו וְקִיְּמָהּ וְהַוְיָא אִשְׁתּוֹ לֹא מִתְּסְרָא צָרָתָהּ מִשּׁוּם צָרַת הַבַּת. וּפֵרֵשׁ רַשִׁ"י דְּכִי אָסַר רַחֲמָנָא צָרַת עֶרְוָה בִּמְקוֹם מִצְוָה הוּא דַּאֲסָרָהּ, וְהָא כֵּיוָן דְּאַיְלוֹנִית הִיא דִּבְלָאו עֶרְוָה נַמִּי לֹא חַזְיָא לְיִבּוּם כְּדִתְנַן בְּמִשְׁנָה ה'‏ פֶּרֶק ח', הַוְיָא צָרָתָהּ צָרַת עֶרְוָה שֶׁלֹּא בִּמְקוֹם מִצְוָה. וְאֶלָּא הָא דְּקָתָנֵי שֶׁנִּמְצְאוּ, תָּנֵי שֶׁהָיוּ:

(א) מַאֲמָר כוּ'. לְמַאי דְּפָסְקִינַן כְּרַבִּי יְהוּדָה בֶּן בְּתֵירָא דְּמִשְׁנָה ט'‏ פֶּרֶק ד'‏ דְּיֵשׁ זִקָּה, כְּלוֹמַר דְּמִשָּׁעָה שֶׁמֵּת בַּעֲלָהּ נַעֲשֵׂית שׁוֹמֶרֶת יָבָם וּזְקוּקָה לוֹ וַחֲשׁוּבָה כַּאֲרוּסָתוֹ, מְפָרֵשׁ הָכִי בַּגְּמָרָא הוּא הַדִּין אַף עַל גַּב דְּלֹא עָבֵיד בָּהּ מַאֲמָר שְׁנִיָּה מִחְלָץ חָלְצָה יִבּוּמִי לֹא מְיַבְּמָהּ, וְהָא דְּקָתָנֵי מַאֲמָר לַאֲפוֹקֵי מִדְּבֵית שַׁמַּאי דְּמִשְׁנָה ה'‏ פֶּרֶק ג'‏ דְּמַאֲמָר קוֹנֶה קִנְיָן גָּמוּר וְתִפָּטֵר אַף בְּלֹא חֲלִיצָה, קָא מַשְׁמַע לָן דְּאֵינוֹ אֶלָּא מִדְּרַבָּנָן:
(ב) וְנִקְרָא מַאֲמָר לְפִי שֶׁאֵין קִדּוּשִׁין אֶלָּא בַּאֲמִירָה וְלֹא בִּשְׁתִיקָה כוּ', אֲבָל יִבּוּם אֲפִלּוּ בְּשׁוֹגֵג דְּקָסָבַר אִשָּׁה אַחֶרֶת הִיא קַנְיָא, וְהָא דְּנָקַט בְּכֶסֶף הוּא הַדִּין בִּשְׁטָר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

=== 3 | chapter 3 ===
(א) אַרְבָּעָה אַחִין. לֹא מִבַּעְיָא שְׁלֹשָׁה אַחִין שְׁנַיִם מֵהֶן נְשׂוּאִים שְׁתֵּי אֲחָיוֹת וּמֵתוּ, דְּתַרְוַיְהוּ עֲלֵיהּ רַמְיָא וְזִקָּתוֹ חֲמוּרָה כוּ', אֶלָּא אֲפִלּוּ אַרְבַּע דְּנָפְלֵי קָמֵי תְּרֵי אַחֵי, דְּלֹא אַלִּימָא זִיקָתָא דְּקַמָּא דִּמְיַבֵּם, דְּמָצִינוּ לְמֵימַר הַאִיךְ הִיא דְרַמְיָא עֲלֵיהּ וַאֲחוֹתָהּ עַל אָחִיו, אֲפִלּוּ הָכִי אֲסִירֵי. רַשִׁ"י:
(ב) שְׁתֵּי אֲחָיוֹת. אוֹ אִשָּׁה וּבִתָּהּ אוֹ אִשָּׁה וּבַת בִּתָּהּ וְכַיּוֹצֵא בּוֹ. עִנְיַן הַדְּבָרִים שֶׁהֵם שְׁתֵּי נָשִׁים שֶׁהָאַחַת מֵהֶן עֶרְוָה עִם הָאַחֶרֶת וְאִי אֶפְשָׁר לִשָּׂא שְׁתֵּיהֶן מִשּׁוּם עֶרְוָה. הָרַמְבַּ"ם. וְנָקַט אֲחָיוֹת מִשּׁוּם דְּמִנַּיְהוּ יָלְפִינַן (עַיֵּן רֵישׁ פֶּרֶק קַמָּא בְּהָרַ"ב):
(ג) וְלֵיכָּא לְמֵימַר דְּכִי חָלִיץ חַד לַחֲדָא דִּלְיַבֵּם אִידָךְ לְאִידָךְ, דְּכָל יְבָמָה שֶׁנֶּאֶסְרָה שָׁעָה אַחַת עַל יְבָמָהּ הֲרֵי הִיא כְּאֵשֶׁת אָח שֶׁיֵּשׁ לוֹ בָּנִים וַאֲסוּרָה לְעוֹלָם. וְלֹא דָּמֵי לְמִשְׁנָה ז'‏ פֶּרֶק ב', דְּהָתָם לָאו יְבָמָה הִיא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

=== 4 | chapter 4 ===
(א) וְנִמְצֵאת. לֹא בָּעֵי לְמֵימַר הַחוֹלֵץ לִיבִמְתּוֹ מְעֻבֶּרֶת, מִשּׁוּם דְּלָאו אֹרַח אַרְעָא, וְכָל שֶׁכֵּן בַּסֵּיפָא דְּהַכּוֹנֵס שֶׁהוּא רָשָׁע אִי מַכְנִיס לְאֵשֶׁת אָחִיו מְעֻבֶּרֶת, דְּרֻבָּן יוֹלְדוֹת וָלָד שֶׁל קַיָּמָא. נִמּוּקֵי יוֹסֵף:
(ב) אָסוּר. אַף עַל גַּב דִּתְנַן לְקַמָּן מִשְׁנָה י'‏ דְּלֹא תַחְלֹץ עַד שֶׁיְּהֵא לָהּ שְׁלֹשָׁה חֳדָשִׁים, וְהָכָא בְּתוֹךְ שְׁלֹשָׁה חֳדָשִׁים כֵּיוָן שֶׁנִּמְצֵאת מְעֻבֶּרֶת קָתָנֵי, מִכָּל מָקוֹם בְּדִיעֲבַד הֲוֵי חֲלִיצָה. תּוֹסָפוֹת:

=== 5 | chapter 5 ===
(א) יֵשׁ גֵּט כוּ'. מִדְּרַבָּנָן. וְטַעְמָא דְּגָזְרוּ מְפָרֵשׁ בַּגְּמָרָא:
(ב) לֹא קָאֵי אֶלָּא אַגֵּט וּמַאֲמָר, וְלֹא אַבְּעִילָה וַחֲלִיצָה, כְּמוֹ שֶׁאִיתָא בְּמִשְׁנָה ג':

(א) לֹא תִנָּשֵׂא. דְּאַף עַל גַּב דְּרֹב נָשִׁים מִתְעַבְּרוֹת וְיוֹלְדוֹת וְאִם כֵּן שֶׁיָּלְדָה צָרָתָהּ וָלָד שֶׁל קַיָּמָא מֻתֶּרֶת אִשָּׁה זוֹ לַשּׁוּק, מִכָּל מָקוֹם אִיכָּא מִעוּט דְּמַפִּילוֹת, וַחֲזָקָה שֶׁאֲסוּרָה לַשּׁוּק דִּבְחֶזְקַת יִבּוּם קַיְמָא כֵּיוָן דְּאִית לָהּ יָבָם, וּסְמֹךְ מִעוּטָא לַחֲזָקָה וַהֲוָה לֵיהּ פַּלְגָא וּפַלְגָא, דַּחֲזָקָה לֹא עָדִיף כִּי רֻבָּא, וְאֵין צָרִיךְ לוֹמַר דְּאַתְיָא כְּרַבִּי מֵאִיר דְּחַיִשׁ לְמִעוּטָא. נִמּוּקֵי יוֹסֵף. וְכֵן נִרְאֶה דַּעַת הָרַ"ב:
(ב) הִיא צָרָתָהּ. צָרָתָהּ מְיֻתָּר הוּא. וּמְפָרֵשׁ בַּגְּמָרָא הָא קָא מַשְׁמַע לָן לְהָא צָרָתָהּ הוּא דְּחָיְשִׁינַן, לְצָרָה אַחֲרִיתָא לֹא חָיְשִׁינַן. פֵּרֵשׁ רַשִׁ"י אֵינָהּ חוֹשֶׁשֶׁת שֶׁמָּא אִשָּׁה אַחֶרֶת נָשָׂא:
(ג) מְלֵאָה. דְּכֵיוָן שֶׁיָּצְאָה מְלֵאָה וְעָבַר זְמַן לֵדָתָהּ אֵין אוֹמְרִים שֶׁמָּא הִפִּילָה, דַּהֲוָה לֵיהּ כְּמִיתָה, וַאֲסִיקְנָא בְּפֶרֶק כָּל הַגֵּט דִּלְמָא מֵת לֹא חָיְשִׁינַן, וְכֵיוָן שֶׁנִּסְתַּלֵּק מִעוּט מַפִּילוֹת וַחֲזָקָה [דַּאֲסוּרָה לַשּׁוּק] לֹא נִשְׁאָר כָּאן אֶלָּא סָפֵק דְּמֶחֱצָה זְכָרִים וּמֶחֱצָה נְקֵבוֹת, וּלְפִיכָךְ חוֹשֶׁשֶׁת. נִמּוּקֵי יוֹסֵף:
