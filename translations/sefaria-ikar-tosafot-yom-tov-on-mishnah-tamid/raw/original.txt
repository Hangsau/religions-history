(א) שׁוֹמְרִים כוּ'. מִשּׁוּם דְּבָעֵי לְאוֹרוֹיֵי בְּמַעֲשֵׂה דְתָמִיד עֲבוֹדַת כֹּהֲנִים, נָקַט תְּחִלָּה שְׁמִירָתָן וּמְקוֹם שְׁכִיבָתָן עַד שֶׁהוּא מְסַדֵּר כָּל הַסֵּדֶר שֶׁתְּחִלָּה טוֹבְלִין כוּ'. (מְפָרֵשׁ הַמַּסֶּכְתָּא. וְאֵינוֹ פֵּרוּשׁ רַשִׁ"י):
(ב) וְאַף עַל גַּב דִּקְרָא קָאָמַר וְהַחוֹנִים וְגוֹ'‏ מֹשֶׁה וְאַהֲרֹן וּבָנָיו, הָכִי קָאָמַר וְהַחוֹנִים וְגוֹ'‏ מֹשֶׁה, וַהֲדַר אַהֲרֹן וּבָנָיו. מִדִּכְתִיב וְהַחוֹנִים וּכְתִיב שׁוֹמְרֵי, חוֹנִים לְחוּד וְשׁוֹמְרִים לְחוּד. גְּמָרָא:
(ג) הַנִּיצוֹץ. לֹא שָׁמַעְתִּי טַעַם לָמָּה נִקְרָא כֵן. מְפָרֵשׁ:
(ד) וַאֲפִלּוּ הָיוּ אוֹתָן עֲלִיּוֹת מִקְצָתָן בָּעֲזָרָה, יְכוֹלִין כֹּהֲנִים לֵישֵׁב כְּשֶׁהָיוּ עֲיֵפִים, דְּגַגִּים וַעֲלִיּוֹת לֹא נִתְקַדְּשׁוּ. וּבָעֲזָרָה אֵין אָדָם יָכוֹל לֵישֵׁב, דְּאֵין יְשִׁיבָה בָעֲזָרָה אֶלָּא לְמַלְכֵי בֵּית דָּוִד. מְפָרֵשׁ:
(ה) וְהֵם כֹּהֲנִים פְּחוּתִים מִשְּׁלֹשׁ עֶשְׂרֵה, וּלְפִי שֶׁעֲדַיִן לֹא הִגִּיעוּ לַעֲשׂוֹת עֲבוֹדָה עָבְדִינַן לְהוּ שׁוֹמְרִים, שֶׁאוֹתָן שֶׁיְּכוֹלִין לַעֲבֹד אֵינָן רוֹצִין לִשְׁמֹר. וְגַם אֵין מַנִּיחִין אוֹתָן לִשְׁמֹר. מְפָרֵשׁ:
(ו) וּבִלְשׁוֹן אַשְׁכְּנַז גיוועל"ב:
(ז) יְשֵׁנִים כוּ'. בְּחֶצְיָהּ שֶׁבָּחֹל:
(ח) בְּיָדָם. כְּלוֹמַר, בִּידֵי אוֹתָם הַכֹּהֲנִים הַיְשֵׁנִים לְשָׁם הָיוּ מֻנָּחוֹת כָּל מַפְתְּחוֹת הָעֲזָרָה. וּמְפָרֵשׁ בְּמַסֶּכֶת מִדּוֹת סוֹף פֶּרֶק קַמָּא הֵיכָן הָיְתָה הַנָּחָתָן, מָקוֹם הָיָה שָׁם אַמָּה עַל אַמָּה. מְפָרֵשׁ:
(ט) וּבַגְּמָרָא אָמְרִינַן, הָתָם קָרֵי לְהוּ רוֹבִים הָכָא קָרֵי לְהוּ פִרְחֵי כְהֻנָּה. אָמְרֵי, אִין, הָתָם דְּלֹא מְטוֹ לְמֶעֱבַד עֲבוֹדָה קָרֵי לְהוּ רוֹבִים, הָכָא דִמְטוֹ לְהוּ לְמֶעֱבַד עֲבוֹדָה קָרֵי לְהוּ פִּרְחֵי:
(י) אֲבָל לֵהָנוֹת נִתְּנוּ אֲפִלּוּ שֶׁלֹּא בִּשְׁעַת עֲבוֹדָה. כְּמוֹ שֶׁכָּתַב הָרַ"ב בְּרֵישׁ פֶּרֶק ז'‏ דְּיוֹמָא:
(יא) וְטָבַל. וְהָא דִתְנַן בְּמַסֶּכֶת מְגִלָּה פֶּרֶק ב'‏ מִשְׁנָה ד'‏ לֹא מַזִּין וְלֹא טוֹבְלִין אֶלָּא בַיּוֹם, הַיְנוּ בְּחַיָּבֵי טְבִילָה שֶׁצְּרִיכִין סְפִירָה כְּמוֹ טְמֵא מֵת אוֹ זָב אוֹ מְצֹרָע כוּ'. וְנִדָּה וְיוֹלֶדֶת טְבִילָתָן בְּלֵיל שְׁמִינִי. מְפָרֵשׁ:
(יב) עַיֵּן פֵּרוּשׁ הָרַ"ב בְּסוֹף פֶּרֶק קַמָּא דְּמִדּוֹת:

(א) הוֹאִיל וְרוֹצִין לְקַדֵּשׁ יְדֵיהֶם וְרַגְלֵיהֶם לְצֹרֶךְ עֲבוֹדָה רַשָּׁאִים לָלֶכֶת בֵּין הָאוּלָם לַמִּזְבֵּחַ. מְפָרֵשׁ:
(ב) הַמַּגְרֵפוֹת. יָעִים לְהָסִיר אֶת הַדֶּשֶׁן וּלְפַנּוֹתוֹ לַתַּפּוּחַ:
(ג) הַצִּנּוֹרוֹת. מַזְלְגוֹת לְהָסִיר אֶת הָאֵבָרִים. מְפָרֵשׁ:
(ד) וְעָלוּ כוּ'. וְקֹדֶם תְּרוּמַת הַדֶּשֶׁן לֹא הָיוּ יְכוֹלִים לַעֲשׂוֹת, לְפִי שֶׁצָּרִיךְ לְהָרִים תְּרוּמַת הַדֶּשֶׁן קֹדֶם שֶׁיְּפַנּוּ אֶת הַדֶּשֶׁן לַתַּפּוּחַ. מְפָרֵשׁ:
(ה) וְהַפְּדָרִים. הַחֲלָבִים. וְכֻלָּם נִקְרְאוּ כֵן עַל שֵׁם הַפֶּדֶר שֶׁבַּכָּתוּב:
(ו) סוֹדְרִין כוּ'. כְּדֵי שֶׁלֹּא יִפָּסְלוּ בְּלִינָה, שֶׁאֵין יְכוֹלִין לְהַקְטִירָן עַד [לְמָחָר] לְאַחַר שֶׁיַּקְטִירוּ אֵבְרֵי תָמִיד, וְאִם לֹא סִדְּרָן עַל הַמִּזְבֵּחַ הָיוּ נִפְסָלִים בְּלִינָה כוּ'. וְאָמְרִינַן בִּזְבָחִים דַּף פ"ז דְּאֵין לִינָה בַּמִּזְבֵּחַ, וְכֶבֶשׁ הֲרֵי הוּא כְמִזְבֵּחַ לְכָל דָּבָר. וּלְאַחַר שֶׁיְּסַדְּרוּ אֶת הַמַּעֲרָכוֹת יַחֲזִירוּ אֶת הָאֵבָרִים עַל הַמַּעֲרָכָה וְיָצִיתוּ בָהּ הָאוּר:

(א) מִזְבֵּחַ כוּ'. אַקְדְּמֵיהּ לְדִשּׁוּן הַמְּנוֹרָה, דְּכִי עָיֵל לַהֵיכָל בַּמִּזְבֵּחַ פָּגַע בְּרֵישָׁא, דְּתַנְיָא, שֻׁלְחָן בַּצָּפוֹן כוּ'‏ וּמְנוֹרָה בַּדָּרוֹם כוּ', מִזְבֵּחַ מְמֻצָּע וְעוֹמֵד בָּאֶמְצַע וּמָשׁוּךְ כְּלַפֵּי חוּץ קִמְעָא, דִּכְתִיב וְאֶת הַמְּנוֹרָה נֹכַח הַשֻּׁלְחָן, בָּעִינַן דַּחֲזוּ לַהֲדָדֵי. גְּמָרָא:
(ב) וְכֵן כָּתַב עוֹד לְהַלָּן וְהַשְּׁלִישִׁי מְדַשֵּׁן כוּ'‏ וּמַקְטִיר קְטֹרֶת. טָעוּת הוּא בְּיָדוֹ, דְּהָא תְנַן בְּפֶרֶק ה'‏ מִשְׁנָה ב'‏ חֲדָשִׁים לַקְּטֹרֶת. וְעוֹד, בְּרֵישׁ פֶּרֶק ו'‏ מִי שֶׁזָּכָה בְּדִשּׁוּן מִזְבֵּחַ הַפְּנִימִי כוּ'‏ מַקְדִּימִים כוּ'. וּלְשׁוֹן הָרַמְבַּ"ם, וְהַשְּׁלִישִׁי מֵרִים דִּשּׁוּן מִזְבֵּחַ הַפְּנִימִי לְפִי שֶׁבּוֹ מַקְטִירִין הַקְּטֹרֶת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ג) וְאֵין נִרְאֶה כֵן מִסּוֹף פֶּרֶק דִּלְקַמָּן. וְהַמְפָרְשִׁים פֵּרְשׁוּ שָׁם הַזָּנָב. אֲבָל הֶעָרוּךְ כָּתַב שֶׁהוּא הֶעָצֶה כְּדִכְתִיב לְעֻמַּת הֶעָצֶה:
(ד) וּבַפֶּרֶק דִּלְקַמָּן מִשְׁנָה ג'‏ תְּנַן דְּכָבֵד תָּלוּי בַּדֹּפֶן הַיְמָנִית. וּלְשׁוֹן הַמְפָרֵשׁ, וּבוֹ מְחֻבָּר הַקָּנֶה וְהַלֵּב:

=== 4 | chapter 4 ===
(א) אוֹחֲזִים בּוֹ. בִּשְׁעַת שְׁחִיטָה וּמְסַיְּעִים אוֹתוֹ. הַמְפָרֵשׁ:
(ב) לַמַּעֲרָב. שֶׁצָּרִיךְ שֶׁיְּהֵא בֵּית שְׁחִיטָתוֹ כְּלַפֵּי מַעֲרָב, דִּכְתִיב וְשָׁחַט אוֹתוֹ וְגוֹ'‏ לִפְנֵי ה'. מְפָרֵשׁ. וּמַה שֶּׁכָּתַב הָרַ"ב שֶׁאִם יַרְבִּיץ, אַרֹאשׁוֹ לַדָּרוֹם קָאֵי:
(ג) כְּמוֹ שֶׁנֶּאֱמַר וּבִתְחַפְנְחֵס חָשַׁךְ הַיּוֹם [יְחֶזְקֵאל ל]. הָרַמְבַּ"ם. וּבַבְּרַיְתָא, אוֹ אֵינוֹ אֶלָּא חוֹבַת הַיּוֹם [כְּלוֹמַר כָּל יוֹם וְיוֹם תַּקְרִיבוּ שְׁנַיִם]. כְּשֶׁהוּא אוֹמֵר אֶת הַכֶּבֶשׂ אֶחָד תַּעֲשֶׂה בַבֹּקֶר וְאֶת הַכֶּבֶשׂ הַשֵּׁנִי תַּעֲשֶׂה בֵּין הָעַרְבַּיִם, הֲרֵי חוֹבַת הַיּוֹם אָמוּר:
(ד) כְּלוֹמַר, וְעַל טַבַּעַת שְׁנִיָּה לָאו בְּטַבַּעַת שְׁנִיָּה שֶׁל סֵדֶר רִאשׁוֹן שֶׁהִיא אֵצֶל הַמִּזְבֵּחַ אֶלָּא בְטַבַּעַת הָרִאשׁוֹנָה שֶׁהִיא אֵצֶל הַקֶּרֶן בַּסֵּדֶר הַשֵּׁנִי הַמְשׁוּכָה לְצַד צָפוֹן, וּפְנֵי הַשּׁוֹחֵט וְצַוַּאר הַבְּהֵמָה לְצַד מַעֲרָב צָפוֹן בַּאֲלַכְסוֹן כְּדֵי שֶׁלֹּא יְעַכֵּב אוֹר הַשֶּׁמֶשׁ לָבֹא כוּ'. הַמְפָרֵשׁ. וְקָשֶׁה כוּ'. וְהַתּוֹסָפוֹת כָּתְבוּ הַטַּעַם שֶׁהִרְחִיקוֹ מִן הַמִּזְבֵּחַ, שֶׁמָּא יַטִּיל גְּלָלִים. וְהַבַּעַל הַמָּאוֹר כָּתַב, לְפִי שֶׁנִּשְׁחָט נֶגֶד יוֹם, וְתַנְיָא מֵעוֹלָם לֹא יָצְאָה חַמָּה מִקֶּרֶן מִזְרָחִית צְפוֹנִית וְלֹא שָׁקְעָה בְּקֶרֶן מַעֲרָבִית צְפוֹנִית, וּכְמוֹ שֶׁהַחַמָּה מֻפְלֶגֶת מִן הַקְּרָנוֹת שֶׁל צְפוֹן הָעוֹלָם כָּךְ הָיָה הַתָּמִיד מֻפְלָג בִּשְׁחִיטָתוֹ מִן הַקְּרָנוֹת שֶׁל צְפוֹן הַמִּזְבֵּחַ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) וְזֶה לִשְׁאָר קָרְבָּנוֹת, חוּץ מִתְּמִידִים דִּכְתִיב בְּהוּ לַיּוֹם:
(ו) אַרְבַּע רַגְלַיִם יַחַד. מְפָרֵשׁ. וְדַעַת הָרַמְבַּ"ם שֶׁאוֹחֲזִין יָדָיו וְרַגְלָיו בִּידֵיהֶן בִּלְבָד:
(ז) הַמְקַבֵּל. שֶׁעוֹמֵד כְּנֶגֶד הַשּׁוֹחֵט בַּאֲלַכְסוֹן, וּמְצַדֵּד עַצְמוֹ כְּדֵי שֶׁיְּהֵא פָנָיו כְּלַפֵּי דָרוֹם. הַמְפָרֵשׁ:
(ח) מְפֹרָשׁ בְּפֵרוּשׁ הָרַ"ב מִשְׁנָה ד'‏ פֶּרֶק ה'‏ דִּזְבָחִים. וּבַגְּמָרָא פָרֵיךְ דְּנֵיתַב בָּרֵישָׁא מַעֲרָבִית דְּרוֹמִית וַהֲדַר מִזְרָחִית צְפוֹנִית, וּמְשָׁנֵי, כֵּיוָן דְּאָמַר מַר כָּל פִּנּוֹת כוּ'‏ דֶּרֶךְ יָמִין, בָּרֵישָׁא בְּהַהִיא פָּגַע. וּפֵרֵשׁ רַשִׁ"י, כְּשֶׁעוֹלֶה בַכֶּבֶשׁ וּפוֹנֶה לְיָמִין פּוֹגֵעַ בְּמִזְרָחִית צְפוֹנִית קֹדֶם מַעֲרָבִית דְּרוֹמִית, וְאַף עַל גַּב דְּהָכָא עַל הָרִצְפָּה עוֹמֵד וְהַכְּלִי בְיָדוֹ וְזוֹרֵק לְמֵרָחוֹק, אֲפִלּוּ הָכִי כֵּיוָן שֶׁבְּחַטָּאת שֶׁנִּתְּנָה בְּעֶלְיוֹנָה שֶׁל קֶרֶן צָרִיךְ לַעֲלוֹת בַּכֶּבֶשׁ דֶּרֶךְ הַדָּרוֹם וּפוֹנֶה לַמִּזְרָח, הָכִי נַמִּי בְּכָל הַדָּמִים אַף עַל פִּי שֶׁנִּתָּנִין לְמַטָּה מַתְחִיל לְהַקִּיף מִדֶּרֶךְ הַדָּרוֹם שֶׁהוּא פָנָיו שֶׁל מִזְבֵּחַ וּמִשָּׁם פּוֹנֶה לְיָמִין:
(ט) דְּרוֹמִית. בַּגְּמָרָא, מִדִּכְתִיב בִּשְׂעִיר נָשִׂיא אֶל יְסוֹד מִזְבַּח הָעוֹלָה, שֶׁאֵין תַּלְמוּד לוֹמַר אֶלָּא לוֹמַר תֵּן יְסוֹד בְּמִזְבְּחָהּ שֶׁל עוֹלָה [וְעַיֵּן פ"ז דִּזְבָחִים אוֹת ה'‏ וּפ"ה אוֹת י"ט]. וּפֵרְשׁוּ הַתּוֹסָפוֹת דְּתַרְתֵּי יָלְפִינַן מִנַּהּ, לִתְחִלַּת מַתַּן דָּמִים שֶׁצְּרִיכִים יְסוֹד, וּמִדְּלֹא כְתִיב אֶל יְסוֹד הָעוֹלָה וּכְתִיב יְסוֹד מִזְבַּח, לְמִדְרַשׁ נַמִּי לַשְּׁיָרִים דְּבָעִינַן יְסוֹד. וְדָרְשִׁינַן תַּרְתֵּי, דְּקֹדֶם יֵשׁ לְהַעֲמִיד פְּשָׁטֵיהּ דִּקְרָא לִתְחִלַּת מַתָּנוֹת, וְהַיִּתּוּר לַשְּׁיָרִים:

(א) עִקַּר הַדָּת וְרֵאשִׁיתוֹ. הָרַמְבַּ"ם:
(ב) אֶת הָעָם. עִם הָעָם. וַעֲבוֹדָה, בִּשְׁבִיל עֲבוֹדָה שֶׁעָשׂוּ הָיוּ מְבָרְכִין אַחֲרֶיהָ רְצֵה כוּ'‏ בָּרוּךְ הַמְקַבֵּל עֲבוֹדַת עַמּוֹ יִשְׂרָאֵל בְּרָצוֹן. אִי נַמִּי שֶׁאוֹתְךָ לְבַדְּךָ בְּיִרְאָה נַעֲבֹד. וּבִרְכַּת כֹּהֲנִים, לְבָרֵךְ אֶת הָעָם. רַשִׁ"י:
(ג) וְהָרַמְבַּ"ם בִּמְקוֹם בִּרְכַּת כֹּהֲנִים כָּתַב שִׂים שָׁלוֹם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(א) בְּדִשּׁוּן כוּ'. וּכְפִי הַסֵּדֶר הַשָּׁנוּי בְּמִשְׁנָתֵנוּ שְׁנֵיהֶם הִקְדִּימוּ לִכָּנֵס וְלָצֵאת קֹדֶם שֶׁיַּקְטִיר הַמַּקְטִיר. וּכְבָר כָּתַבְתִּי בְּסוֹף פֶּרֶק ג'‏ דְּאָתְיָא כְּאַבָּא שָׁאוּל. אֲבָל לְרַבָּנָן, לְאַחַר שֶׁהִקְטִיר נִכְנָס זֶה שֶׁזָּכָה בְּדִשּׁוּן הַמְּנוֹרָה וּמֵיטִיב שְׁתֵּי הַנֵּרוֹת. וְכֵן כָּתַב הָרַמְבַּ"ם, וּמְסַיֵּם וְיוֹצֵא זֶה שֶׁהִקְטִיר עִם מְדַשֵּׁן הַמְּנוֹרָה:

(ג) לָאו דַּוְקָא, דְּהָא בְּסוֹף פֶּרֶק ג'‏ תְּנַן וּמַדְלִיקָן מִן הַדּוֹלְקִים, וּפֵרֵשׁ הָרַ"ב וְאִם אֵין הַנֵּרוֹת דּוֹלְקִין מַדְלִיקָן מִמִּזְבַּח הָעוֹלָה, שְׁמַע מִנַּהּ דְּכִי יֵשׁ דּוֹלְקִין שַׁפִּיר טְפֵי לְמִדְלְקִינְהוּ מִנַּיְהוּ. וְלֹא בָא לוֹמַר אֶלָּא שֶׁאֵין מַדְלִיקִין בְּשׁוּם אֵשׁ אַחֵר שֶׁבַּמִּקְדָּשׁ, וְלֹא לְמַעוֹטֵי נֵרוֹת דִּמְנוֹרָה אִם יֵשׁ בָּהֶם דּוֹלְקִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(א) דְּהָא דִכְתִיב וְאַל יָבֹא בְּכָל עֵת אֶל הַקֹּדֶשׁ [וַיִּקְרָא טז], דָּרְשִׁינַן בִּמְנָחוֹת דְּקֹדֶשׁ הוּא הַהֵיכָל בְּאַל יָבֹא, וְאֶל מִבֵּית לַפָּרֹכֶת כוּ'‏ בְּמִיתָה. וּמִלְּשׁוֹן הָרַ"ב מַשְׁמַע דְּכָל הִשְׁתַּחֲוָיָה שָׁרְיָא וְלֹא הַוְיָא שֶׁלֹּא לְצֹרֶךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(ג) יוֹצֵא. פֵּרוּשׁ, יוֹצֵא מִמְּקוֹמוֹ בַּאֲשֶׁר הוּא שָׁם וְרוֹצֶה לִכָּנֵס לַהֵיכָל, קָדַם הוּא וְהִגְבִּיהַּ כוּ'. אֲבָל הָרַמְבַּ"ם הֶעְתִּיק כוּ'‏ וְיִכָּנֵס לַהֵיכָל וְיִשְׁתַּחֲוֶה וְכֵיוָן שֶׁיִּשְׁמַע הַסְּגָן קוֹל רַגְלָיו שֶׁל כֹּהֵן גָּדוֹל שֶׁהוּא יוֹצֵא מַגְבִּיהַּ לוֹ אֶת הַפָּרֹכֶת. נִרְאֶה דְּשֶׁהוּא יוֹצֵא הַיְנוּ מֵהַהֵיכָל כוּ'. וְקָשֶׁה, דְּלָמָּה לֹא מַגְבִּיהַּ נַמִּי בִּכְנִיסָתוֹ לְשָׁם. וְנִרְאֶה לִי שֶׁהָיָה הַפָּרֹכֶת כָּל הַיּוֹם מֻגְבָּהּ וְעוֹמֵד עַל יְדֵי קְשָׁרִים וִיתֵרוֹת לִכָּנֵס וְלָצֵאת הַכֹּהֲנִים הַצְּרִיכִים לַעֲבוֹדַת מְנוֹרָה וּמִזְבֵּחַ פְּנִימִי, אֶלָּא שֶׁכְּשֶׁנִּכְנָס הַכֹּהֵן גָּדוֹל הָיוּ מְסִירִים הַפָּרֹכֶת מֵהַקְּשָׁרִים וִיתֵרוֹת כְּדֵי שֶׁיְּהֵא נִתְלֶה עַל עָמְדוֹ, וְכָבוֹד גָּדוֹל נַעֲשָׂה לוֹ לַכֹּהֵן גָּדוֹל בָּזֶה לְהַצְנִיעַ אַחֲרָיו הַקֹּדֶשׁ כְּאִלּוּ מִתְיַחֵד שָׁם עִם הַשּׁוֹכֵן בַּבַּיִת בָּרוּךְ הוּא, וּלְפִיכָךְ אַחַר שֶׁחָזַר וְרָצָה לָצֵאת הֻצְרָךְ לְהַגְבִּיהַּ, וְלֹא כֵן בִּשְׁעַת כְּנִיסָתוֹ. כֵּן נִרְאֶה לִי:
