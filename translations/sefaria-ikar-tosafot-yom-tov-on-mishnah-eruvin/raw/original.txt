(א) מָבוֹי. פֵּרוּשׁ הָר"ר יוֹנָתָן הַכֹּהֵן שֶׁהוּא מִלְּשׁוֹן מְבוֹא הָעִיר [שׁוֹפְטִים א]:
(ב) כְּלוֹמַר כְּדִין הַכַּרְמְלִית שֶׁהִזְכִּיר בְּרֵישׁ פֶּרֶק י"א דְּשַׁבָּת, אֲבָל לֹא שֶׁיִּהְיֶה דִּינוֹ כִּרְשׁוּת הַיָּחִיד גָּמוּר לְחַיֵּב הַזּוֹרֵק מֵרְשׁוּת הָרַבִּים לְתוֹכוֹ, וּכְמוֹ שֶׁאִיתָא בַּמִּשְׁנָה דִּלְקַמָּן דַּאֲפִלּוּ הֻכְשַׁר בְּקוֹרָה לֹא הָוֵי כִּרְשׁוּת הַיָּחִיד גָּמוּר:
(ג) וְלֹא יוֹעִיל בָּהּ הֶכֵּירָא דְּקוֹרָה. וּלְרַבִּי יְהוּדָה דְּקוֹרָה מִשּׁוּם מְחִצָּה, אֵין צָרִיךְ לְמַעֵט, כִּדְאִיתָא בַּבְּרַיְתָא. הָרִי"כ:
(ד) שֶׁהוּא רָחָב. וַאֲפִלּוּ הָיָה גָּבוֹהַּ יוֹתֵר מֵעֶשְׂרִים אַמָּה. הָרַמְבַּ"ם:

=== 2 | chapter 2 ===
(א) גָּבְהָן עֲשָׂרָה כוּ'. גָּבְהָן עֲשָׂרָה שֶׁאֵין נִקְרֵאת מְחִצָּה בְּפָחוֹת מֵעֲשָׂרָה, וְרָחְבָּן שִׁשָּׁה טְפָחִים דְּזֶהוּ אַמָּה, שֶׁאַף עַל פִּי שֶׁבִּמְחִצַּת חָצֵר סַגִּי בְּפַס אַרְבָּעָה, לֹא דָּמִי, דְּהָתָם בְּחָצֵר אִיכָּא שָׁלֹשׁ מְחִצּוֹת שְׁלֵמוֹת אֲבָל בְּפַסֵּי בֵּירָאוֹת שֶׁרֻבּוֹ פָּרוּץ מִכָּל צַד מַצְרִיכִין פַּס שֶׁל שִׁשָּׁה טְפָחִים. רִי"כ:
(ב) קְשׁוּרוֹת. וְאַף עַל פִּי שֶׁאֵין הַפָּרוֹת מוֹשְׁכוֹת בָּעֹל בְּשַׁבָּת אֲפִלּוּ הָכִי מְשַׁעֲרִין בְּהוּ, לְפִי שֶׁגַּם בִּימוֹת הַחוֹל נִכְנָסוֹת שָׁם לִשְׁתּוֹת, וּפְעָמִים שֶׁכְּשֶׁהַצֶּמֶד הָאֶחָד נִכְנָס לִשְׁתּוֹת מוֹצֵא הַשֵּׁנִי שֶׁשָּׁתָה מִשָּׁם וְרוֹצֶה לָצֵאת, וְאִם יִהְיוּ בֵּין הַדְּיוּמְדִין פָּחוֹת מִזֶּה הַשִּׁעוּר יָבֹאוּ הַפָּרוֹת וְיַפִּילוּ הַדְּיוּמְדִין לָאָרֶץ. הָרִי"כ:
(ג) הָכִי תָּנוּ רַבָּנָן. וּפֵרֵשׁ רַשִׁ"י וְלֹא פָרָה נִכְנֶסֶת וּפָרָה יוֹצֵאת, דְּאִם כֵּן לֹא הֲוָה קְשׁוּרוֹת:
(ד) רְצוֹנוֹ לוֹמַר דְּאִלּוּ עֶשֶׂר אַמּוֹת לְרַבִּי מֵאִיר וְכֵן הַשְּׁלֹשׁ עֶשְׂרֵה וּשְׁלִישׁ לְרַבִּי יְהוּדָה לֹא בָּעִינַן מְצֻמְצָמוֹת. וְכֵן הוּא בְּרַשִׁ"י וְרַמְבַּ"ם:

=== 3 | chapter 3 ===
(א) וְאַף עַל גַּב דִּתְחוּמִין חֲמִירֵי, שֶׁיֵּשׁ לָהֶם סֶמֶךְ מִן הַמִּקְרָא, הֵקֵלוּ בִּתְחוּמִין שֶׁאֵינָהּ אֶלָּא לִדְבַר מִצְוָה. תּוֹסָפוֹת. וְהָא דְּקָרֵי לֵיהּ עֵרוּב, לְפִי שֶׁהוּא מְעָרֵב הַתְּחוּמִין שֶׁלֹּא הָיָה יָכוֹל לֵילֵךְ בּוֹ תְּחִלָּה עִם הַתְּחוּם שֶׁהָיָה יָכוֹל לֵילֵךְ בּוֹ. הַמַּגִּיד מִשְׁנֶה:
(ב) חוּץ כוּ'. דְּטַעֲמָא דְּעֵרוּבִין מִשּׁוּם דְּדַעְתּוֹ וְדִירָתוֹ בִּמְקוֹם מְזוֹנוֹתָיו הוּא, וְנַעֲשֶׂה הַמְעָרֵב כְּאִלּוּ אוֹכֵל וְשׁוֹבֵת שָׁם בַּמָּקוֹם שֶׁהִנִּיחַ הָעֵרוּב, וּמִשָּׁם יֵשׁ לוֹ אַלְפַּיִם אַמָּה לְכָל רוּחַ, וְהִלְכָּךְ דָּבָר [מָזוֹן] בָּעִינַן. רַשִׁ"י:
(ג) דְּכֻלָּהּ סְפֵקָא מִשּׁוּם עֲצָמוֹת הוּא, שֶׁהַבָּשָׂר כְּבָר נִרְקַב כַמָּה שָׁנִים, וַעֲצָמוֹת בְּלֹא בָּשָׂר אֵין מְטַמְּאִין בְּאֹהֶל אֶלָּא בְּמַגָּע, וּכְשֶׁמְּנַפֵּחַ לִפְנֵי רַגְלָיו אִם יֵשׁ עֶצֶם גָּדוֹל נִרְאֶה הוּא וְיִזָּהֵר שֶׁלֹּא יִגַּע בּוֹ רַגְלָיו, וְאִם עֶצֶם קָטָן הוּא נִדְחֶה בַּנְּפִיחָה וְאֵינוֹ מְסִיטוֹ בְּרַגְלָיו, וְאִם מְכֻסֶּה בְּעָפָר אֵינוֹ מְטַמֵּא, דְּהָא עֶצֶם כִּשְׂעוֹרָה אֵינוֹ מְטַמֵּא אֶלָּא בְּמַגָּע. רִי"כ:
(ד) בַּעֲגָלוֹת, מִבְּעוֹד יוֹם (רַשִׁ"י). וְעַיֵּן רֵישׁ פֶּרֶק ח'‏ דְּאָהֳלוֹת שֶׁצָּרִיךְ שֶׁמַּחְזִיקִים אַרְבָּעִים סְאָה כוּ', וּפָרֵיךְ בַּגְּמָרָא הֵיכִי מַיְתֵי לַהּ כוּ', וּמִדְּנָקַט רַבִּי יְהוּדָה וְלֶאֱכֹל שְׁמַע מִנָּהּ דְּרָאוּי לְאָכְלוֹ בָּעֵי, וּמְשָׁנֵי בִּפְשׁוּטֵי כְּלֵי עֵץ שֶׁאֵין בְּרָחְבָּן טֶפַח, דְּלֹא מְקַבְּלֵי טֻמְאָה וְאֵין מְבִיאִין אֶת הַטֻּמְאָה:
(ה) וְכָל תַּשְׁמִישֵׁי הַמֵּת אֲסוּרִין בַּהֲנָאָה [דְּנַפְקָא לָן] מִוַּתָּמָת שָׁם מִרְיָם, וְגָמַר שָׁם שָׁם מֵעֶגְלָה עֲרוּפָה. רַשִׁ"י:

=== 4 | chapter 4 ===
(א) נְתָנוּהוּ. לְפִי שֶׁדֶּרֶךְ הוּא כְּשֶׁמּוֹלִיכִין הָאָדָם מוֹלִיכִין אוֹתוֹ אֶל עִיר, הִלְכָּךְ קָתָנֵי הוֹלִיכוּהוּ גַּבֵּי עִיר. אֶלָּא שֶׁאִם אֵרַע לִפְעָמִים שֶׁלֹּא יוּכְלוּ לָבֹא אֶל עִיר אוֹ מִפְּנֵי סִבָּה וּמִקְרֶה שֶׁיִּתְּנוּהוּ לְמָקוֹם אַחֵר הַמִּשְׁתַּמֵּר כְּמוֹ בְּדִיר אוֹ סַהַר, לְכָךְ תָּנֵי גַּבַּיְהוּ נְתָנוּהוּ:
(ב) וְאֵין הָכִי נַמִּי דְּבָעִיר נַמִּי גָּזְרוּ דְּהָא וַדַּאי מַשְׁמַע דְּאַכָּל מַה שֶּׁהֻזְכַּר בְּדִבְרֵי רַבָּן גַּמְלִיאֵל וְרַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה קָמִפְלְגֵי, וְנָקְטֵי חֲדָא מִתַּרְתֵּי בָּבֵי דְּתַנָּא קַמָּא, וְהוּא הַדִּין לְאִידָךְ. תּוֹסְפוֹת יוֹם טוֹב:
(ג) לְפִי שֶׁאֵינוֹ דּוֹמֶה לְדִיר וְסַהַר אֶלָּא כְּשֶׁהִיא קְבוּעָה בַּנָּמָל כְּמוֹ שֶׁכָּתַב הָרַמְבַּ"ם. אִי נַמִּי כְּפֵרוּשׁ רַשִׁ"י דִּכְשֶׁהַסְּפִינָה מְהַלֶּכֶת כְּשֶׁעוֹקֵר רַגְלוֹ קֹדֶם שֶׁיַּנִּיחֵהוּ מוֹצִיאָתוֹ הַסְּפִינָה מֵאַרְבַּע אַמּוֹת שֶׁלּוֹ וְנִכְנָס בְּאַרְבַּע אַמּוֹת אֲחֵרוֹת עַל כָּרְחוֹ, וְהָוֵי כְּמוֹ שֶׁהוֹצִיאוּהוּ לִסְטִים וְנוֹתְנִין אוֹתוֹ תָדִיר בְּאַרְבַּע אַמּוֹת אֲחֵרוֹת, שֶׁנּוֹתְנִין לוֹ אַרְבַּע אַמּוֹת, וְכֵן לְעוֹלָם:
(ד) [וְטַעֲמָא] כִּדְאִיתָא בַּגְּמָרָא דִּלְמָא עָמְדָה פִּתְאוֹם וְלָאו אַדַּעְתַּיְהוּ. וּלְטַעֲמָא דְּשָׁבְתוּ בַּאֲוִיר כוּ'‏ נִרְאֶה דְּהוֹאִיל וְיָצְאוּ חוּץ לַתְּחוּם רָצוּ לְהַחְמִיר, שֶׁלֹּא תִטְעֶה וְלֹא תְחַלֵּק בֵּין יוֹצֵא לְחוּץ לַתְּחוּם וְשָׁבַת בַּמְּחִצּוֹת הָהֵן מִבְּעוֹד יוֹם כְּמוֹ בִּסְפִינָה לְהֵיכָא דְּלֹא שָׁבַת כְּמוֹ בְּדִיר וְסַהַר וְתַתִּיר הַכֹּל:

=== 5 | chapter 5 ===
(א) וְאַכֻּלְּהוּ קָאֵי, וְלָאו דַּוְקָא בַּקְּרָנוֹת שֶׁהֵן קֶרֶן זָוִית, אֶלָּא הוּא הַדִּין בָּאֶמְצַע:
(ב) הַיְנוּ אַרְבָּעָה טְפָחִים, כְּמוֹ שֶׁאִיתָא בְּפֶרֶק ב'‏ מִשְׁנָה ה':
(ג) וּגְשָׁרִים. שֶׁהֵן מְכֻסִּין מִלְּמַעְלָה. וְיֵשׁ לוֹ כְּתָלִים מֵאַרְבַּע רוּחוֹתָיו, אֶלָּא שֶׁהַכִּפָּה חֲלוּלָה, וְיֵשׁ בַּכְּתָלִים פִּלּוּשׁ שֶׁהַמַּיִם יַעַבְרוּ תַּחְתָּיו אוֹ בְּנֵי אָדָם, וְיֵשׁ תַּחְתָּיו דִּירָה לַמּוֹכֵס. הָרִי"כ:
(ד) וְקָשֶׁה [לְגִרְסָתוֹ דְגָרַס אוֹתָן], הָא תָּנָא לֵיהּ פֶּרֶק ד'‏ מִשְׁנָה ח', וְנֻסָּח אַחֵר [גָּרַס] אוֹתָהּ, וּפֵרֵשׁ הָרַמְבַּ"ם רְצוֹנוֹ לוֹמַר כִּי כְּשֶׁתִּהְיֶה הַמְּדִינָה עֲגֻלָּה שֶׁיַּעֲשׂוּ בּוֹ מְרֻבָּע דָּבוּק לָהּ וְאָז יִמְדְּדוּ חוּץ לַמְרֻבָּע אַלְפַּיִם אַמָּה לְכָל רוּחַ וּמַרְוִיחִים אַנְשֵׁי הַמְּדִינָה אוֹתָן הַתּוֹסָפוֹת שֶׁיּוֹסִיפוּ הַמְרֻבָּע עַל הָעֲגֻלָּה שֶׁבְּתוֹכוֹ:
(ו) טַבְלָא מְרֻבַּעַת. בְּעִיר מְרֻבַּעַת עָסְקִינַן, שֶׁאֵין בָּהּ אֶלָּא אַלְפַּיִם עַל אַלְפַּיִם, דַּהֲווּ נַמִּי תְּחוּמִין מְרֻבָּעִין, וְאִם אֵינָהּ מְרֻבַּעַת כְּגוֹן שֶׁהִיא יְתֵרָה עַל אַלְפַּיִם לְאָרְכָּהּ אוֹ פְּחוּתָה מֵאַלְפַּיִם, לֹא הָווּ מְרֻבָּעִים, דִּתְחוּמִין מְרֻבָּעִים לָאו דַּוְקָא שֶׁיְּהֵא אָרְכָּן כְּרָחְבָּן, שֶׁהֲרֵי רָחְבָּן לְמִדַּת הָעִיר הִיא כוּ', אֶלָּא מְרֻבָּעִים דְּקָתָנֵי לִמְעוּטֵי עֲגֻלִּים. רַשִׁ"י:

=== 10 | chapter 10 ===
(א) רְצוֹנוֹ לוֹמַר וְיֵשׁ לָחוּשׁ שֶׁיִּתְבַּזּוּ, וּכְמוֹ שֶׁכָּתַב [לְקַמָּן] בְּדִבּוּר הַמַּתְחִיל בַּיְשָׁנוֹת, אֲבָל מֵחֲשַׁשׁ גַּנָּבֵי לֹא הִתִּירוּ:
(ב) וְכֵן כָּתְבוּ הַפּוֹסְקִים, וְאִם כֵּן סְבִירָא לְהוּ דְּשַׁבָּת זְמַן תְּפִלִּין הִיא, וְתֵימַהּ דִּבְמִשְׁנָה ג'‏ פֶּרֶק ג'‏ דִּבְרָכוֹת מְפָרֵשׁ הָרַ"ב דְּלָאו זְמַן תְּפִלִּין הוּא, וְכֵן הַסְכָּמַת כָּל הַפּוֹסְקִים, וְצָרִיךְ עִיּוּן, תּוֹסְפוֹת יוֹם טוֹב. וּמַהַרְשָׁ"א מְיַשֵּׁב עַל זֶה דַּאֲפִלּוּ לְמַאן דְּאָמַר לָאו זְמַן תְּפִלִּין הוּא אָסוּר לִלְבֹּשׁ יוֹתֵר מִזּוּג אֶחָד, דְּלֹא שָׁרֵי לֵיהּ יוֹתֵר מִמַּלְבּוּשׁוֹ בַּחֹל. וְהָא דְּאוֹקִימְנָא כְּמַאן דְּאָמַר זְמַן כוּ'‏ מִשּׁוּם דִּסְתָמָא דְּמַתְנִיתִין כְּרַבִּי מֵאִיר וְרַבִּי מֵאִיר סְבִירָא לֵיהּ זְמַן תְּפִלִּין הוּא. וְהַמָּגֵן אַבְרָהָם תֵּרֵץ כֵּיוָן דַּאֲנַן לֹא בְּקִיאִין לְכַוֵּן מְקוֹם הֲנָחָתָן וְאִם יָנִיחַ שֶׁלֹּא בִּמְקוֹמוֹ יִהְיֶה מַשְּׂאוֹי, לָכֵן לֹא יָבִיא אֶלָּא זוּג זוּג:
(ג) וְאַף עַל גַּב דְּהַשְׁתָּא אֵין בּוֹ קֶשֶׁר רַשַּׁאי לְלָבְשׁוֹ, דַּאֲפִלּוּ הָכִי הָוֵי דֶּרֶךְ מַלְבּוּשׁ, אִי נַמִּי דְּעָנְבוֹ. תּוֹסָפוֹת.:
(ד) דְּאִסּוּר נַמִּי אִיכָּא:
(ה) אַכֻּלָּהּ מַתְנִיתִין קָאֵי, בֵּין מְעַט בֵּין צְבָתִים, מְכַסָּן בְּטַלִּיתוֹ שֶׁלֹּא יֻטְּלוּ כָּל כָּךְ בְּבִזָּיוֹן, וּמַנִּיחָן שָׁם וְהוֹלֵךְ וְלֹא יְזִיזֵם מִשָּׁם שֶׁמָּא יַרְגִּישׁוּ בּוֹ וְיִתְחַיֵּב לַמַּלְכוּת. רַשִׁ"י:
(ו) וְנִרְאֶה לִי דְּרַבֵּנוּ הַקָּדוֹשׁ הֻכְרַח לְהַחְסִיר לְמִשְׁנָתֵנוּ, דְּהָא חַכְמֵי הַמִּשְׁנָה לֹא רָצוּ לְגַלּוֹת הָא דְּפָחוֹת פָּחוֹת מֵאַרְבַּע אַמּוֹת, כְּמוֹ שֶׁכָּתַבְנוּ בְּרֵישׁ פֶּרֶק כ"ד דְּשַׁבָּת:
