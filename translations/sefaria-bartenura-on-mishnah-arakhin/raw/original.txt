הַכֹּל מַעֲרִיכִין. הַכֹּל לַאֲתוֹיֵי בֶּן שְׁלֹשׁ עֶשְׂרֵה שָׁנָה וְיוֹם אֶחָד וְלֹא הֵבִיא שְׁתֵּי שְׂעָרוֹת, שֶׁהוּא מַעֲרִיךְ. וְהוּא מֻפְלָא הַסָּמוּךְ לְאִישׁ. וּמֻפְלָא אִקְּרֵי לֵיהּ לְפִי שֶׁבּוֹדְקִים אוֹתוֹ אִם יוֹדֵעַ לְהַפְלוֹת וּלְפָרֵשׁ לְשֵׁם מִי נָדַר וּלְשֵׁם מִי הִקְדִּישׁ. אַף עַל גַּב דְּאִישׁ כִּי יַפְלִיא כְתִיב (ויקרא כז), כְּשֶׁהוּא סָמוּךְ לְפִרְקוֹ קָרוּי אִישׁ לְעִנְיָן זֶה. וְהַכֹּל דְּקָאֵי נַמִּי עַל נֶעֱרָכִים, לַאֲתוֹיֵי מְנֻוָּל וּמֻכֶּה שְׁחִין, שֶׁאֵין שָׁוִין דָּמִים, דְּסָלְקָא דַעְתָּךְ אֲמֵינָא נֶדֶר בְּעֶרְכְּךָ כְּתִיב (שם), כָּל שֶׁיֶּשְׁנוֹ בְדָמִים יֶשְׁנוֹ בַעֲרָכִין, וְכָל שֶׁאֵינוֹ בְדָמִים אֵינוֹ בַעֲרָכִין, קָמַשְׁמַע לָן. וְהַכֹּל דְּקָאֵי עַל נוֹדְרִים, מִשּׁוּם נִדָּרִים נָקַט לֵיהּ. וְנִדָּרִים גּוּפֵיהּ אִצְטְרִיךְ, לַאֲתוֹיֵי פָּחוּת מִבֶּן חֹדֶשׁ שֶׁהוּא נִדָּר אַף עַל פִּי שֶׁאֵינוֹ נֶעֱרָךְ:
מַעֲרִיכִין. אִם אָמַר עַל אֶחָד, עֵרֶךְ פְּלוֹנִי עָלָי, נוֹתֵן עֵרֶךְ אוֹתוֹ פְּלוֹנִי כְּפִי שָׁנָיו כְּמוֹ שֶׁקָּצוּב בְּפָרָשַׁת עֲרָכִין, שֶׁהָעֵרֶךְ נוֹתֵן לְפִי שָׁנִים שֶׁל נֶעֱרָכִין:
וְנֶעֱרָכִין. אִם אָמַר אַחֵר עָלָיו עֵרֶךְ פְּלוֹנִי זֶה עָלָי, אוֹ אָמַר הוּא עַצְמוֹ עֶרְכִּי עָלָי:
נוֹדְרִים. אָמַר דְּמֵי פְּלוֹנִי עָלָי, צָרִיךְ שֶׁיִּתֵּן כְּמוֹ שֶׁהוּא יָפֶה לִמָּכֵר בַּשּׁוּק:
וְנִדָּרִין. אָמַר אַחֵר עָלָיו דְּמֵי פְּלוֹנִי עָלָי, אוֹ אָמַר הוּא עַצְמוֹ דָּמַי עָלָי:
כֹּהֲנִים וּלְוִיִּם. כֹּהֲנִים אִצְטְרִיכָא לֵיהּ לְאַשְׁמְעִינַן דְּנֶעֱרָכִין, דְּסָלְקָא דַעְתָּךְ אֲמֵינָא הוֹאִיל וּכְתִיב (שם) וְהֶעֱמִידוֹ לִפְנֵי הַכֹּהֵן וְהֶעֱרִיךְ אוֹתוֹ הַכֹּהֵן, יִשְׂרָאֵל לִפְנֵי כֹהֵן וְלֹא כֹהֵן לִפְנֵי כֹהֵן, הִלְכָּךְ כֹּהֵן לֵיתֵיהּ בַּעֲרָכִין, דְּאִי עָנִי הוּא לֹא קָרֵינָא בֵּיהּ וְהֶעֱמִידוֹ לִפְנֵי הַכֹּהֵן, קָמַשְׁמַע לָן דְּכֹהֲנִים אִיתַנְהוּ בְתוֹרַת עֲרָכִין. וְאַיְדֵי דִתְנַן כֹּהֲנִים, תְּנַן לְוִיִּם וְיִשְׂרְאֵלִים:
נָשִׁים וַעֲבָדִים. וְהָאִשָּׁה מְשַׁלֶּמֶת כְּשֶׁתִּתְאַלְמֵן אוֹ תִּתְגָּרֵשׁ, וְהָעֶבֶד כְּשֶׁיִּשְׁתַּחְרֵר:
טֻמְטוּם וְאַנְדְּרוֹגִינוֹס נוֹדְרִים וְנִדָּרִים. שֶׁהֲרֵי יֵשׁ לָהֶם דָּמִים:
וּמַעֲרִיכִין. אֶת אֲחֵרִים, אִם אָמְרוּ עֵרֶךְ פְּלוֹנִי עָלָי, נוֹתְנִים עֵרֶךְ אוֹתוֹ פְּלוֹנִי:
אֲבָל לֹא נֶעֱרָכִין. אִם אָמַר עֶרְכִּי עָלָי, אוֹ אָמַר אַחֵר עָלֵיהֶם עֵרֶךְ פְּלוֹנִי עָלָי, לֹא אָמַר כְּלוּם. דְּזָכָר וּנְקֵבָה אֲמוּרִים בְּפָרָשַׁת עֲרָכִין, עַד שֶׁיִּהְיֶה זָכָר וַדָּאִי אוֹ נְקֵבָה וַדָּאִית:
פָּחוּת מִבֶּן חֹדֶשׁ נִדָּר. אִם אָמַר דָּמָיו עָלָי. שֶׁהֲרֵי שָׁוֶה דָּמִים כָּל שֶׁהֵן:
אֲבָל לֹא נֶעֱרָךְ. שֶׁלֹּא נֶאֱמַר עֵרֶךְ בַּפָּרָשָׁה אֶלָּא מִבֶּן חֹדֶשׁ וָמַעְלָה:

אֵין בַּעֲרָכִין פָּחוּת מִסֶּלַע. אֲפִלּוּ עָנִי שֶׁהֶעֱרִיךְ וְאֵין יָדוֹ מַשֶּׂגֶת אֵינוֹ נִדּוֹן בְּפָחוּת מִסֶּלַע, דִּכְתִיב (ויקרא כז) וְכָל עֶרְכְּךָ יִהְיֶה בְּשֶׁקֶל הַקֹּדֶשׁ, כָּל עֲרָכִין שֶׁאַתָּה מַעֲרִיךְ לֹא יִהְיוּ פְּחוּתִים מִשֶּׁקֶל:
וְלֹא יוֹתֵר עַל חֲמִשִּׁים. דְּזֶהוּ גָדוֹל שֶׁבָּעֲרָכִים דִּכְתִיבֵי בַּפָּרָשָׁה:
נָתַן סֶלַע וְהֶעֱשִׁיר. בֶּן עֶשְׂרִים שֶׁעֶרְכּוֹ חֲמִשִּׁים וְהָיָה עָנִי וְנָתַן סֶלַע בְּעֶרְכּוֹ, דְּעָנִי נִדּוֹן בְּהַשָּׂגַת יָדוֹ, וְהֶעֱשִׁיר, פָּטוּר:
פָּחוֹת מִסֶּלַע. נָתַן פָּחוֹת מִסֶּלַע:
וְהֶעֱשִׁיר נוֹתֵן חֲמִשִּׁים. דְּבִנְתִינָה קַמַּיְתָא לֹא יָצָא יְדֵי עֶרְכּוֹ וַעֲדַיִן עֶרְכּוֹ עָלָיו:
הָיוּ בְיָדָיו חָמֵשׁ סְלָעִים. עָנִי בֶּן עֶשְׂרִים שֶׁהֶעֱרִיךְ עַצְמוֹ, וְהָיוּ לוֹ חָמֵשׁ סְלָעִים:
רַבִּי מֵאִיר אוֹמֵר אֵינוֹ נוֹתֵן אֶלָּא אֶחָת. דְּסָבַר רַבִּי מֵאִיר אוֹ נוֹתֵן כָּל הַחֲמִשִּׁים שֶׁעָלָיו אוֹ אֵינוֹ נוֹתֵן אֶלָּא שֶׁקֶל:
וַחֲכָמִים אוֹמְרִים נוֹתֵן כָּל מַה שֶּׁבְּיָדוֹ. דִּקְרָא דִכְתִיב וְכָל עֶרְכְּךָ יִהְיֶה בְּשֶׁקֶל, לַפָּחוּת שֶׁבַּעֲרָכִין הוּא דַאֲתָא, שֶׁאֲפִלּוּ עָנִי שֶׁבָּעֲנִיִּים לֹא יִתֵּן פָּחוֹת. וּמִיהוּ הֵיכָא דְאִית לֵיהּ טְפֵי יָהֵיב, דִּכְתִיב (שם) עַל פִּי אֲשֶׁר תַּשִּׂיג. וַהֲלָכָה כַּחֲכָמִים:
אֵין פֶּתַח בַּטּוֹעָה פָּחוּת מִשִּׁבְעָה. אַיְדֵי דְאַיְרֵי בְאֵין בַּעֲרָכִין פָּחוּת וְלֹא יוֹתֵר, תָּנָא נַמִּי לְכָל הָנָךְ דְּאִית בְּהוּ אֵין פָּחוּת וְלֹא יוֹתֵר:
פֶּתַח. תְּחִלַּת נִדּוּת. שֶׁדִּין נִדָּה דְאוֹרַיְתָא הוּא. אִם הִתְחִילָה וְרָאֲתָה הַיּוֹם, מוֹנָה שִׁשָּׁה וְהוּא. רָאֲתָה שְׁנַיִם, מוֹנָה חֲמִשָּׁה וְהֵם. רָאֲתָה שְׁלֹשָׁה, מוֹנָה אַרְבָּעָה וְהֵם. [וַאֲפִלּוּ] רָאֲתָה כָל שִׁבְעָה וּפָסְקָה לָעֶרֶב, טוֹבֶלֶת וּמְשַׁמֶּשֶׁת. וּמִיּוֹם שְׁבִיעִי וָאֵילָךְ הָוְיָן יְמֵי זִיבָה אַחַד עָשָׂר יוֹם, שֶׁאִם רָאֲתָה בָהֶן יוֹם אֶחָד אוֹ שְׁנַיִם שׁוֹמֶרֶת יוֹם כְּנֶגֶד יוֹם וְטוֹבֶלֶת. וְאִם רָאֲתָה בָהֶן שְׁלֹשָׁה רְצוּפִים, הַוְיָא זָבָה גְדוֹלָה שֶׁטְּעוּנָה שִׁבְעָה נְקִיִּים וְקָרְבָּן. וְאִם מָשְׁכוּ יְמֵי זִיבָה חֹדֶשׁ אוֹ שָׁנָה, אֵינָהּ חוֹזֶרֶת לִתְחִלַּת נִדּוּת עַד שֶׁתֵּשֵׁב שִׁבְעָה נְקִיִּים. וּכְשֶׁיָּשְׁבָה שִׁבְעָה נְקִיִּים וְרָאֲתָה, הַוְיָא תְחִלַּת נִדָּה וּמוֹנָה שִׁשָּׁה וְהוּא, וְחוֹזֶרֶת לְעִנְיָנָהּ כְּכָל הַמְפֹרָשׁ. וְאִם לֹא רָאֲתָה בְאוֹתָן אַחַד עָשָׂר יוֹם שְׁלֹשָׁה רְצוּפִים, אֵינָהּ זָבָה. וּכְשֶׁתִּרְאֶה אַחֲרֵיהֶן, בֵּין שֶׁהִיא רוֹאָה סָמוּךְ בֵּין שֶׁהִרְחִיקָה רְאִיָּתָהּ הָוֵי תְחִלַּת נִדָּה. וַאֲפִלּוּ תִרְאֶה שְׁלֹשָׁה רְצוּפִים, אֵינָהּ מוֹנָה אֶלָּא אַרְבָּעָה וְהֵם, שֶׁאֵין יְמֵי זוֹב אֶלָּא אוֹתָן אַחַד עָשָׂר יוֹם שֶׁבֵּין סוֹף נִדּוּת זֶה לִתְחִלַּת נִדּוּת הַבָּא:
