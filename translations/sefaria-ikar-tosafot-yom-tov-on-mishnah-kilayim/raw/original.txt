(א) הַחִטִּים. כְּדֵי לְהוֹרוֹת שֶׁשְּׁנוּיִם זוּגוֹת זוּגוֹת לְכָךְ הִפְסִיק בְּזוּג קַמָּא וְתָנֵי עֲלַהּ אֵינָן כִּלְאַיִם זֶה בָזֶה:
(ב) כִּלְאַיִם. כְּתִיב שָׂדְךָ לֹא תִזְרַע כִּלְאָיִם, הָיִיתִי אוֹמֵר אֲפִלּוּ שְׁנֵי מִינֵי חִטִּים אוֹ שְׂעוֹרִים, וְכֵן לְגַבֵּי בְּהֵמָה אֲפִלּוּ שׁוֹר שָׁחוֹר עַל גַּבֵּי לָבָן, וְכֵן לְגַבֵּי בְּגָדִים, פֵּרוּשׁ בִּבְגָדִים צֶמֶר וּפִשְׁתִּים יַחְדָּיו, מַה זֶּה לֹא זֶה מִמִּין זֶה, אַף כִּלְאַיִם שֶׁאָסַרְתִּי לְךָ בְּכָל מָקוֹם לֹא זֶה מִמִּין זֶה וְלֹא זֶה מִמִּין זֶה. יְרוּשַׁלְמִי:
(ג) דְּאִי לָאו הָכִי לֹא הָיוּ אוֹסְרִין אֶלָּא בְּאוֹתוֹ מָקוֹם בִּלְבָד כִּבְסוֹף פֶּרֶק ה':
(ד) וְהַשְּׁעוּעִית. מִפְּנֵי שֶׁמְּשַׁעְשְׁעוֹת אֶת הַלֵּב וּמְהַלֶּכֶת אֶת בְּנֵי מֵעַיִם, יְרוּשַׁלְמִי. כְּלוֹמַר שֶׁמְּטַמְטֶמֶת אֶת הַלֵּב. וּמְשַׁעְשַׁעַת מִלְּשׁוֹן וְעֵינָיו הָשַׁע (יְשַׁעְיָה ו). וְטָח אֶת הַבָּיִת, תַּרְגּוּם וִישׁוּעַ:

=== 2 | chapter 2 ===
(א) דִּדְבַר תּוֹרָה חַד בִּתְרֵי בָּטֵל אֶלָּא דַחֲכָמִים אָסְרוּ מִפְּנֵי מַרְאִית הָעַיִן וְהֵם אָמְרוּ דְּפָחוֹת מֵרֹבַע לִסְאָה מִתְבַּטֵּל:
(ב) יְמַעֵט. אוֹ מוֹסִיף עַל סְאָה אוֹ פּוֹחֵת מֵרֹבַע, וְאַף דְּכָל הָאִסּוּרִין שֶׁרִבָּה עֲלֵיהֶם מֵזִיד אָסוּר, הָכָא הָאִסּוּר אֵינוֹ אֶלָּא מַרְאִית הָעַיִן:

=== 3 | chapter 3 ===
(א) עֲרוּגָה. לְשׁוֹן עִבְרִי הוּא, מֵעֲרֻגוֹת מַטָּעָהּ (יְחֶזְקֵאל יז):
(ב) זוֹרְעִים כוּ'. כְּדִתְנַן בְּפֶרֶק ט'‏ דְּשַׁבָּת כִּי כָאָרֶץ תּוֹצִיא צִמְחָהּ וּכְגַנָּהּ זֵרוּעֶיהָ תַצְמִיחַ. וְשָׁם פֵּרַשְׁתִּיו בְּסִיַּעְתָּא דִשְׁמַיָּא:
(ג) זֵרְעוֹנִים. זֶה הַשֵּׁם זֵרְעוֹנִים כּוֹלֵל לְכָל הַנִּזְרָע, אֲבָל בְּכָאן הַכַּוָּנָה לִזְרָעִים מְיֻחָדִים וְהֵם הַזֶּרַע שֶׁל מִינֵי יְרָקוֹת וּכְדִתְנַן בְּרֵישׁ מַתְנִיתִין דִּלְקַמָּן. הָרַ"מ:
(ד) וְאִם תֹּאמַר יִזָּרְעוּ תִּשְׁעָה מִינִין בָּעֲרוּגָה, אַרְבָּעָה בְּאַרְבַּע קְרָנוֹת, וְאַרְבָּעָה בְּאֶמְצַע כָּל רוּחַ, וְאֶחָד בָּאֶמְצַע, דְּכָל אֶחָד רָחוֹק מֵחֲבֵרוֹ שְׁלֹשָׁה טְפָחִים. וְיֵשׁ לוֹמַר דְּכֻלֵּי הַאי הֲוֵי עִרְבּוּב. עוֹד יֵשׁ לוֹמַר דְּהַשְׁתָּא יָכוֹל לִזְרֹעַ הַרְבֵּה מִכָּל מִין שֶׁבָּרוּחַ, תּוֹסָפוֹת. וְכָתְבוּ עוֹד דְּאַף עַל גַּב דְּבַעֲרוּגָה שֶׁהִיא חֲמִשָּׁה עַל חֲמִשָּׁה יֵשׁ בַּאֲלַכְסוֹנָהּ שִׁבְעָה, דְּכָל אַמָּתָא בְּרִבּוּעַ אַמָּתָא וּתְרֵי חֻמְשֵׁי בַּאֲלַכְסוֹנָהּ, וְיָכוֹל לִזְרֹעַ בְּאַרְבַּע קְרָנוֹת וּבָאֶמְצַע, וְתֵרֵץ רִ"י דְּהָא מִקְּרָא יָלְפִינַן וְאֵין לְהַעֲמִיד הַמִּקְרָא בְּעִנְיָן זֶה שֶׁאֵינוֹ שִׁעוּר לֹא לִינִיקָה וְלֹא לְשׁוּם דָּבָר:
(ה) שֶׁלֹּא אָסְרָה תּוֹרָה אֶלָּא בְּמַפֹּלֶת יָד וּקְרָא דְּהָכָא דְּקָפֵּיד אַעִרְבּוּב אֵינוֹ אֶלָּא אַסְמַכְתָּא:
(ו) וּמִכָּל מָקוֹם צָרִיךְ שֶׁיַּנִּיחַ קְצָת לְהַפְרִישׁ בֵּינֵיהֶן כִּדְפֵרֵשׁ רַשִׁ"י בַּגְּמָרָא:
(ז) וְשִׁעוּר שְׁלֹשָׁה טְפָחִים הַיְנוּ עִם מְקוֹם הַזְּרָעִים. כָּתַב הָרַ"שׁ וְהָא דִתְנַן לְעֵיל בְּפֶרֶק ב'‏ יָרָק בְּיָרָק שִׁשָּׁה טְפָחִים הַנֵּי מִילֵי בְּשָׂדֶה גְדוֹלָה שֶׁיֵּשׁ בָּהּ יָרָק הַרְבֵּה, אֲבָל הָכָא דְּאֵין זוֹרֵעַ מִכָּל אֶחָד וְאֶחָד אֶלָּא מְעַט לֹא חַיְשִׁינַן, וְדִכְוָתַהּ אַשְׁכְּחָן גַּבֵּי תְבוּאָה בֵּית רֹבַע דְּאָמְרִינַן בַּיְרוּשַׁלְמִי שֶׁרַשַּׁאי לִזְרֹעַ חֲמִשָּׁה מִינִין בְּבֵית רֹבַע אֶחָד דִּבְכָל מָקוֹם שֶׁאֵין זוֹרֵעַ מִכָּל מִין וָמִין אֶלָּא מְעַט לֹא חַיְשִׁינַן אֶלָּא לִינִיקָה:
(ח) גָּבוֹהַּ טֶפַח. וְגַם בְּרָחְבּוֹ טֶפַח כִּדְאִיתָא בִּגְמָרָא שַׁבָּת דַּף פ"ה מִדְּרַבִּי יְהוּדָה דְּאָמַר לְקַמָּן מִשְׁנָה ג'‏ רֹחַב הַפַּרְסָה, וְכָתְבוּ הַתּוֹסָפוֹת דְּאַף עַל גַּב דְּרַבִּי יְהוּדָה לְקַמָּן לֹא מַיְרֵי אֶלָּא לְעִנְיַן הֶפְסֵק, נִרְאֶה לַגְּמָרָא דְּלֹא שְׁנָא לִגְבוּל דְּהוּא הַדִּין נַמִּי שֶׁשִּׁעוּרוֹ טֶפַח:
(ט) שְׁלֹשָׁה כוּ'. וְתֵימַהּ שֶׁלִּשְׁנֵים עָשָׂר זְרָעִים צָרִיךְ שְׁתֵּים עֶשְׂרֵה מֶרְחָקִין וְכָל מֶרְחָק שְׁלֹשָׁה טְפָחִים הֵם שְׁלֹשִׁים וְשִׁשָּׁה טְפָחִים, וְאֵין בְּכָל הַגְּבוּלִים סָבִיב כִּי אִם עֶשְׂרִים וּשְׁמֹנָה טְפָחִים, לִשְׁנַיִם לְכָל אֶחָד שְׁמֹנָה טְפָחִים וְלִשְׁנַיִם לְכָל אֶחָד שִׁשָּׁה טְפָחִים. וְלָכֵן נִרְאֶה לִי שֶׁעַל כָּל גְּבוּל הוּא זוֹרֵעַ שְׁלֹשָׁה. וְכָךְ יָכוֹל לַעֲשׂוֹת, שֶׁמַּנִּיחַ טֶפַח הַסְּמוּכָה לַקֶּרֶן וְזוֹרֵעַ מִין אֶחָד וּמַנִּיחַ שְׁלֹשָׁה טְפָחִים וְזוֹרֵעַ, וּמַנִּיחַ עוֹד שְׁלֹשָׁה טְפָחִים וְזוֹרֵעַ, וְנִמְצָא שֶׁיֵּשׁ עוֹד טֶפַח פָּנוּי בַּקֶּרֶן הַשֵּׁנִי, וְכֵן הוּא עוֹשֶׂה לְכָל גְּבוּל וּגְבוּל, וְנִמְצָא שֶׁיֵּשׁ בֵּין מִין וָמִין שְׁלֹשָׁה טְפָחִים שִׁעוּר הַרְחָקָה, מִלְּבַד בְּאוֹתָן שֶׁסְּמוּכוֹת לַקְּרָנוֹת שֶׁאֵין בֵּינֵיהֶם הַרְחָקָה לְפִי שֶׁאֵין צְרִיכִין, שֶׁיֵּשׁ הֶכֵּר בַּדֶּרֶךְ שֶׁנִּזְרָעִין שֶׁזֶּה נִזְרָע צָפוֹן וְדָרוֹם וְזֶה מִזְרָח וּמַעֲרָב כִּדְכָתַב הָרַ"ב בְּרֵישָׁא. שֶׁהֲרֵי כָּתַבְתִּי שֶׁרֹחַב הַגְּבוּל טֶפַח, וְנִמְצָא שֶׁזּוֹרֵעַ בְּאֹרֶךְ טֶפַח וְיֵשׁ הֶכֵּר בִּזְרִיעָתָן כְּמוֹ שֶׁכָּתַבְתִּי. מַה שֶּׁאֵין כֵּן בְּרֵישָׁא שֶׁאִם יִזְרַע תִּשְׁעָה הֲוֵי עִרְבּוּב כְּמוֹ שֶׁכָּתְבוּ הַתּוֹסָפוֹת, לְפִי שֶׁאֵין כָּאן הֶכֵּר, וְעוֹד שֶׁלֹּא יוּכַל לִזְרֹעַ מִכָּל מִין כִּי אִם גַּרְעִין אֶחָד. תּוֹסְפוֹת יוֹם טוֹב:
(י) מִפְּנֵי כוּ'. וְהַרְחָקַת שְׁלֹשָׁה טְפָחִים לֹא מְהַנֵּי דְּאַכַּתִּי הֲוֵי נִרְאֶה מְעֻרְבָּב וְאַף עַל פִּי שֶׁאֵין יוֹנְקִין זֶה מִזֶּה. אֲבָל בַּעֲרוּגָה פְּשׁוּטָה דְרֵישָׁא רַשַּׁאי לִטַּע אֲפִלּוּ רֹאשׁ לֶפֶת בָּאֶמְצַע, לְפִי שֶׁאֵין כָּאן אֶלָּא אַרְבָּעָה מִינִין לְאַרְבַּע הָרוּחוֹת לֹא נִרְאֶה כְּעִרְבּוּבְיָא. כָּךְ נִרְאֶה לִי. תּוֹסְפוֹת יוֹם טוֹב:
(יא) שִׁשָּׁה. דָּרַשׁ זֶרַע זַרְעָהּ זְרָעֶיהָ זֵרוּעֶיהָ כְּלוֹמַר דְּמָצֵי לְמִכְתָּב זֶרַע דְּמַשְׁמַע חַד וְזַרְעָהּ מַשְׁמַע תְּרֵי וּזְרָעֶיהָ מַשְׁמַע תְּלָת וּכְתִיב זֵרוּעֶיהָ הֲרֵי אַרְבָּעָה וּכְתִיב תּוֹצִיא צִמְחָהּ תּוֹצִיא חַד וְצִמְחָהּ חַד כְּמוֹ לְתַנָּא קַמָּא, הֲרֵי שִׁית. וְכִי רָמֵיז קְרָא שִׁשָּׁה זֵרְעוֹנִים בַּעֲרוּגָה בַּת שִׁשָּׁה טְפָחִים, דְּקִים לֵיהּ לְרַבִּי יְהוּדָה דְּשִׁית בְּשִׁית לֹא יָנְקֵי, כִּי הֵיכִי דִּלְרַבָּנָן בַּחֲמִשָּׁה בְּשִׁשָּׁה כוּ'. וְלֹא אִתְבָּרֵר כַּמָּה הוּא שִׁעוּר הֶרְחֵק לְרַבִּי יְהוּדָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(יב) בָּאֶמְצַע. וְרַבִּי יְהוּדָה פָּלֵיג בֵּין בְּרֵישָׁא וּבֵין בְּסֵיפָא, וְהָכִי אִיתָא בַּיְרוּשַׁלְמִי:
