(ב) שֶׁזֶּה טוֹעֲנוֹ שֶׁשָּׁלַח יָד בְּפִקְדוֹנוֹ, דְּהַיְנוּ גְזֵלוֹת. אֲבָל לָוָה וְלֹא שִׁלֵּם לֹא מִקְּרֵי גַזְלָן, דְּמִלְוָה לְהוֹצָאָה נִתְּנָה. וֵאלֹהִים מַשְׁמַע לְשׁוֹן שְׂרָרָה וְרַבָּנוּת. כְּדִמְתַרְגְּמִינָן רְאֵה נְתַתִּיךָ אֱלֹהִים לְפַרְעֹה (שְׁמוֹת ז) לְשׁוֹן רַבָּנוּת. רַשִׁ"י:
(ג) וַחֲבָלוֹת. נִרְאֶה לְרַבֵּנוּ תָם דְּכֹפֶר וּשְׁלשִׁים דְּעֶבֶד וְעֶשְׂרִים וְאַרְבָּעָה רָאשֵׁי אֵבָרִים דְּעֶבֶד, בִּכְלַל חֲבָלוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ד) דְּשׁוֹר הַמּוּעָד דְּאָמְרָן דְּלָאו דְּשֵׁן וָרֶגֶל אֶלָּא דְּקֶרֶן דְּלֹא מַגְבִּין בְּבָבֶל שֶׁאֵין מֻמְחִין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(א) וְדָנִין כוּ'. פְּשִׁיטָא (אַטּוּ מִשּׁוּם חֲשִׁיבוּתוֹ מִגְרַע גָּרַע, וְכֵיוָן דְּדָן אִיהוּ נַמִּי דָנִין אוֹתוֹ מִשּׁוּם הִתְקוֹשְׁשׁוּ וָקוֹשּׁוּ. תּוֹסָפוֹת). אַיְדֵי דְקָא בָעֵי לְמִתְנֵי מֶלֶךְ לֹא דָן וְלֹא דָנִין אוֹתוֹ, תָּנֵי נַמִּי כֹהֵן גָּדוֹל. וְאִי בָעֵית אֵימָא הָא קָא מַשְׁמַע לָן דְּכָל שְׁאָר דִּינִין חוּץ מִדִּינֵי נְפָשׁוֹת דָּן וְדָנִין אוֹתוֹ כְהֶדְיוֹט בִּשְׁלשָׁה כוּ'. גְּמָרָא:
(ב) מֵעִיד. דַּוְקָא כְשֶׁהַמֶּלֶךְ יוֹשֵׁב שָׁם, שֶׁאִם הַכֹּהֵן גָּדוֹל יוֹדֵעַ עֵדוּת לִבְנוֹ הוֹלֵךְ הַמֶּלֶךְ לְשָׁם מִשּׁוּם יְקָרָא דְכֹהֵן גָּדוֹל, וְקָא אָתֵי וּמַסְהִיד. אֲבָל בְּלָאו הָכִי אֵינוֹ מֵעִיד, דְּזִילוּתָא הוּא לְכֹהֵן גָּדוֹל שֶׁיָּבֹא וְיָעִיד, דְּתַנְיָא וְהִתְעַלַּמְתָּ, פְּעָמִים שֶׁאַתָּה מִתְעַלֵּם, כְּגוֹן זָקֵן וְאֵינוֹ לְפִי כְבוֹדוֹ. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ג) מִשּׁוּם דְּקַשְׁיָא לֵיהּ דְּאִי הָכִי אֲפִלּוּ לְבֵיתוֹ נַמִּי לֹא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ד) גְּמָרָא. וּפֵרוּשׁ בֵּינוֹ, רְצוֹנוֹ לוֹמַר בֵּין עַצְמוֹ שֶׁהוּא הַמְמֻנֶּה לְבֵין הָעָם:
(ה) דְּאִלּוּ בְיוֹם אַחֵר אֵין מִצְוָה בְכֹהֵן גָּדוֹל דַּוְקָא וְרַשַּׁאי כֹּהֵן הֶדְיוֹט לַעֲבֹד אִם אֵרַע פְּסוּל בְּכֹהֵן גָּדוֹל וְאֵין צָרִיךְ לְהַמְמֻנֶּה:
(ו) הַסַּפְסָל. הוּא כִסֵּא קָטָן. הָרַמְבַּ"ם:

(א) זֶה בּוֹרֵר לוֹ אֶחָד. וְהָא דְכָתַב הָרַ"ב בְּרֵישׁ פֶּרֶק קַמָּא, דִּשְׁלֹשָׁה הֶדְיוֹטוֹת כּוֹפִין לָדוּן בִּפְנֵיהֶם, מַיְרֵי בִמְסָרֵב וְאֵינוֹ רוֹצֶה לָדוּן כְּלָל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ב) רַשִׁ"י. וּפֵרֵשׁ הָרֹא"שׁ דִּרְצוֹנוֹ לוֹמַר שֶׁסּוֹבֵר כָּךְ שֶׁיַּהֲפֹךְ בִּזְכוּתוֹ יוֹתֵר מִבִּזְכוּת הָאַחֵר, וּמִתּוֹךְ כָּךְ צָיְתֵי לְדִינַיְהוּ. אֲבָל הַדַּיָּן עַצְמוֹ חָלִילָה לִמְצֹא סְבָרָא לְזַכּוֹתוֹ אִם לֹא שֶׁיֵּרָאֶה לוֹ דִּין גָּמוּר. אֲבָל אִם הָיָה יָכוֹל לְהַטְעוֹת אֶת חֲבֵרוֹ שֶׁיְּקַבֵּל אֶת סְבָרָתוֹ אַף עַל פִּי שֶׁהוּא מְסֻפָּק בָּהּ, הוּא בִכְלַל מַטֶּה מִשְׁפָּט. אֲבָל דַּעַת הָרַ"ב מִתּוֹךְ שֶׁזֶּה בוֹרְרוֹ מֵבִין דְּבָרָיו לַאֲשׁוּרוֹ כוּ'‏ נִמְצָא שֶׁלֹּא יִשָּׁאֵר שׁוּם צַד זְכוּת נֶעְלָם מֵעֵינֵי הַדַּיָּנִים שְׁנֵיהֶם, וְהַשְּׁלִישִׁי שׁוֹמֵעַ מַשָּׂא וּמַתָּן שֶׁל שְׁנֵיהֶם וּמַכְרִיעַ בֵּינֵיהֶם וְיוֹצֵא הַדִּין לַאֲמִתּוֹ. עַד כָּאן:
(ג) וּבַגְּמָרָא מַסְּקִינַן דְּכֻלֵּי עָלְמָא דַּעַת הַדַּיָּנִים בָּעִינַן שֶׁיְּבָרְרוּ מִי שֶׁיֵּשֵׁב עִמָּהֶן אָדָם הָגוּן, וְכִי פְלִיגֵי בְּדַעַת בַּעֲלֵי דִינִין, רַבִּי מֵאִיר סָבַר דַּעַת בַּעֲלֵי דִינִין נַמִּי בָעִינַן כוּ':
(ד) פּוֹסֵל דַּיָּנוֹ. בַּגְּמָרָא, כֹּל כְּמִנֵּיהּ דְּפָסֵיל דַּיָּנֵי. אָמַר רַבִּי יוֹחָנָן בְּעַרְכָּאוֹת שֶׁבְּסוּרְיָא שָׁנוּ, שֶׁהֵם הֶדְיוֹטוֹת שֶׁאֵין יוֹדְעִין דִּבְרֵי תוֹרָה אֶלָּא שֶׁהֻרְגְּלוּ לָדוּן. וְרַבִּי מֵאִיר יָהֵיב לְהוּ דִין יוֹשְׁבֵי קְרָנוֹת, וְיָכוֹל לְפָסְלָן. וְרַבָּנָן סָבְרֵי כֵּיוָן שֶׁהִמְחוּם רַבִּים עֲלֵיהֶם לֹא כֹל הֵימֶנּוּ לְפָסְלָן, וַעֲדִיפֵי מִיּוֹשְׁבֵי קְרָנוֹת. אַחֲרוֹנִים:
(ה) זֶה הוּא לְשׁוֹן רַשִׁ"י בְּאוֹקִימְתָא קַמַּיְתָא. אֲבָל בַּמַּסְקָנָא כָתַב לֹא אָדוּן לְפָנָיו:
(ו) וַאֲפִלּוּ נָשָׂא וְנָתַן בַּיָּד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ז) וְקַיְמָא לָן כְּמַאן דְּדָאֵין דִּינָא דְגַרְמֵי:
(ח) וְלֹא יָדַע שֶׁכְּבָר פָּשַׁט הַמַּעֲשֶׂה בְּכָל הָעוֹלָם כְּדִבְרֵי הָאֶחָד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ט) וְחוֹזֵר וְנוֹטֵל כוּ'‏ בְּמִכָּל שֶׁכֵּן מֵהָא דִּלְקַמָּן:
(י) נָפְקָא מִנֵּיהּ לְעִנְיַן זְמָן. אִי נַמִּי זִמְנִין דְּלֵיכָּא לְבָרוֹרֵי טָעוּתָא לֹא מֵהַמִּשְׁנָה וְלֹא מֵהַסּוּגְיָא דְעָלְמָא, וַאֲפִלּוּ הָכִי אֶפְשָׁר שֶׁאֵינוֹ דִין צֶדֶק וִיכוֹלִים הַדַּיָּנִים לִפְסֹק דְּלֹא כְמוֹ שֶׁפָּסַק זֶה הַדַּיָּן, דְּאַטּוּ כֻלְּהוּ דִינֵי נִשְׁנוּ בַמִּשְׁנָה וְסוּגְיָא דְעָלְמָא:
(יא) שֶׁזֶּה מִתְכַּוֵּן לְהַזִּיק הוּא כֵּיוָן דְּלָאו דַּיָּנָא הוּא:
(יב) דְּהָכָא בְּבָבֶל שֵׁבֶט, לְשׁוֹן שְׂרָרָה, וְיֵשׁ לָהֶם רְשׁוּת [לְהַפְקִיר, דְּהֶפְקֵר בֵּית דִּין הֶפְקֵר]. וְהָתָם בְּאֶרֶץ יִשְׂרָאֵל מְחוֹקֵק, שְׂרָרָה מֻעֶטֶת. כִּדְאִיתָא בַגְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(יג) כְּלוֹמַר וְדִינָיו דִּין. אֲבָל אִם טָעָה, אֵינוֹ פָטוּר, הוֹאִיל וְלֹא נָקַט רְשׁוּתָא:
(יד) רַשִׁ"י. וּקְצָת קָשֶׁה לוֹמַר כֵּן דְּמִפְּנֵי שֶׁסָּבוּר הָיָה שֶׁיֵּשׁ לוֹ שְׁתֵּי כִתֵּי עֵדִים וְאָמַר כֵּן וְאַחַר כָּךְ בִּקֵּשׁ וְלֹא מָצָא יַפְסִיד. וְהָרַמְבַּ"ם מְפָרֵשׁ דִּפְלוּגְתַיְהוּ כְּשֶׁהֵבִיא שְׁתֵּי כִתֵּי עֵדִים שֶׁאַחַר שֶׁהֵעִידוּ כַת אַחַת כִּדְבָרָיו הֵבִיא עוֹד אַחֶרֶת וְהֵעִידוּ גַּם כֵּן כְּמוֹ הָרִאשׁוֹנִים, וְטָעַן הַלָּה שֶׁאַרְבָּעָה הָעֵדִים פְּסוּלִים הֵם, רַבִּי מֵאִיר אוֹמֵר שׁוֹמְעִים לוֹ, וְנֹאמַר לְזֶה שֶׁהֱבִיאָם הָבֵא רְאָיָה שֶׁהֵם כְּשֵׁרִים, לְפִי שֶׁאַתָּה לֹא הָיִיתָ מֵבִיא עֵדִים אַחַר עֵדִים אֶלָּא שֶׁאַתָּה יוֹדֵעַ שֶׁיֵּשׁ בָּהֶן מִי שֶׁאֵינוֹ כָשֵׁר, וַאֲנַחְנוּ לֹא נֵדַע אֵיזֶהוּ הָרִאשׁוֹנִים אוֹ הָאַחֲרוֹנִים. וַחֲכָמִים אוֹמְרִים אֵין חוֹשְׁשִׁים לָזֶה, לְפִי שֶׁלֹּא עָשָׂה זֶה אֶלָּא לְחַזֵּק טַעֲנוֹתָיו וּלְהוֹדִיעַ כִּי זֶה יָדוּעַ לָרַבִּים:

(א) הָרַמְבַּ"ם בְּפֵרוּשׁוֹ. אֲבָל בְּחִבּוּרוֹ כָתַב אֵין עֵדֵי מָמוֹן צְרִיכִין דְּרִישָׁה וַחֲקִירָה. וְכֵן הוּא בַגְּמָרָא:
