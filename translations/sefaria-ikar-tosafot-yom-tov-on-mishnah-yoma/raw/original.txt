(א) מִבֵּיתוֹ. פֵּרֵשׁ הָרַמְבַּ"ם מֵאִשְׁתּוֹ, כְּדֵי שֶׁלֹּא תִּמָּצֵא נִדָּה וְיִהְיֶה הוּא טָמֵא שִׁבְעָה יָמִים, כְּמוֹ שֶׁכָּתוּב [וַיִּקְרָא טו] וּתְהִי נִדָּתָהּ עָלָיו וְטָמֵא שִׁבְעָה יָמִים. וְהָכִי פֵּרוּשׁוֹ, שֶׁלֹּא תִּמְצָא אַחַר בִּיאָה דָּם עַל הַסָּדִין שֶׁקִּנְּחָה בּוֹ וְאֵין יָדוּעַ אִם בִּשְׁעַת בִּיאָה הָיָה הַדָּם אוֹ אַחַר כֵּן, וְיִהְיֶה בְּתוֹךְ כְּדֵי שֶׁתֵּרֵד מִן הַמִּטָּה וְתָדִיחַ, דְּבִכְהַאי גַּוְנָא אַף לְתַנָּא קַמָּא דְּמִשְׁנָה ג'‏ פֶּרֶק ב'‏ דְּנִדָּה מְטַמְּאָה אֶת בּוֹעֲלָהּ. גְּמָרָא:
(ב) אֵין לַדָּבָר סוֹף. אֲבָל בְּטֻמְאָה, כֹּהֵן גָּדוֹל זָרִיז הוּא וְנִשְׁמָר מֵהַטֻּמְאָה, וְלֹא מַתְקִינִין לוֹ כֹּהֵן אַחֵר אֶלָּא מִשּׁוּם דְּכִי עָבְדִינַן לֵיהּ צָרָה כָּל שֶׁכֵּן דְּמִזָּרִיז טְפֵי. גְּמָרָא:

(א) הַכַּף וְאֶת הַמַּחְתָּה. מַחְתָּה מְלֵאָה קְטֹרֶת, וּמִמֶּנָּה חָפַן כוּ'. גְּמָרָא:
(ב) דְּאִי לָאו הָכִי, יַצִּיבָא בְּאַרְעָא וְגִיּוֹרָא בִּשְׁמֵי שְׁמַיָּא [בִּתְמִיהַּ] אֲבָל לֹא מִשּׁוּם דְּהוֹלָכָה בִּשְׂמֹאל פְּסוּלָה. וְהָרַמְבַּ"ם מְצָרֵף גַּם הַטַּעַם הַזֶּה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ג) וְאֵינָהּ יְכוֹלָה לַעֲמֹד בְּגֹבַהּ (מֵאָה) [מ'] אַמָּה. וּלְהוֹסִיף עַל עָבְיָהּ אִי אֶפְשָׁר, דִּכְתִיב (דִּבְרֵי הַיָּמִים א כה) הַכֹּל בִּכְתָב מִיַּד ה'‏ עָלַי הִשְׂכִּיל:
(ד) פָּרֹכֶת אַחַת. שֶׁעָבְיָהּ אַמָּה, דִּסְבִירָא לֵיהּ וְהִבְדִּילָה הַפָּרֹכֶת הַיְנוּ בֵּין חָלָל לְחָלָל, דְּחָלָל שֶׁחוּצָהּ לָהּ קֹדֶשׁ וְחָלָל שֶׁלִּפְנִים הֵימֶנָּה קֹדֶשׁ קָדָשִׁים. אִי נַמִּי דִּלְמָא פְּשִׁיטָא לֵיהּ דִּקְדֻשַּׁת הָאַמָּה כְּלִפְנִים, נִמְצָא דִּלְדִידֵיהּ מִתְּחִלַּת עָבְיָהּ וְלִפְנִים קֹדֶשׁ קָדָשִׁים, כְּמוֹ בַּמִּשְׁכָּן. תּוֹסָפוֹת:
(ה) וּרְצוֹנוֹ לוֹמַר שֶׁהַפָּרֹכֶת עַל פְּנֵי כָּל הַדְּבִיר מִכֹּתֶל הַצָּפוֹן עַד כֹּתֶל הַדָּרוֹם. וְשָׁם בְּכֹתֶל דָּרוֹם נִכְפְּלָה, רְצוֹנוֹ לוֹמַר הָיְתָה אֲרֻכָּה יוֹצֵאת לַחוּץ לְתוֹךְ הַהֵיכָל וְנֶאֱחֶזֶת בְּכֹתֶל הַדָּרוֹם בַּקֶּרֶס כוּ', וְעַל שֵׁם הַקֶּרֶס מִקְּרֵי פְּרוּפָה. קַרְסֵי זָהָב מְתַרְגֵּם פּוּרְפִין דִּדְהַב:
(ו) מִן הַדָּרוֹם. וְאַף עַל גַּב דְּנִמְצָא [כְּשֶׁהוּא] מְהַלֵּךְ בַּהֵיכָל עַד שֶׁמַּגִּיעַ לִפְרוּפָה זוֹ שֶׁהוּא הוֹלֵךְ לִשְׂמֹאלוֹ, שֶׁהַהוֹלֵךְ מִמִּזְרָח לְמַעֲרָב וְהוֹפֵךְ פָּנָיו לַדָּרוֹם פּוֹנֶה לִשְׂמֹאלוֹ, וּבְמִשְׁנָה ה'‏ [בְּפֶרֶק דִּלְעֵיל] לְעֵיל כָּתַב הָרַ"ב דְּכָל פִּנּוֹת כוּ'‏ לֹא יִהְיוּ אֶלָּא לְיָמִין, מִכָּל מָקוֹם כְּדֵי שֶׁכְּשֶׁהוּא מְהַלֵּךְ עִם הַפָּרֹכֶת הַשֵּׁנִי בְּתוֹךְ הַדְּבִיר, שֶׁנִּמְצָא שְׂמֹאלוֹ עִם הַפָּרֹכֶת וִימִינוֹ מוּל אֲרוֹן הַקֹּדֶשׁ. אֲבָל לַעֲשׂוֹת גַּם פְּרִיפָה הָרִאשׁוֹנָה בַּצָּפוֹן לֹא בָּעוּ לְמֶיעֱבַד, דְּכָבוֹד הוּא לְקֹדֶשׁ הַקָּדָשִׁים שֶׁלֹּא יִהְיוּ נִפְתָּחִים שְׁתֵּי הַפָּרֹכוֹת בְּצַד אֶחָד וְיִהְיוּ עֵינָיו נִזּוֹנוֹת מִיָּד מִבֵּית קֹדֶשׁ הַקָּדָשִׁים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ז) כִּי בְּבַיִת רִאשׁוֹן שֶׁהָיָה שָׁם אָרוֹן לֹא הָיָה שָׁם אֶלָּא פָּרֹכֶת אַחַת, וְהוּא יְסַפֵּר בְּכָאן סֵדֶר בַּיִת שֵׁנִי שֶׁלֹּא הָיָה שָׁם אָרוֹן וְהָיוּ שָׁם שְׁתֵּי פָּרֹכוֹת. הָרַמְבַּ"ם. וּבַגְּמָרָא וְהָא קָתָנֵי לְבֵין שְׁנֵי הַבַּדִּים, אֵימָא כְּבֵין שְׁנֵי הַבַּדִּים:
(ח) עָשָׁן. לְשׁוֹן הָרַמְבַּ"ם וּמַמְתִּין שָׁם עַד שֶׁיִּתְמַלֵּא הַבַּיִת עָשָׁן וְיֵצֵא. דְּאִי לֹא תֵּימָא הָכִי מַאי לְמֵימְרָא דְּנִתְמַלֵּא כוּ':
(ט) וְיָלְפִינַן לַהּ מִדְּאָמַר קְרָא (דִּבְרֵי הַיָּמִים ב כא) וַיָּבֹא שְׁלֹמֹה לַבָּמָה אֲשֶׁר בְּגִבְעוֹן יְרוּשָׁלַיִם, דַּהֲוָה לֵיהּ לְמִכְתַּב מִירוּשָׁלַיִם, אוֹ מִן הַבָּמָה כוּ'‏ לִירוּשָׁלַיִם. אֶלָּא הָכִי קָאָמַר, וַיָּבֹא שְׁלֹמֹה הַבָּמָה אֲשֶׁר בְּגִבְעוֹן דֶּרֶךְ הֲלִיכָתוֹ, וְכֵן לִירוּשָׁלַיִם בְּשׁוּבוֹ מִגִּבְעוֹן הָיְתָה כְּבִיאָה רִאשׁוֹנָה:
(י) כִּי הַרְבֵּה כֹּהֲנִים גְּדוֹלִים הָיוּ מֵתִים בְּקֹדֶשׁ הַקָּדָשִׁים אִם לֹא הָיוּ רְאוּיִין אוֹ אִם יְשַׁנֶּה דָּבָר בְּמַעֲשֵׂה הַקְּטֹרֶת. הֲלֹא תִּרְאֶה אָמַר ה'‏ יִתְעַלֶּה לְאַהֲרֹן [וַיִּקְרָא טז], וְלֹא יָמוּת, כִּי בֶעָנָן אֵרָאֶה עַל הַכַּפֹּרֶת. הָרַמְבַּ"ם:
