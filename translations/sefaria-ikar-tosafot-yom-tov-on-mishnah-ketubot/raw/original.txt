(א) שֶׁפַּעֲמַיִם כוּ'. אַבְּתוּלָה קָאֵי. רַשִׁ"י. וְכָתַב הָרַ"ן דְּמַקְשׁוּ הָכָא, דְּלִתְנֵי נַמִּי טַעֲמָא דְּשָׁקְדוּ לְמִעוּטֵי יוֹם רִאשׁוֹן בְּשַׁבָּת, וְטַעֲמָא דְּאַלְמָנָה בְּיוֹם חֲמִישִׁי כוּ'. עוֹד יֵשׁ לוֹמַר דְּמִשּׁוּם שָׁקְדוּ לֹא הֲווֹ מְתַקְּנֵי רַבָּנָן לְעִכּוּבֵי מִצְוָה דְנִשּׂוּאִין, אֶלָּא מִשּׁוּם טַעֲמָא דִבְתוּלִים דְּמִלְּתָא דְאִסּוּרָא הִיא הֻצְרְכוּ לְיַחֵד לַבְּתוּלָה יָמִים, וְכֵיוָן שֶׁהֻצְרְכוּ לְיַחֵד יִחֲדוּ לַבְּתוּלָה רְבִיעִי דְּמַעֲלֵי טְפֵי מֵאֶחָד בְּשַׁבָּת מִשּׁוּם שָׁקְדוּ, וְאַגַּב, יִחֲדוּ לָאַלְמָנָה גַּם כֵּן, דְּאִי לֹא, אָתוּ לְמֵימַר דְּנִשּׂוּאֵי אַלְמָנָה זִילוּ לְרַבָּנָן וְלֹא מְתַקְּנֵי בְהוּ מִלְּתָא, וּמִשּׁוּם הָכִי תָּנֵי טַעֲמָא דִבְתוּלִים, לוֹמַר דְּעִקַּר הַתַּקָּנָה הָיְתָה בִּשְׁבִילָהּ. וְעַיֵּן עוֹד:
(ב) הַשֵּׁנִי כוּ'. דִּשְׁכִיחֵי, דְּאָתוּ לְמִקְרֵי בְּסִפְרָא. גְּמָרָא:
(ג) פֵּרוּשׁ, לְאַחַר שֶׁנִּתְאָרְסָה, וַהֲרֵי הִיא אֲסוּרָה עָלָיו, כְּגוֹן אֵשֶׁת כֹּהֵן, אוֹ שֶׁקִּבֵּל בָּהּ אָבִיהָ קִדּוּשֶׁיהָ פְּחוּתָה מִבַּת שָׁלֹשׁ, דְּלֵיכָּא אֶלָּא חַד סְפֵקָא, בְּכֹהֵן, שֶׁמָּא קֹדֶם שֶׁנִּתְאָרְסָה, וּבִפְחוּתָה מִבַּת שָׁלֹשׁ, שֶׁמָּא בְּאֹנֶס. וּמִשּׁוּם הָכִי אַף בִּשְׁאָר נָשִׁים דְּאִיכָּא סְפֵק סְפֵקָא וְאֵינָהּ אֲסוּרָה עָלָיו לֹא פְּלוּג רַבָּנָן בְּתַקַּנְתָּא. תּוֹסָפוֹת:
