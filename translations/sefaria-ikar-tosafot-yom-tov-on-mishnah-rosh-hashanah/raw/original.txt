(א) שֶׁהֲרֵי מָנָה שָׁנָה זוֹ שְׁנַת אַרְבַּע מֵאוֹת וּשְׁמוֹנִים לִיצִיאַת מִצְרַיִם, וְשָׁנָה רְבִיעִית לִמְלֹךְ שְׁלֹמֹה. רַשִׁ"י:
(ב) כִּדְּיָלֵיף לַהּ מִקְּרָאֵי בַּגְּמָרָא:
(ג) לְמַלְכֵי יִשְׂרָאֵל אַסְבְּרַא לֵיהּ, דִּבְהוּ קַיְמִינַן הַשְׁתָּא:
(ד) דְּאִלּוּ בְּנִיסָן שֶׁלְּאַחַר אֲדָר וְכָתַב שֶׁל שָׁנָה רִאשׁוֹנָה, אַף עַל פִּי שֶׁלְּפִי הָאֱמֶת הִיא שָׁנָה הַשְּׁנִיָּה, לֵיכָּא לְמֵיחַשׁ כוּ', שֶׁכֵּיוָן שֶׁזֶּה הַמֶּלֶךְ לֹא מָלַךְ שׁוּם נִיסָן, וְשֵׁם הַמֶּלֶךְ הֲרֵי כָּתוּב בַּשְּׁטָר, וְנִמְצָא שֶׁלֹּא יוּכַל לִטְרֹף לָקוֹחוֹת אֶלָּא שֶׁלָּקְחוּ אַחַר הַנִּיסָן שֶׁמָּלַךְ זֶה הַמֶּלֶךְ, וּלְאוֹתוֹ כְּדִין הוּא טוֹרֵף:
(ה) עַיֵּן בְּהָרַ"ב מִשְׁנָה ה'‏ פֶּרֶק י'‏ דִּשְׁבִיעִית:
(ו) כְּלוֹמַר וְרַבִּי שִׁמְעוֹן הִיא, גְּמָרָא. וְטַעֲמַיְיהוּ דְּרַבִּי שִׁמְעוֹן וְרַבָּנָן מִקְּרָאֵי יָלְפִינַן לְהוּ בַּגְּמָרָא:
(ז) וְהַאי גָּמְרוּ הַיְנוּ שֶׁנִּגְמַר לְהִתְחַיֵּב בְּמַעֲשֵׂר, שֶׁאֵין אוֹכְלִים אֲפִלּוּ עֲרַאי, וְהוּא מִשֶּׁיִרְאֶה פְּנֵי הַבַּיִת:
(ח) גְּמָרָא. וּזְמַן לֵידַת בְּהֵמָה גַּסָּה אַף עַל גַּב דְּיוֹלֶדֶת לְתִשְׁעָה חֳדָשִׁים, נַמִּי בְּאָב הוּא, אֶלָּא שֶׁזְּמַן עִבּוּרָן קָדִים אַרְבָּעָה חֳדָשִׁים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ט) בְּתִשְׁרֵי. וְאַרְבָּעָה רָאשֵׁי שָׁנִים דְּחָשְׁבֵי לַחֲמִשָּׁה עָשָׂר בְּנִיסָן שֶׁהוּא רֹאשׁ הַשָּׁנָה לָרְגָלִים, הָכִי קָאָמַר, אַרְבָּעָה רָאשֵׁי שָׁנִים לְדִבְרֵי הַכֹּל. לְרַבִּי שִׁמְעוֹן דְּבָעֵי שְׁלֹשָׁה רְגָלִים כְּסִדְרָן דַּל מַעֲשַׂר בְּהֵמָה, וּלְדִפְלִיגֵי עֲלֵיהּ דַּל רְגָלִים. גְּמָרָא:
(י) לַשָּׁנִים כוּ'. מִקְּרָאֵי יָלְפִינַן לְהוּ בַּגְּמָרָא:
(יא) שֶׁיּוֹם הַנְּטִיעָה אֵינוֹ מִכְּלָל יְמֵי הַקְּלִיטָה:
(יב) בַּגְּמָרָא יָלֵיף לַהּ:
(יג) הָכִי יָלְפִינַן לְהוּ בַּגְּמָרָא:
(יד) כִּדְמְפָרֵשׁ בַּגְּמָרָא דְתַנָּא תָּנֵי יְרָקוֹת שֶׁמַּעַשְׂרוֹתֵיהֶן מִדְּרַבָּנָן. וְכָל שֶׁכֵּן מַה שֶּׁחַיָּבִים מִדְּאוֹרַיְתָא. וּדְתָנֵי יְרָקוֹת לְשׁוֹן רַבִּים, מִשּׁוּם דִּתְרֵי גַּוְנֵי יָרָק הֵן דְּאָזְלִינַן בָּתַר לְקִיטָה כִּדְאָמְרִינַן לְעִנְיַן גָּרְנוֹ מִשֶּׁנֶּאֱגַד [וְכוּ'‏ בְּמַסֶּכֶת מַעַשְׂרוֹת פֶּרֶק א'‏ מִשְׁנָה ה']:
(טו) וּבַגְּמָרָא אִיתָא נַמִּי וְאִם הָיְתָה שְׁנִיָּה נִכְנֶסֶת לַשְּׁלִישִׁית כוּ'‏ כִּדְבַסָּמוּךְ בְּהָרַ"ב:
(טז) דִּלְעִנְיַן עָרְלָה וּרְבָעִי כְּבָר נִתְבָּאֵר לְעֵיל:
(יז) בִּשְׁבָט. הוֹאִיל וְיָצְאוּ רֹב גִּשְׁמֵי שָׁנָה וְעָלָה הַשָּׂרָף בָּאִילָנוֹת, וְנִמְצָא הַפֵּרוֹת חוֹנְטִין מֵעַתָּה. גְּמָרָא:

=== 2 | chapter 2 ===
(א) דְּאִם לֹא כֵן חַד מִי מְהֵימָן לְהַכְשִׁירוֹ. אֲבָל לְעִנְיָן חִלּוּל הַשַּׁבָּת יָכֹל גַּם אֶחָד לְחַלֵּל כְּדֵי לְהָעִיד עָלָיו, דְּשֶׁמָּא יִצְטָרֵף עִם אַחֵר. גְּמָרָא:
(ב) מִכָּל אָדָם. מִיִּשְׂרָאֵל, שֶׁכָּל יִשְׂרָאֵל בְּחֶזְקַת כַּשְׁרוּת עַד שֶׁיִּוָּדַע לְךָ שֶׁהוּא פָּסוּל זֶהוּ דִּין תּוֹרָה. הָרַמְבַּ"ם:

=== 3 | chapter 3 ===
(א) עַד שֶׁחֲשֵׁכָה. הַיְנוּ צֵאת הַכּוֹכָבִים. רוֹצֶה לוֹמַר שְׁלֹשָׁה כּוֹכָבִים כִּדְפֵרֵשׁ הָרַ"ב בְּסוֹף פֶּרֶק ב'‏ דְּשַׁבָּת שֶׁאִם לֹא יָצְאוּ אֶלָּא שְׁנֵי כּוֹכָבִים קָרֵי לֵיהּ תַּנָּא דְּהָתָם סְפֵק חֲשֵׁכָה כוּ':
(ב) בַּגְּמָרָא דְּאָמַר קְרָא [תְּהִלִּים פא] כִּי חֹק לְיִשְׂרָאֵל הוּא מִשְׁפָּט לֵאלֹהֵי יַעֲקֹב. אֵימַת הֲוֵי חֹק, בִּגְמַר דִּין. דְּהַאי קְרָא לְעִנְיַן רֹאשׁ הַשָּׁנָה דָּרְשִׁינַן לֵיהּ, וְהוּא הַדִּין כָּל הֶחֳדָשִׁים, וְקָא קְרִי לֵיהּ מִשְׁפָּט, וּמִשְׁפָּט בַּיּוֹם:
(ג) לְשׁוֹן הָרַ"ב צָרִיךְ עִיּוּן, דְּמַשְׁמַע שֶׁבָּא לְחַלֵּק בֵּין הַלֵּילוֹת, וְהָא לֵיתָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב שֶׁהֶאֱרִיךְ:
(ד) בֵּית דִּין. פֵּרוּשׁ שֶׁל עֶשְׂרִים וּשְׁלֹשָׁה. תּוֹסָפוֹת:
(ה) וְהִלְכָּךְ גַּם לְמָחָר אֵין סוֹמְכִין עַל רְאִיָּתָן, דְּאִם לֹא כֵן הָיְתָה רְאִיָּה בַּלַּיְלָה בִּמְקוֹם קַבָּלַת עֵדוּת, וְאֵין קַבָּלַת עֵדִים אֶלָּא בַּיּוֹם כוּ'. תּוֹסָפוֹת:
(ו) וּלְהָכִי הוּא דְּאִצְטְרִיךְ לְמִתְנֵי הַךְ סֵיפָא כִּדְאִיתָא בַּגְּמָרָא:
(ז) גְּמָרָא. וְאֵין בֵּית דִּין שָׁקוּל, מוֹסִיפִין עֲלֵיהֶם עוֹד אֶחָד. תּוֹסָפוֹת:
(ח) הַשְּׁנַיִם. הָכָא שֶׁאֵין הַבֵּית דִּין אֶלָּא שְׁלֹשָׁה קָתָנֵי הַשְּׁנַיִם לְפִי שֶׁהֵם יְדוּעִים, שֶׁאֵין כָּאן אֶלָּא שְׁנַיִם וְאֶחָד, וּמִן הַסְּתָם אֵין שְׁלָשְׁתָּן שָׁוֶה וְאֶחָד גָּדוֹל בְּתוֹכָם וְהוּא נִשְׁאַר יוֹשֵׁב. מַה שֶּׁאֵין כֵּן בְּרֵישָׁא שֶׁהַבֵּית דִּין הֵם עֶשְׂרִים וּשְׁלֹשָׁה, שֶׁאֵין שְׁנַיִם מְבֹרָרִים, קָתָנֵי שְׁנַיִם:

(א) בֵּית דִּין. מִפְּנֵי שֶׁבֵּית דִּין זְרִיזִין הֵן וְלֹא יָבֹאוּ הַתּוֹקְעִים לְהַעֲבִיר הַשּׁוֹפָר בִּפְנֵיהֶם בִּרְשׁוּת הָרַבִּים, שֶׁבֵּית דִּין מַזְהִירִין אֶת הָעָם וּמוֹדִיעִים אוֹתָם. הָרַמְבַּ"ם:
