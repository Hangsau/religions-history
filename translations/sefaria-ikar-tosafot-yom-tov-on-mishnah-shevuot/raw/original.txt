(א) כְּרַבִּי עֲקִיבָא מִשְׁנָה ה'‏ פֶּרֶק ג'. גְּמָרָא:
(ב) שֶׁאֵין הַכָּתוּב מְדַבֵּר אֶלָּא בְטֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו, כִּדְמוֹכְחֵי לַהּ בַּגְּמָרָא:
(ג) וּדְקָרֵי לְהוּ יְדִיעוֹת, פֵּרֵשׁ הָרַ"ב בְּרֵישׁ פֶּרֶק ב':
(ד) כְּרַבִּי יִשְׁמָעֵאל דְּסוֹף פֶּרֶק ב'. גְּמָרָא:
(ה) עַיֵּן מַה שֶּׁכָּתַבְתִּי בְרֵישׁ פֶּרֶק קַמָּא דְשַׁבָּת. וְהָא דְהָתָם תָּנָא שְׁתַּיִם שֶׁהֵן אַרְבַּע בִּפְנִים כוּ'‏ וּשְׁתַּיִם כוּ'‏ בַּחוּץ וְאִלּוּ הָכָא לֹא תָנֵי אֶלָּא שְׁתַּיִם שֶׁהֵן אַרְבַּע, מְתָרְצִים בַּגְּמָרָא, הָתָם דְּעִקַּר שַׁבָּת תָּנָא חִיּוּבֵי וּפְטוֹרֵי, הָכָא חִיּוּבֵי תָּנָא וּפְטוֹרֵי לֹא תָּנֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

=== 2 | chapter 2 ===
(א) רַשִׁ"י. מַשְׁמַע שֶׁפֵּרֵשׁ כֻּלַּהּ מַתְנִיתִין בִּידִיעָה גְמוּרָה. שֶׁפֵּרֵשׁ וְיָדַע שֶׁנִּטְמָא כוּ'. וְהַמְדַקְדֵּק יָכוֹל לְפָרֵשׁ כֻּלַּהּ מַתְנִיתִין בִּידִיעַת בֵּית רַבּוֹ. תּוֹסָפוֹת. וְעַיֵּן פֶּרֶק קַמָּא מִשְׁנָה ג'. וְעוֹד דַּע, שֶׁעִם זֶה שֶׁצָּרִיךְ שֶׁתְּהֵא לוֹ יְדִיעָה שֶׁנִּטְמָא צָרִיךְ גַּם כֵּן שֶׁתְּהֵא לוֹ יְדִיעָה שֶׁזֶּה מִקְדָּשׁ וְשֶׁזֶּה קֹדֶשׁ:
(ב) וְהַיְנוּ טַעֲמָא נַמִּי דְּלֹא תָנֵי אַרְבַּע שֶׁהֵם שְׁמֹנֶה, וְהַיְנוּ עוֹד אַרְבַּע יְדִיעוֹת בַּסּוֹף, דְּלֹא חָשֵׁיב הָנֵי יְדִיעוֹת דְּאִיתְנְהוּ בְכָל הַתּוֹרָה כֻּלָּהּ, שֶׁאֵין אָדָם מֵבִיא חַטָּאת אֶלָּא אִם כֵּן נוֹדַע לוֹ שֶׁחָטָא, כִּדְאִיתָא בַגְּמָרָא:
(ג) בְּהֶדְיָא. אֶלָּא מֵרִבּוּיָא דִּוְנֶעְלַם דָּרְשִׁינַן לַהּ בְּסוֹף פִּרְקִין:
(ד) וְלֹא קַשְׁיָא אַמַּאי לֹא חָשֵׁיב לְהוּ וְלֶהֱוֵי שֵׁשׁ, דְּמִכֵּיוָן דְּחָשְׁבִינַן יְדִיעַת הַטֻּמְאָה דְלַקֹּדֶשׁ וְכֵן דְּלַמִּקְדָּשׁ, וְכֵן יְדִיעַת הַמִּקְדָּשׁ גַּם יְדִיעַת הַקֹּדֶשׁ, אֵין לְהוֹסִיף בְּמִנְיַן יְדִיעוֹת הֶעְלֵם שְׁנֵיהֶם יַחַד, כֵּיוָן שֶׁכְּבָר נִמְנוּ כָּל אַחַת בִּפְרָט:

=== 3 | chapter 3 ===
(א) כְּרַבִּי עֲקִיבָא בְּמִשְׁנָה ה':
(ב) הֵיכָן כוּ'. וְלֹא, וַהֲרֵי נְמָלָה. בְּרִיָּה שָׁאנֵי. וַהֲרֵי הֶקְדֵּשׁ שֶׁהוּא חַיָּב מְעִילָה בְּשָׁוֶה פְרוּטָה וַאֲפִלּוּ בְּפָחוֹת מִכַּזַּיִת, וּמְשָׁנֵי הָא בָעִינַן שָׁוֶה פְרוּטָה שֶׁזֶּהוּ שִׁעוּרוֹ. וְהָא מְפָרֵשׁ דְּכֻלֵּי עָלְמָא מוֹדוּ דְמִחַיַּב בְּכָל שֶׁהוּא. מְפָרֵשׁ נַמִּי כִּבְרִיָּה דָמֵי. פֵּרֵשׁ רַשִׁ"י כִּבְרִיָּה בִשְׁאָר אִסּוּרִים דָּמֵי. בְּרִיָּה טַעֲמָא מַאי מִשּׁוּם דְּחָשִׁיבָה, מְפָרֵשׁ נַמִּי אִיהוּ אַחַשְּׁבֵיהּ דְּאַסְרֵיהּ עֲלֵיהּ. גְּמָרָא:
(ג) בִּמְדַבֵּר. כְּלוֹמַר מְדַבֵּר וְאוֹסֵר. דְּהָא אַשְׁכְּחַן שְׁבוּעַת הָעֵדוּת, אֶלָּא דְהָתָם מְדַבֵּר וְחוֹטֵא הוּא. וְכֵן מַשְׁכַּחַת מְגַדֵּף כוּ'. גְּמָרָא. וְתוֹסָפוֹת:
(ד) מְפָרֵשׁ בַּגְּמָרָא דְּיָלְפִינַן מִקְּרָא כוּ'. וְאִי בָעֵית אֵימָא סְבָרָא, דְּאָמַר לֵיהּ אֱינִישׁ לְחַבְרֵיהּ נִטְעֹם מִידֵי וְעָיְלֵי וְאָכְלוּ וְשָׁתוּ. וּבִלְשׁוֹנָם הָווּ קָרֵי לַאֲכִילָה טְעִימָה. וְאִצְטְרִיךְ סְבָרָא לִקְרָא, דְּמַהוּ דְתֵימָא בִּנְדָרִים הַלֵּךְ אַחַר לְשׁוֹן בְּנֵי אָדָם, קָא מַשְׁמַע לָן. הָרַ"ן בְּשֵׁם בַּעַל הֲלָכוֹת גְּדוֹלוֹת:

=== 4 | chapter 4 ===
(א) שְׁבוּעַת כוּ'. שְׁבוּעָה לֹא כְתִיב בְּהֶדְיָא בַקְּרָא, אֶלָּא יָלְפִינַן מִקּוֹל, כְּמוֹ שֶׁכָּתוּב בְּרֵישׁ מַתְנִיתִין דְּסוֹף פִּרְקִין:
(ב) כְּלוֹמַר, אַף עַל גַּב דְּאֵינוֹ בַר מַלְקוֹת, שֶׁנִּתָּק לַעֲשֵׂה דְהִשָּׁבוֹן:
(ג) מִכְּלָל דְּמֶלֶךְ מִדְּאוֹרַיְתָא אֵינוֹ כָשֵׁר לְהָעִיד, וְהָכִי אִיתָא בְהֶדְיָא בַגְּמָרָא דִּמְמַעֲטִינַן מְשַׂחֵק בְּקֻבְיָא וְכָל שֶׁכֵּן מֶלֶךְ. וּכְלוֹמַר שֶׁהוּא מִדְּאוֹרַיְתָא. וּפֵרֵשׁ רַשִׁ"י, דִּכְתִיב (דְּבָרִים יז) שׂוֹם תָּשִׂים עָלֶיךָ מֶלֶךְ, שֶׁתְּהֵא אֵימָתוֹ עָלֶיךָ, וְהַבָּא לְהָעִיד צָרִיךְ לַעֲמֹד לִפְנֵי הַדַּיָּן וְדֶרֶךְ בִּזָּיוֹן הוּא לוֹ. וּלְפִי זֶה אַף בְּמַלְכֵי בֵית דָּוִד אָיְרִינַן, דְּאֵינָן רְאוּיִין לְהָעִיד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ד) מִפִּי עַצְמוֹ. דְּיָלֵיף לַהּ מִגְּזֵרָה שָׁוָה דְתֶחֱטָא תֶּחֱטָא מִפִּקָּדוֹן. וְסָבַר דּוּן מִנַּהּ. וּמִנַּהּ כְּלוֹמַר, דְּדָיְנִינַן מִנַּהּ לְחַיְּבוֹ אַף מִפִּי עַצְמוֹ, וּמִנַּהּ כִּי הָתָם דְּמִחַיְּבֵי בֵּין בְּבֵית דִּין בֵּין שֶׁלֹּא בְבֵית דִּין. דְּנִשְׁבַּע עַל שָׁקֶר כְּתִיב, מַשְׁמַע בֵּין בְּבֵית דִּין בֵּין שֶׁלֹּא בְבֵית דִּין. רַשִׁ"י:
(ה) וְכֵן אִם עָנָה אָמֵן אַחַר הַשְּׁבוּעָה. רַשִׁ"י:
(ו) אַף עַל גַּב דְּלֹא עָנָה אָמֵן. שֶׁכֵּן הוּא אוֹמֵר וְשָׁמְעָה קוֹל אָלָה אִם לוֹא יַגִּיד וְנָשָׂא עֲוֹנוֹ:
(ז) שֶׁנֶּאֱמַר וְשָׁמְעָה קוֹל אָלָה:
(ח) בֵּין כוּ'. דִּסְבִירָא לְהוּ דּוּן מִנַּהּ וְאוֹקֵי בְאַתְרַהּ. כְּלוֹמַר, לְאַחַר שֶׁהֵבֵאתָ מִפִּי עַצְמוֹ לִשְׁבוּעַת הָעֵדוּת מִפִּקָּדוֹן, אוֹקֵי בְאַתְרַהּ בְּדִין הַכָּתוּב בִּשְׁבוּעַת הָעֵדוּת, דְּמִפִּי אֲחֵרִים שֶׁמִּשְׁפָּטוֹ דַוְקָא בְבֵית דִּין, הָכִי נַמִּי מִפִּי עַצְמוֹ שֶׁהֵבֵאתָ אֵינוֹ אֶלָּא בְבֵית דִּין:

=== 5 | chapter 5 ===
(א) וּכְלוֹמַר דְּאִלּוּ לֹא הוֹדָה, מִי מְחַיְּבוֹ קָרְבָּן. וְהוּא הַדִּין בִּשְׁבוּעַת הָעֵדוּת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ב) רַשִׁ"י. וּפֵרֵשׁ הָרַשַׁ"ל דְּבָא לוֹמַר דְּמִפִּי עַצְמוֹ חַיָּב אַף בְּלֹא טַעֲנַת הַתּוֹבֵעַ. וְכִדְאִיתָא בַגְּמָרָא דַּף ל"ב:
(ג) עַד כוּ'. מִשּׁוּם דְּמִשְּׁבוּעַת הָעֵדוּת גָּמְרִינַן לַהּ, וּסְבִירָא לֵיהּ דּוּן מִנַּהּ וּמִנַּהּ, וְהָתָם בְּבֵית דִּין. רַשִׁ"י:
(ד) כֵּיוָן כוּ'. דִּסְבִירָא לְהוּ דּוּן מִנַּהּ וְאוֹקֵי בְאַתְרַהּ. רַשִׁ"י:
(ה) רַשִׁ"י. וּמַשְׁמַע, הָא שֶׁיֵּשׁ אִסּוּר בִּשְׁבוּעָה זוֹ יוֹדְעִים הֵם, וְכִדְפֵרְשׁוּ לְעִנְיַן שְׁבוּעַת הָעֵדוּת, וְסָמְכוּ אַדִּלְעֵיל. וְכֵן מַה שֶּׁכָּתַב אֲבָל זָכוּר כוּ', הוּא הַדִּין הָתָם, וְסָמַךְ אַדְּהָכָא:
(ו) דְּבִכְרֵתוֹת יָלְפִינַן לַהּ אַכָּל הָאֲשָׁמוֹת. תּוֹסָפוֹת וְהָרַ"ן:

(א) הַדַּיָּנִין. הָא דְלָא נָקַט נוֹהֶגֶת בַּאֲנָשִׁים וּבְנָשִׁים כְּדִתְנַן לְעֵיל גַּבֵּי פִקָּדוֹן, נִרְאֶה לִי מִשּׁוּם דִּשְׁבוּעַת הַדַּיָּנִין הִיא עַצְמָהּ שְׁבוּעַת הַפִּקָּדוֹן. וְזֶה לְשׁוֹן הָרַמְבַּ"ם, שֶׁדָּבָר בָּרוּר וְגָלוּי שֶׁכָּל הַנִּשְׁבָּע שְׁבוּעַת הַדַּיָּנִים אוֹ הֶסֵּת בְּשֶׁקֶר, שֶׁיְּהֵא חַיָּב מִשּׁוּם שְׁבוּעַת הַפִּקָּדוֹן:
(ב) רַשִׁ"י. פֵּרוּשׁ לְפֵרוּשׁוֹ, שֶׁאַף עַל פִּי שֶׁשְּׁבוּעַת הַפִּקָּדוֹן הוּא בְטוֹעֵן נֶאֶנְסוּ אוֹ נֶאֶבְדוּ אוֹ נִגְנְבוּ, וְכֵן שְׁבוּעַת עֵד אֶחָד שְׁבוּעַת הַדַּיָּנִין נַמִּי הִיא, הָכִי קָתָנֵי, שְׁבוּעַת הַדַּיָּנִין שֶׁל מוֹדֶה בְמִקְצָת צָרִיךְ שֶׁתְּהֵא הַטַּעֲנָה שְׁתֵּי כֶסֶף וְהַהוֹדָאָה שָׁוֶה פְרוּטָה. וְלָאו כָּל שֶׁשְּׁבוּעַת הַדַּיָּנִין קָתָנֵי. אִי נַמִּי, דִּשְׁבוּעַת הַפִּקָּדוֹן לָאו דְּדַיָּנִין הִיא, דְּשֵׁם אַחֵר יֵשׁ לָהּ. וְכֵן שְׁבוּעַת עֵד אֶחָד כֵּיוָן דְּלֹא נִכְתְּבָה בְהֶדְיָא לֵיתָא בִכְלַל זֶה. הָרַ"ן. וְטַעֲמָא דְעֵד אֶחָד לֹא בָעֵי שְׁתֵּי כֶסֶף, מִשּׁוּם דְּכָל מָקוֹם שֶׁשְּׁנַיִם מְחַיְּבִין מָמוֹן עֵד אֶחָד מְחַיְּבוֹ שְׁבוּעָה:
(ג) וְטַעֲמָא מְפֹרָשׁ בַּיְרוּשַׁלְמִי, שֶׁאָמְרוּ סוֹף מַטְבֵּעַ כֶּסֶף, מָעָה, כְּלוֹמַר כֵּיוָן שֶׁאָמְרָה תוֹרָה כֶּסֶף, כְּדִכְתִיב כִּי יִתֵּן כוּ'‏ כֶּסֶף כוּ', וְאִי אַתָּה מוֹצֵא מַטְבֵּעַ כֶּסֶף מֻזְכָּר בַּתּוֹרָה פָּחוֹת מִמָּעָה, כְּדִמְתַרְגְּמִינַן עֶשְׂרִים גֵּרָה, עֶשְׂרִים מָעִין, אֵין לְךָ לְחַיְּבוֹ בְּפָחוֹת מִמַּטְבֵּעַ זוֹ. וּמַקְשֶׁה, וּתְהֵא מָעָה, כְּלוֹמַר מִנַּיִן לְךָ שֶׁיִּהְיוּ שְׁתַּיִם. וּמְפָרֵק, אוֹ כֵלִים, מַה כֵּלִים שְׁנַיִם, אַף כֶּסֶף שְׁנַיִם. הָרַ"ן:
(ד) וְרַב הוּא דְּאָמַר הָכִי בַגְּמָרָא. וּפֵרֵשׁ רַשִׁ"י טַעֲנָה דִתְנַן בְּמַתְנִיתִין לָאו אַתְּחִלַּת טַעֲנָה קָאֵי, אֶלָּא אַכְּפִירָה שֶׁהוּא כוֹפֵר בְּטַעֲנַת חֲבֵרוֹ לְבַד הוֹדָאַת הַפְּרוּטָה, דְּבָעִינַן שֶׁתְּהֵא תְחִלַּת הַטַּעֲנָה שְׁתֵּי כֶסֶף וּפְרוּטָה. וְיָלֵיף לֵיהּ בַּגְּמָרָא, דְּכֶסֶף לָמָּה לִּי (וַהֲוָה דָרְשִׁינַן מַה כֵּלִים שְׁנַיִם אַף כָּל שְׁנַיִם), אִם אֵינוֹ עִנְיָן לְטַעֲנָה תְּנֵהוּ עִנְיָן לִכְפִירָה כוּ'. וְנִמְצָא שֶׁמַּה שֶּׁכָּתַב הָרַ"ב בִּתְחִלַּת הַמִּשְׁנָה שֶׁלֹּא תְהֵא הַטַּעֲנָה, לֹא בָא לְדַקְדֵּק בְּפֵרוּשׁ הַטַּעֲנָה מַמָּשׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) כִּדְאַשְׁכְּחַן בְּאַבְרָהָם דְּאַתְפְּסֵיהּ לֶאֱלִיעֶזֶר בִּבְרִית מִילָה, שֶׁנֶּאֱמַר (בְּרֵאשִׁית כד) שִׂים נָא יָדְךָ תַּחַת יְרֵכִי וְאַשְׁבִּיעֲךָ. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ו) לְשׁוֹן שִׂימָה. שֶׁשָּׂמוּ חֲכָמִים עָלָיו שְׁבוּעָה. כְּמוֹ אִם ה'‏ הֱסִיתְךָ (שְׁמוּאֵל א כו):
(ז) אַף עַל פִּי שֶׁהִיא נִזְכֶּרֶת בַּמִּשְׁנָה. הָרַמְבַּ"ם:
(ח) בְּשָׁוֶה פְרוּטָה. דְּמִשְׁנָתֵנוּ בִּשְׁבוּעַת הַדַּיָּנִים עַל יְדֵי הוֹדָאַת מִקְצָת הַטַּעֲנָה. כִּדְפֵרֵשׁ הָרַ"ב. וּמְסַיֵּם רַשִׁ"י, שֶׁלְּמָדוּהָ מֵאֲשֶׁר יֹאמַר כִּי הוּא זֶה. (דְּעֵרוּב פָּרָשִׁיּוֹת כְּתִיב כָּאן וְכִי הוּא זֶה דִּכְתִיב בְּפָרָשַׁת שׁוֹמְרִים שָׁדֵי אַפָּרָשַׁת אִם כֶּסֶף תַּלְוֶה) וּכְתִיב עֲלַהּ וְנִקְרַב בַּעַל הַבַּיִת אֶל הָאֱלֹהִים אִם לֹא שָׁלַח יָדוֹ, דְּהַיְנוּ שְׁבוּעָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ט) פָּטוּר. דְּכִי הוּא זֶה מַשְׁמַע מִמִּין הַטַּעֲנָה:
(י) וְאַלִּיבָּא דִשְׁמוּאֵל מְפָרְשִׁינַן הָכִי בַגְּמָרָא, וּסְבִירָא לֵיהּ דְּטַעֲנָה דְמַתְנִיתִין בִּתְחִלַּת טַעֲנָה, וְלֹא בָעִינַן שְׁתֵּי כֶסֶף וּפְרוּטָה, וְהָכָא שְׁתֵּי כֶסֶף דַּוְקָא אָמַר וְלֹא שָׁוֶה, וּלְכָךְ פָּטוּר, דְּמַה שֶּׁטְּעָנוֹ לֹא הוֹדָה לוֹ. וְכֵיצַד אַמַּאי דִּסְמִיךְ לֵיהּ קָאֵי, דְּאִם אֵין הַהוֹדָאָה כוּ'. אֲבָל לְרַב, שָׁוֶה קָאָמַר. וְכֵיצַד אָרֵישָׁא קָאֵי. וְקַשְׁיָא לְהָרַ"ב דִּמְפָרֵשׁ לָרֵישָׁא אַלִּיבָּא דְרַב, וְהָכָא (וּבְבָבָא מְצִיעָא פּ"ד מ"ז) מְפָרֵשׁ אַלִּיבָּא דִשְׁמוּאֵל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(יא) כְּמֵשִׁיב אֲבֵדָה. שֶׁהָיָה יָכוֹל לְהָעִיז פָּנָיו וְלִכְפֹּר, וּמִדְּלֹא הֵעִיז הֲרֵי הוּא כְמֵשִׁיב אֲבֵדָה. רַשִׁ"י:

(א) רֵישָׁא דִקְרָא, שְׁבוּעַת ה'‏ תִּהְיֶה בֵּין שְׁנֵיהֶם וְגוֹ'‏ וְלָקַח בְּעָלָיו [אֶת הַשְּׁבוּעָה] וְלֹא יְשַׁלֵּם, וְהַאי לֹא יְשַׁלֵּם קְרָא יְתֵרָה הוּא, דְּכֵיוָן דִּכְתִיב אִם לֹא שָׁלַח כוּ'‏ מִמֵּילָא מַשְׁמַע דִּלְאַחַר הַשְּׁבוּעָה לֹא יְשַׁלֵּם, וְאִם אֵינוֹ עִנְיָן כָּאן, תְּנֵהוּ לְמוֹדֶה בְמִקְצָת וְעֵד אֶחָד כוּ'. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ב) וְאֵלּוּ כוּ'. וְאִיכָּא עוֹד הָנָךְ דִּלְקַמָּן מִשְׁנָה ז', וּדְפֶרֶק י'‏ דְּבָבָא קַמָּא מִשְׁנָה ז', וּדְפֶרֶק ח'‏ דִּכְתֻבּוֹת מִשְׁנָה ה'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ג) וְאִי נַמִּי שְׂכָרוֹ בְעֵדִים אֶלָּא שֶׁאֵין עֵדִים שֶׁעָשָׂה עִמּוֹ מְלָאכָה:
(ד) וְאִם תֹּאמַר הָא לֹא חָשְׁדִינַן לְבַעַל הַבַּיִת דִּמְשַׁקֵּר אֶלָּא שֶׁשָּׁכַח, וְאֵיךְ שַׁיָּךְ מִגּוֹ. יֵשׁ לוֹמַר דְּהָכִי קָאָמְרִינַן, וַדַּאי קוּשְׁטָא קָאָמַר שֶׁזּוֹכֵר שֶׁפְּרָעוֹ, דְּאִי מְסָפְקָא לֵיהּ וְטוֹעֵן שֶׁקֶר שֶׁזּוֹכֵר שֶׁפְּרָעוֹ, לֹא הֲוָה אָמַר מִלְּתָא דְּיָדַע דִּמְסַפְּקֵי בָהּ רַבָּנָן מִשּׁוּם טִרְדָּא, וַהֲוָה טָעֵין דְּלֹא שְׂכָרוֹ כְלָל. הָרַ"ן. וְתוֹסָפוֹת:
(ה) וְהַיְנוּ כְּדַעַת הָרִי"ף, דְּבִקְצִיצָה לֹא תִקְנוּ כְלוּם אֶלָּא הִנִּיחוּ עַל דִּינוֹ, דְּאִם הוּא מוֹדֶה מִקְצָת נִשְׁבָּע שְׁבוּעָה דְאוֹרַיְתָא, וְאִם כּוֹפֵר בַּכֹּל נִשְׁבָּע הֶסֵּת. וְטַעֲמָא, כֵּיוָן דְּלֵיכָּא כָאן טַעֲמָא דִכְדֵי חַיָּיו. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ו) מִקְצָת כוּ'. לְרַבִּי יְהוּדָה לֹא תִקְנוּ שְׁבוּעָה לִטֹּל, אֶלָּא בְמָקוֹם שֶׁיֵּשׁ שְׁבוּעָה עַל הַנִּתְבָּע, וַהֲפָכוּהָ חֲכָמִים עַל הַתּוֹבֵעַ בְּאֵלּוּ הַשְּׁנוּיִים כָּאן. רַשִׁ"י:

=== 8 | chapter 8 ===
(א) מְשַׁלֵּם כוּ'. וְיֵשׁ עָלָיו שְׁבוּעָה שֶׁאֵינוֹ בִרְשׁוּתוֹ:
(ב) וְהַאי דַּהֲדַרַהּ וְתַנְיַהּ הָכָא, מִשּׁוּם דְּבָעֵי לְמִתְנֵי חִיּוּבֵי וּפְטוֹרֵי דְּקָרְבַּן שְׁבוּעָה דִּידְהוּ, תָּנָא בָרֵישָׁא חִיּוּבֵי וּפְטוֹרֵי דְּמָמוֹן דִּידְהוּ, לְפִי שֶׁקָּרְבַּן הַשְּׁבוּעָה בְּחִיּוּבֵי וּפְטוֹרֵי תָּלוּי בְּכָךְ כוּ', כְּדִמְפָרֵשׁ וְאָזֵיל אָמַר לְשׁוֹמֵר חִנָּם כוּ':
