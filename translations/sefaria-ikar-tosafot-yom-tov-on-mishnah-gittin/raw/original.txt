(א) גֵּט. לְשׁוֹן אֲרַמִּי. סֵפֶר כְּרִיתוּת (דְּבָרִים כד) מְתֻרְגָּם גֵּט פִּטּוּרִין. וּלְכָךְ נָהֲגוּ לִכְתֹּב בּוֹ שְׁתֵּים עֶשְׂרֵה שׁוּרוֹת כְּמִנְיַן גֵּט. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ב) וְכֵן פָּסַק רַשִׁ"י. וְהָרַ"ן פֵּרֵשׁ דְּחוּץ לָאָרֶץ שֶׁל צַד מַעֲרָבִי שֶׁל אֶרֶץ יִשְׂרָאֵל נָקַט, לְפִי שֶׁבְּאוֹתוֹ צַד לֵיכָּא עֲיָרוֹת מֻבְלָעוֹת וּסְמוּכוֹת שֶׁהֲרֵי הַיָּם מַפְסִיק, וּבִצְדָדִים אֲחֵרִים לֹא פְּסִיקָא לֵיהּ מִלְּתָא מִשּׁוּם דְּאִיכָּא עֲיָרוֹת סְמוּכוֹת וּמֻבְלָעוֹת (וּלְקַמָּן אָמְרִינַן דְּתַנָּא קַמָּא מַפִּיק אֵלּוּ מִכְּלָל). וּמִשּׁוּם הָכִי אָמַר מִמְּדִינַת הַיָּם שֶׁהִיא מַעֲרַב אֶרֶץ יִשְׂרָאֵל, וְהוּא הַדִּין לְכָל צְדָדִים אֲחֵרִים מֵאוֹתָן עֲיָרוֹת שֶׁהֵן דּוֹמוֹת לְאֵלּוּ שֶׁבִּמְדִינַת הַיָּם, שֶׁאֵינָן לֹא מֻבְלָעוֹת וְלֹא סְמוּכוֹת. וְאֶפְשָׁר שֶׁלָּזֶה כִּוֵּן רַשִׁ"י זִכְרוֹנוֹ לִבְרָכָה בְּמַה שֶּׁסִּיֵּם, כְּלוֹמַר כָּל חוּץ לָאָרֶץ שֶׁהִיא דוֹמָה לְחוּץ לָאָרֶץ שֶׁבִּמְדִינַת הַיָּם:
(ג) וּקְרָא כְּתִיב וְכָתַב לָהּ, דְּהַיְנוּ לִשְׁמָהּ. וְאֵין בְּקִיאִין הַיְנוּ דְּאֵין חוֹשְׁשִׁין לְהַךְ דְּרָשָׁה, דְּאִי לָאו הָכִי מַאי שְׁנָא מִשְּׁאָר הִלְכוֹת גִּטִּין דְּלֹא חָיְשִׁינַן לְהוּ דְּלֹא בְּקִיאֵי. תּוֹסָפוֹת:
(ד) וְרַשִׁ"י פֵּרֵשׁ דִּסְתָמָא כִּי מַסְהִיד בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם לִשְׁמָהּ קָמַסְהִיד. וְסִיֵּם הָרַ"ן דְּמֵימַר אָמַר הַאי שָׁלִיחַ לָמָּה לְהוּ לְרַבָּנָן לְמִשְׁאָל אִם נִכְתַּב וְנֶחְתַּם בְּפָנַי, שְׁמַע מִנַּהּ שֶׁיֵּשׁ כְּתִיבָה וַחֲתִימָה בְּגֵט שֶׁהִיא פְסוּלָה, וְאִלּוּ לֹא נִכְתַּב לִשְׁמָהּ הָיָה אוֹמֵר הַשָּׁלִיחַ כָּךְ אֵרַע מַעֲשֶׂה דִלְאַחֵר נִכְתַּב. וְהָא דְסַגִּי בְּעֵד אֶחָד, מִשּׁוּם דְּרֹב בְּקִיאִין הֵן. וַאֲפִלּוּ לְרַבִּי מֵאִיר דְּחָיֵישׁ לְמִעוּטָא, הָכָא הָוֵי מִעוּטָא דְמִעוּטָא, וְרַבָּנָן הוּא דְאַצְרוּךְ, וּמִשּׁוּם עִגּוּנָא הֵקֵלּוּ. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) וּפָרֵיךְ בַּגְּמָרָא אִם כֵּן בְּפָנַי נִכְתַּב לָמָּה לִי, וּמְשָׁנֵי דִּלְמָא אָתֵי לְאַחְלוֹפֵי בְּקִיּוּם שְׁטָרוֹת דְּעָלְמָא בְּעֵד אֶחָד:
(ו) דְּעֵדִים הַחֲתוּמִים עַל הַשְּׁטָר כְּמוֹ שֶׁנֶּחְקְרָה עֵדוּתָן בְּבֵית דִּין, וְרַבָּנָן הוּא דְאַצְרוּךְ, וּמִשּׁוּם עִגּוּנָא הֵקֵלּוּ. גְּמָרָא:
(ז) הַחֶגֶר. בַּגְּמָרָא מְפָרֵשׁ דִּבְהָא פְלִיגֵי, דְתַנָּא קַמָּא סָבַר דְּמֵרֶקֶם וְחֶגֶר אַף עַל פִּי שֶׁהֵן מִחוּץ לָאָרֶץ אֵין צָרִיךְ שֶׁיֹּאמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, לְפִי שֶׁהֵן עֲיָרוֹת הַסְּמוּכוֹת לְאֶרֶץ יִשְׂרָאֵל, וְכָל שֶׁכֵּן מִכְּפַר לוּדִים לְלוֹד שֶׁאֵין צָרִיךְ, לְפִי שֶׁהוּא מֻבְלָע בִּתְחוּם אֶרֶץ יִשְׂרָאֵל כוּ'. וְאָתֵי רַבָּן גַּמְלִיאֵל לָאַפְלוֹגֵי עֲלַהּ וּלְמֵימַר דְּדַוְקָא בְּמֻבְלָעוֹת כְּמוֹ מִכְּפַר לוּדִים גְּמִירֵי וּשְׁכִיחֵי, אֲבָל סְמוּכוֹת כְּגוֹן רֶקֶם לֹא גְמִירֵי וְלֹא שְׁכִיחֵי. וַאֲתָא רַבִּי אֱלִיעֶזֶר לְמֵימַר דַּאֲפִלּוּ מִכְּפַר לוּדִים שֶׁהוּא מֻבְלָע צָרִיךְ לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, דִּנְהִי דְמוֹדֶה לְרַבָּנָן קַמָּאֵי דְמֻבְלָעִים גְּמִירֵי וּשְׁכִיחֵי, אֲפִלּוּ הָכִי מַצְרִיךְ בְּהוּ לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, כְּדֵי שֶׁלֹּא תַחְלֹק בִּמְדִינַת הַיָּם. הָרַ"ן:
(ח) מִכְּפַר כוּ'. מִקְרֵי כְּפַר לוּדִים, שֶׁבְּנֵי לוֹד מְצוּיִין בָּהּ תָּמִיד. וּמִשּׁוּם הָכִי נָקַט לְלוֹד לִרְבוּתָא. הָרַ"ן:
(ט) וְהַמּוֹלִיךְ. וּבְהָא פְלִיגֵי אַתַּנָּא קַמָּא, דִּלְתַנָּא קַמָּא הַמּוֹלִיךְ מֵאֶרֶץ יִשְׂרָאֵל לְחוּץ לָאָרֶץ אֵין צָרִיךְ, לְפִי שֶׁבְּנֵי אֶרֶץ יִשְׂרָאֵל בְּקִיאִין לִשְׁמָהּ, וַחֲכָמִים גָּזְרוּ מוֹלִיךְ אַטּוּ מֵבִיא. וּלְמַאן דְּאָמַר טַעֲמָא מִשּׁוּם קִיּוּם, חֲכָמִים לִפְרוּשֵׁי מִלְּתָא דְּתַנָּא קַמָּא אָתוּ דְּמֵבִיא לָאו דַּוְקָא, אֶלָּא הוּא הַדִּין מוֹלִיךְ. גְּמָרָא:
(י) בִּמְדִינַת הַיָּם. לְטַעֲמָא דְאֵין בְּקִיאִין הוּא הַדִּין בְּאוֹתָהּ מְדִינָה נַמִּי צָרִיךְ שֶׁיֹּאמַר, אֶלָּא דְּנָקַט מִמְּדִינָה לִמְדִינָה לְמֵידַק מִנַּהּ דְּהָא מִמְּדִינָה לִמְדִינָה בְּאֶרֶץ יִשְׂרָאֵל לֹא צָרִיךְ, וַאֲפִלּוּ לְכַתְּחִלָּה. וּלְאִידָךְ טַעֲמָא נַמִּי בְּאֶרֶץ יִשְׂרָאֵל דִּקְבִיעֵי בֵית דִּין שְׁכִיחֵי שַׁיָּרוֹת. גְּמָרָא:
(יא) גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י שֶׁלֹּא יֵלֵךְ מִזּוֹ לְזוֹ. וּבְאֶרֶץ יִשְׂרָאֵל נַמִּי לְטַעֲמָא דְקִיּוּם, דְּמַקְפִּידִין. וּלְמַאן דְּאָמַר לִשְׁמָהּ נַמִּי, דְּאִיהוּ סְבִירָא לֵיהּ נַמִּי טַעֲמָא דְקִיּוּם, דְּאִלּוּ בְּחוּץ לָאָרֶץ אֲפִלּוּ בְּאוֹתָהּ הֶגְמוֹנְיָא כִדְאִיתָא בַּגְּמָרָא:
(יב) הֶגְמוֹנְיָא. פְּקִידוּת:

(א) וַאֲפִלּוּ לְמַאן דְּאָמַר מִשּׁוּם קִיּוּם, וְצָרִיךְ לוֹמַר בְּפָנַי נִכְתַּב מִשּׁוּם דִּלְמָא אָתֵי לְאַחְלוֹפֵי, בְּחֶצְיוֹ הָרִאשׁוֹן שֶׁיֵּשׁ בּוֹ שֵׁם כוּ'‏ שֶׁהוּא עִקַּר הַגֵּט מִנְכַּר מִלְּתָא וְלֹא אָתֵי לְאַחְלוֹפֵי. תּוֹסָפוֹת:
(ב) וְאֵין צָרִיךְ שֶׁיִּהְיוּ שְׁנֵיהֶם אֲדוּקִים וַחֲתוּמִים בַּגֵּט, אֶלָּא שֶׁשְּׁנֵיהֶם אוֹמְרִים שֶׁהַבַּעַל עָשָׂה שְׁנֵיהֶם שְׁלוּחִים עַל כָּךְ וְאָמַר לָהֶם שְׁנֵיכֶם הוֹלִיכוּ גֵט לְאִשְׁתִּי, וּנְתָנוֹ הָאֶחָד בְּמַעֲמַד חֲבֵרוֹ כוּ'. תּוֹסָפוֹת:
(ג) לְמַאן דְּאָמַר מִשּׁוּם קִיּוּם הָאִיכָּא שְׁנַיִם שֶׁמְּצוּיִין לְקַיְּמוֹ, וּלְמַאן דְּאָמַר מִשּׁוּם לִשְׁמָהּ, מַיְרֵי לְאַחַר שֶׁלָּמְדוּ, וּבִתְרֵי לֹא שְׁכִיחֵי, הִלְכָּךְ לֹא גָזְרִינַן שֶׁמָּא יַחֲזֹר הַדָּבָר לְקִלְקוּלוֹ:
(ד) וְאֶחָד. שְׁלִישִׁי. רַשִׁ"י:
(ה) דְּכֵיוָן דְּהַהוּא דְאָמַר בְּפָנַי נֶחְתַּם לֹא אָמַר בְּפָנַי נִכְתַּב, אָתֵי לְאַחְלוֹפֵי בְּקִיּוּם כוּ'. וּמַאן דְּאָמַר לִשְׁמָהּ, הָא סְבִירָא לֵיהּ נַמִּי טַעֲמָא דְּקִיּוּם. גְּמָרָא:
(ו) מַכְשִׁיר. כֵּיוָן דְּאִיכָּא תְּרֵי דִמְסַהֲדִי אַכְּתִיבָה, תּוּ לֹא אָתֵי לְאַחְלוֹפֵי. גְּמָרָא:
(ז) כָּשֵׁר. לְמַאן דִּסְבִירָא לֵיהּ טַעֲמָא דְקִיּוּם, אֲפִלּוּ אֵין שָׁם עֵד מֵעִיד עַל הַכְּתִיבָה כָּשֵׁר, שֶׁהֲרֵי נִתְקַיֵּם עַל פִּי שְׁנַיִם. וְלֹא נָקַט אֶחָד אוֹמֵר בְּפָנַי נִכְתַּב אֶלָּא אַגַּב רֵישָׁא דְתָנָא אֶחָד אוֹמֵר כוּ'. תּוֹסָפוֹת. וַאֲפִלּוּ יוֹצֵא מִתַּחַת יַד אֶחָד נַמִּי כָּשֵׁר. וּלְטַעֲמָא דְלִשְׁמָהּ דַּוְקָא בְּיוֹצֵא מִתַּחַת יְדֵי שְׁנֵיהֶם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

=== 3 | chapter 3 ===
(א) לְגָרֵשׁ כוּ'. כֻּלְּהוּ יָלֵיף לְהוּ בַּגְּמָרָא מִקְּרָאֵי:
(ב) אֲפִלּוּ לְמַאן דְּאָמַר בְּעָלְמָא יֵשׁ בְּרֵרָה, הָכָא מוֹדֶה, מִשּׁוּם דִּוְכָתַב לָהּ, לִשְׁמָהּ, מַשְׁמַע שֶׁיְּהֵא מְבֹרָר בִּשְׁעַת כְּתִיבָה. תּוֹסָפוֹת:
