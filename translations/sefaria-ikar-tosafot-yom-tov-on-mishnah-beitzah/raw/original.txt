(א) בַּגְּמָרָא. וְאַף עַל גַּב דְּמַסְקִינַן דְּיוֹם טוֹב דְּעָלְמָא נַמִּי אֲסוּרָה, אֲפִלּוּ הָכִי מוֹקְמִינַן לְמַתְנִיתִין בְּעִקַּר הַדִּין. וְהָא דְנָקַט יוֹם טוֹב וְלֹא שַׁבָּת וְיִהְיֶה פֵּרוּשׁוֹ שַׁבָּת שֶׁאַחַר יוֹם טוֹב, דַּהֲוָה אֲמִינָא דַּוְקָא שַׁבָּת סְעוּדָתוֹ בָּעֲיָא הֲכָנָה וְאֵין יוֹם טוֹב מֵכִין לְשַׁבָּת, אֲבָל יוֹם טוֹב שֶׁאַחַר שַׁבָּת אֵימָא לֹא לִתְּסַר, דִּכְחֹל שַׁוְוִינְהוּ שֶׁאֵין סְעוּדָתוֹ בָּעֵי הֲכָנָה, הִלְכָּךְ אִיצְטְרִיךְ לְמִתְנֵי יוֹם טוֹב, וְשַׁבָּת אָתְיָא בְּמִכָּל שֶׁכֵּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ב) גְּמָרָא. דְּאִלּוּ הַיּוֹם גָּמְרָה קַיְמָא לָן בְּעֵרוּבִין דְּשַׁבָּת מְכִינָה לְעַצְמָהּ וְהוּא הַדִּין יוֹם טוֹב. וְקָאָמַר גָּמְרָה, דְּאִלּוּ לֹא גָּמְרָה אַף עַל פִּי שֶׁנִתְגַּדֵּל לֹא מִיתַּסְרִי, דְּאִי הָכִי בְּיוֹם טוֹב אַחַר שַׁבָּת הֵיכִי שָׁחֲטִינַן, וַהֲלֹא כְּבָר נִתְגַּדְלוּ בְּשַׁבָּת, אֶלָּא דְּלֹא חָשְׁבוּ לַהּ הֲכָנָה אֶלָּא כִּי גָּמְרָה, שֶׁאָז מִתְחַדֶּשֶׁת בְּרִיָּה חֲדָשָׁה לְגַמְרֵי. הָרַ"ן:
(ג) דְּעַל כָּרְחֲךָ בָּא לְאַשְׁמְעִינַן דַּאֲפִלּוּ הֲכָנָה בִּידֵי שָׁמַיִם כְּגוֹן בֵּיצָה אֲסוּרָה, דְּאִי מִשּׁוּם הֲכָנָה בִּידֵי אָדָם כְּלוֹמַר שֶׁיֹּאפוּ וִיבַשְּׁלוּ מִבְּעוֹד יוֹם, הָא בְּהֶדְיָא כְתִיב אֶת אֲשֶׁר תֹּאפוּ אֵפוּ. רַשִׁ"י:
(ד) שֶׁכֵּן כְּתִיב בְּהוּ שַׁבָּתוֹן בְּפָרָשַׁת אֱמֹר, גַּם מִמָּחֳרָת הַשַּׁבָּת דְּעֹמֶר פֵּרְשׁוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה מִמָּחֳרָת יוֹם טוֹב:
(ה) גְּמָרָא. דְּכֵיוָן דְּשָׁמְעִינַן דְּאֵין יוֹם טוֹב מֵכִין לְשַׁבָּת, יָלְפִינַן מִקַּל וָחֹמֶר דְּאֵין שַׁבָּת מְכִינָה לְיוֹם טוֹב. הָרַ"ן:
(ו) וְאַף עַל גַּב דְּמֵאֶתְמוֹל הִיא נִגְמֶרֶת, מִכָּל מָקוֹם כְּשֶׁנּוֹלְדָה בְּשַׁבָּת הֲרֵי זוֹ הֲכָנָה, דְּגַם בִּשְׁעַת לֵידָה שַׁיָּךְ הֲכָנָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ז) שֶׁמִּתְּחִלָּה לֹא מֵחֲמַת סָפֵק הִתְחִילוּ לַעֲשׂוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ח) מַה שֶּׁאֵין כֵּן שַׁבָּת וְיוֹם טוֹב הַסְּמוּכִים, וְאַף עַל פִּי שֶׁהַיּוֹם טוֹב הוּא יוֹם טוֹב שֵׁנִי, דְּשַׁבָּת מִיהוּ וַדַּאי קֹדֶשׁ הוּא, וְיוֹם טוֹב הַסָּמוּךְ לָהּ נַמִּי שֶׁמָּא יוֹם טוֹב גָּמוּר הוּא. הָרַ"ן:
(ט) לֹא תֵאָכֵל. וַאֲפִלּוּ לְטַלְטֵל נַמִּי לֹא כוּ', אֶלָּא מִשּׁוּם דִּסְתָמָא בֵּיצָה לָאֲכִילָה קַיְמָא, נָקַט כִּי הַאי לִישָׁנָא. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(י) גְּמָרָא. וּרְצוֹנוֹ לוֹמַר לֹא יֵרָאֶה לְךָ חָמֵץ וְלֹא יֵרָאֶה לְךָ שְׂאֹר [שְׁמוֹת יג] דְּכָתַב לָאו בְּכָל חֲדָא, מַה שֶּׁאֵין כֵּן בָּאֲכִילָה, דִּנְתִינַת טַעַם, כִּי כָּל אֹכֶל כְּתִיב [שָׁם יב]:
(יא) בְּכַכּוֹתֶבֶת. דְּהוֹאִיל וְיָדַעְנוּ דְּשִׁעוּרוֹ שֶׁל זֶה לֹא כְּשִׁעוּרוֹ שֶׁל זֶה, נוֹקִים לֵיהּ שִׁעוּרָא כִּדְאַשְׁכְּחָנָא בְּעָלְמָא:
(יב) בְּכַזָּיִת. וּקְרָאֵי צְרִיכֵי דְאִי כָּתַב שְׂאֹר הֲוָה אֲמִינָא מִשּׁוּם דְּחִמּוּצוֹ קָשֶׁה כוּ', וְאִי כָּתַב חָמֵץ מִשּׁוּם דְּרָאוּי לָאֲכִילָה. גְּמָרָא:

(א) וְנִקְרָא שְׁמוֹ עֵרוּב, שֶׁהוּא מְעָרֵב צָרְכֵי שַׁבָּת עַל צָרְכֵי יוֹם טוֹב לַעֲשׂוֹתָן יַחַד. הָרַאֲבַ"ד:
(ב) וַאֲפִלּוּ בְּיוֹם טוֹב עַצְמוֹ הָיָה יָכוֹל לַעֲשׂוֹת הָעֵרוּב לִפְנֵי סְעוּדַת יוֹם טוֹב, אֶלָּא גְּזֵרָה שֶׁמָּא יִפְשַׁע וְיִשְׁכַּח וְלֹא יְעָרֵב מִשּׁוּם דְּטָרִיד. גְּמָרָא. וְלִשְׁנֵי הַפֵּרוּשִׁים שֶׁכָּתַב הָרַ"ב אֵינוֹ אֶלָּא מִדִּבְרֵי סוֹפְרִים, וְאַף עַל גַּב דַּהֲכָנָה דְּאוֹרַיְתָא, מֻתָּר לְבַשֵּׁל מִיּוֹם טוֹב לְשַׁבָּת, מִשּׁוּם הוֹאִיל וְאִי מִקַּלְעֵי אוֹרְחִים. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ג) בְּדָג וּבֵיצָה. שֶׁהָיוּ טָחִין בֵּיצָה עַל הַדָּג כְּשֶׁצּוֹלִין אוֹתוֹ. רַשִׁ"י:
(ד) וְזֶה לְשׁוֹן רַשִׁ"י, אוֹ שֶׁאָבַד קֹדֶם שֶׁבִּשֵּׁל כָּל צָרְכֵי שַׁבָּת. אִם כֵּן יֵשׁ לְפָרֵשׁ דִּבְרֵי הָרַ"ב דְּהָכִי קָאָמַר, מֵעֶרֶב שַׁבָּת קֹדֶם שֶׁבִּשֵּׁל כוּ', וְאֵין צָרִיךְ לְהַגִּיהַּ מֵעֶרֶב יוֹם טוֹב:
(ה) לֹא כוּ'. וּלְפֵרוּשׁ רִאשׁוֹן דְּטַעְמָא דְּעֵרוּב תַּבְשִׁילִין מִשּׁוּם כְּבוֹד שַׁבָּת, אֲפִלּוּ הָכִי מוּטָב שֶׁלֹּא יִתְעַדֵּן שַׁבָּת אַחַת וְיִתְעַדֵּן שַׁבָּתוֹת הַרְבֵּה. שֶׁאִם אַתָּה מַתִּיר לוֹ לְבַשֵּׁל כְּשֶׁנֶּאֱבַד עֵרוּבוֹ אַף הוּא יִתְרַשֵּׁל בּוֹ וְלֹא יִשְׁמְרֶנּוּ אוֹ יֵאָבֵד אוֹ יֵאָכֵל וְלֹא יִזְכֹּר לְשַׁבָּת כְּלָל. הָרַ"ן:
(ו) אֶלָּא דְּבַגְּמָרָא גָּרְסִינַן מַאי כָּל שֶׁהוּא, דְּאִית בֵּיהּ כַּזַּיִת, וּפֵרֵשׁ רַשִׁ"י וּלְגַבֵּי כִּכָּר שָׁלֵם קָרֵי לֵיהּ כָּל שֶׁהוּא. וְכֵן הָרַמְבַּ"ם סָתַם וְלֹא חִלֵּק בֵּין תְּחִלָּתוֹ לְסוֹפוֹ, וְכֵן כָּתַב הַטּוּר. וְצָרִיךְ עִיּוּן עַל הָרַ"ב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

=== 3 | chapter 3 ===
(א) וְטַעְמָא כְּמוֹ בִּטְחִינָה, כְּמוֹ שֶׁכָּתַבְנוּ בְּפֶרֶק קַמָּא בְּאוֹת ל'. וְכֵן צִידָה אָסְרוּ, שֶׁפְּעָמִים שֶׁתַּעֲלֶה בִּמְצוּדָתוֹ דָּגִים הַרְבֵּה כוּ'. הָרֹא"שׁ:
(ב) וּלְפִי זֶה בְּדָגִים לְעוֹלָם אֵין צָדִין, דְּאִי לֹא תֵימָא הָכִי לָמָּה חִלְּקָן. וְהַטַּעַם, מִפְּנֵי שֶׁהֵם מְכֻסִּים מִן הָעַיִן:

=== 4 | chapter 4 ===
(א) וּמְסַיֵּם רַשִׁ"י אוֹ עַל יְדֵי עֵרוּב. וְכָתַב בֵּית יוֹסֵף דְּמַשְׁמַע מִדְּבָרָיו דִּבְמוֹלִיךְ הַכַּדִּים בַּחוּץ בִּמְקוֹם הָרוֹאִים דַּוְקָא הוּא דְּאָמְרִינַן דְּלֹא יְבִיאֵם בְּסַל וּבְקֻפָּה, אֲבָל לְטַלְטֵל בִּמְקוֹם שֶׁאֵין רוֹאִין כְּגוֹן מִזָּוִית לְזָוִית בְּאוֹתוֹ בַּיִת, מֵבִיא כִּי הֵיכִי דְבָעֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ב) אוֹרְחָא דְמִלְּתָא נָקַט, דְּעַל כְּתֵפוֹ מֵבִיא אֶחָד אוֹ שְׁנַיִם, וּבְסַל וּבְקֻפָּה מֵבִיא שְׁלֹשָׁה וְאַרְבָּעָה, וְהוּא הַדִּין אִיפְּכָא, דִּלְעוֹלָם אִסּוּרָא בְּסַל וְקֻפָּה וְהַהֶתֵּר בִּכְתֵפוֹ, דִּבְשִׁנּוּי תַּלְיָא מִלְּתָא, לֹא בַּמִּסְפָּר:
(ג) וְשָׁם נוֹתְנִין עֵצִים וְכָל דָּבָר שֶׁאֵין דַּעְתּוֹ לִטֹּל עַד זְמַן מְרֻבֶּה. רַשִׁ"י:

=== 5 | chapter 5 ===
(א) לִישְׁנָא דְּכִי יִשַּׁל זֵיתֶךָ [דְּבָרִים כח]. גְּמָרָא:
(ב) וְכָתַב הָרֹא"שׁ, וְיוֹתֵר נִרְאֶה כְּדִבְרֵי רִ"י דִּבְשַׁבָּת דְּקָתָנֵי בַּסֵּיפָא קָאֵי אַהָנֵי תְּרֵי בָּבֵי, דְּמִדְּקָתָנֵי רֵישָׁא בְּיוֹם טוֹב אֲבָל לֹא בְּשַׁבָּת וּבַסֵּיפָא קָתָנֵי בְּשַׁבָּת, מַשְׁמַע דְּקָאֵי אַתְּרֵי בָּבֵי:
(ג) כַּדֵּי יַיִן כוּ'. מוֹקְמִינַן בַּגְּמָרָא בִּדְטִבְלָא, דְּאָסוּר לְטַלְטְלָן, וְהָא קָא מַשְׁמַע לָן דְּלֹא תֵימָא בְּפֵרוֹת הָרְאוּיִן לְטַלְטֵל דַּוְקָא קָא מַשְׁמַע לָן, דַּאֲפִלּוּ בְּאֵין רְאוּיִן, דִּכְלִי נִטַּל לְצֹרֶךְ דָּבָר שֶׁאֵינוֹ נִטַּל. וּלְמַאן דְּאָמַר שֶׁאֵינוֹ נִטַּל, מוֹקְמִינַן דַּוְקָא בְּפֵרוֹת הָרְאוּיִן, וְהָדַר תָּנֵי כַּדֵּי יַיִן, לְאַשְׁמְעִינַן דַּאֲפִלּוּ בְּהֶפְסֵד מֻעָט הִתִּירוּ:
(ד) הַדֶּלֶף. דַּוְקָא בְּדֶלֶף הָרָאוּי לִרְחִיצָה, דְּאִי בְּשֶׁאֵינוֹ רָאוּי אָסוּר, דִּמְבַטֵּל כְּלִי מֵהֵיכָנוֹ. הָרַ"ן:
