(א) וְהֵ"א מִתְחַלֵּף בְּחֵי"ת. רַשִׁ"י. שֶׁשְּׁנֵיהֶם מִמּוֹצָא הַגָּרוֹן. וְעָיֵף הַיְנוּ לְמַיִם, כְּדִכְתִיב [מִשְׁלֵי כה] מַיִם קָרִים עַל נֶפֶשׁ עֲיֵפָה:
(ב) בַּמּוֹעֵד. הַמִּשְׁנָה תִּקְרָא [מוֹעֵד, מַה שֶּׁאָנוּ קוֹרִים] חֻלּוֹ שֶׁל מוֹעֵד, [וְהֵם] הַיָּמִים שֶׁבֵּין מוֹעֵד לְמוֹעֵד. הָרַמְבַּ"ם:
(ג) רַשִׁ"י. וְהָרִי"ף וְהָרַמְבַּ"ם וְהָרֹא"שׁ וְהַטּוּר פֶּה אֶחָד הִסְכִּימוּ דְּאֵין לְחֻלּוֹ שֶׁל מוֹעֵד עִקָּר מִן הַתּוֹרָה. וּבְרַיְתָא דַּחֲגִיגָה דְּלַמְּדוּהָ מִקְּרָא אֵינוֹ אֶלָּא אַסְמַכְתָּא. וְהַבֵּית יוֹסֵף כָּתַב שֶׁנִּרְאֶה לוֹ מִזֹּאת הַבְּרַיְתָא שֶׁיֵּשׁ לוֹ עִקָּר מִן הַתּוֹרָה, וְהַכָּתוּב מְסָרָהּ לַחֲכָמִים לֶאֱסֹר וּלְהַתִּיר כְּפִי שֶׁנִּרְאֶה לָהֶם:
(ד) אֲבָל הָרַמְבַּ"ם נִרְאֶה שֶׁאֵין סוֹבֵר כָּךְ, אֶלָּא דְּבֵית הַבַּעַל נַמִּי אָסוּר בִּשְׁבִיעִית. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) כְּלוֹמַר שֶׁאֲפִלּוּ דָּבָר הָאָבֵד לֹא הִתִּירוּ כְּשֶׁיֵּשׁ טִרְחָא:
(ו) שֶׁלֹּא. וְאִצְטְרִיךְ לְמִתְנֵי לְמִעוּטֵי שְׂדֵה הַבַּעַל, שֶׁאֲפִלּוּ מִמַּעְיָן שֶׁלֹּא יָצָא בִּתְחִלָּה אָסוּר. גְּמָרָא:
(ז) אֵין מַשְׁקִין. אַמּוֹעֵד קָאֵי, דְּגַבֵּי שְׁבִיעִית לֹא חָיְישִׁינַן לְטִרְחָא יְתֵרָא, וְטַעְמָא מְפָרֵשׁ בַּיְרוּשַׁלְמִי דִּשְׁבִיעִית עַל יְדֵי שֶׁהוּא מֻתָּר בִּמְלָאכָה שֶׁאֵינָהּ דְּקַרְקַע הִתִּירוּ אֲפִלּוּ בְּדָבָר שֶׁל טֹרַח, אִי נַמִּי שְׁבִיעִית עַל יְדֵי שֶׁזְּמַנָּהּ מְרֻבֶּה כוּ'‏ וְלֹא מִסְתַּבֵּר לַעֲשׂוֹת שִׁבְעָה יָמִים הָאַחֲרוֹנִים כְּשִׁבְעָה דְּרֶגֶל:
(ח) הַקִּילוֹן. כֵּיוָן דְּמֵי הַגְּשָׁמִים לֹא נֶאֶסְרוּ אֶלָּא מִשּׁוּם גְּזֵרָה דְּמֵי הַקִּילוֹן, הִלְכָּךְ שַׁפִּיר נַקְטֵיהּ אַחַר מֵי גְּשָׁמִים. הָרַ"ן:
(ט) וּמִלְּשׁוֹן קוּלְתָּא שֶׁהוּא תִּרְגּוּם כַּדָּהּ עַל שִׁכְמָה, נִקְרֵאת קִילוֹן. הָרַ"ן:
(י) וְאֵין עוֹשִׂין כוּ'. מִדְּרֵישָׁא דְּהַךְ אֲבָל אֵין מַשְׁקִין לֹא קָאֵי אַשְּׁבִיעִית, הַךְ סֵיפָא נַמִּי לֹא קָאֵי אַשְּׁבִיעִית. הָרַמְבַּ"ם:
(יא) כְּדֵי שֶׁיִּתְעַכְּבוּ שָׁם הַמַּיִם, מִלְּשׁוֹן [רוּת א] הֲלָהֶן תֵּעָגֵנָה, שֶׁהוּא לְשׁוֹן עִכּוּב. הָרַ"ן:
(יב) לִישְׁנָא דִּבְרַיְתָא, וּבְאַתְרֵיהּ דְּתַנָּא דְּמַתְנִיתִין שְׁכִיחֵי גְּפָנִים וְלֹא זֵיתִים:

=== 2 | chapter 2 ===
(א) כְּלוֹמַר דְּפָתַח [בְּמַתְנִיתִין] בְּאָבֵל, וְלֹא פֵּרֵשׁ אִם יָכֹל לִטְעֹן קוֹרָה רִאשׁוֹנָה בִּימֵי אֶבְלוֹ, אֶלָּא מִלֵּי דְּמוֹעֵד קָא פָּרֵישׁ, דְּאִם אֵרַע לוֹ אֹנֶס קֹדֶם הַמּוֹעֵד דְּטוֹעֵן קוֹרָה רִאשׁוֹנָה. גְּמָרָא:
(ב) וְהוּא מִלְּשׁוֹן מְגֻפָה. וְכָתְבוּ הַתּוֹסָפוֹת יֵשׁ לוֹמַר דְּצָרִיךְ לְכַסּוֹתוֹ מִפְּנֵי הַשְּׁרָצִים. וּמִיהוּ מְגֻפָה לֹא הָיָה צָרִיךְ, דְּדַיּוֹ בְּכִסּוּי מֻעָט, וְשֶׁמָּא חָדָשׁ מִתְקַלְקֵל:

=== 3 | chapter 3 ===
(א) כְּגוֹן שֶׁלֹּא בָּא לַיִּשּׁוּב עַד חֹל הַמּוֹעֵד, אוֹ שֶׁלֹּא הָיָה לוֹ שְׁהוּת לְגַלֵּחַ קֹדֶם הָרֶגֶל כִּי כְּשֶׁבָּא לֹא הָיָה לוֹ שְׁהוּת בַּיּוֹם. בֵּית יוֹסֵף:
(ב) וּבְרֵישׁ פֶּרֶק ט'‏ דִּנְדָרִים כָּתַב דְּקָיְמָא לָן דְּאֵינוֹ צָרִיךְ לִפְתֹּחַ כְּשֶׁמִּתְחָרֵט מֵעַצְמוֹ, וְכֵיוָן שֶׁכֵּן נִמְצָא כִּי לֹא נִתְחָרֵט קֹדֶם הָרֶגֶל דִּבְמִתְכַּוֵּן לֹא גִּלֵּחַ וְלִתְּסַר, וְאִם כֵּן לֹא הֲוָה לֵיהּ לְהָרַ"ב לִכְתֹּב טַעַם זֶה. וּמִדַּעַת הָרִי"ף וְהָרַמְבַּ"ם נִרְאֶה דְּכָל שֶׁהָיָה בְּדַעְתּוֹ לְהַשְׁלִים נִדְרוֹ וְנִמְלַךְ בָּרֶגֶל וְהִתִּירוּ לוֹ חָשׁוּב אֹנֶס:
(ג) וְאִם תֹּאמַר פְּשִׁיטָא. וְיֵשׁ לוֹמַר דְּלֹא גָּזְרִינַן דִּלְמָא אָתֵי לְשַׁהוֹיֵי קָרְבְּנוֹתָיו עַד הָרֶגֶל שֶׁהוּא בָּטֵל מִמְּלָאכָה. הָרַ"ן. אֲבָל בַּגְּמָרָא מַסִּיק דְּנָזִיר וּמְצֹרָע אַף עַל פִּי שֶׁהָיָה לָהֶם פְּנַאי מֻתָּרִין:
(ד) דְּאִי לָאו הָכִי אַף עַל גַּב דְּגִלּוּחַ מְלָאכָה הוּא הָיָה לָנוּ לְהַתִּירָהּ לְצֹרֶךְ הַמּוֹעֵד. תּוֹסָפוֹת:
