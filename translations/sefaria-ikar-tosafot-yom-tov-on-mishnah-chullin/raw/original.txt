(א) דְּוַדַּאי לֹא אָתָא לְאַשְׁמְעִינַן אֶלָּא אַמַּאי דִמְרַבִּינַן. דִּבְמִי דִפְשִׁיטָא לָן שֶׁשּׁוֹחֵט, מַאי אָתֵי לְמֵימְרָא, קְרָא כְּתִיב וְזָבַחְתָּ:
(ב) הַנְהוּ דְכָתַב בְּרֵישׁ פֶּרֶק ב'‏ דְּהִלְכָתָא גְּמִירֵי לְהוּ:
(ג) לְשׁוֹן עֲיֵפוּת וְחַלִּישׁוּת הַלֵּב, שֶׁאֵין יָכוֹל לִרְאוֹת מַכַּת חֶרֶב וְסַכִּין אֲפִלּוּ בִּבְהֵמָה. כְּמוֹ וְתִתְעַלַּפְנָה הַבְּתוּלֹת [עָמוֹס ח]. רַשִׁ"י:
(ד) וּמִיהוּ אַדִּידְהוּ נַמִּי קָאֵי, דְּדַוְקָא בְּדִיעֲבַד מַכְשִׁירִין שְׁחִיטָתָן, אֲבָל לְכַתְּחִלָּה לֹא אֲפִלּוּ בַאֲחֵרִים עוֹמְדִים עַל גַּבֵּיהֶם. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) רַשִׁ"י. וּמְסַיֵּם הָרַ"ן, שֶׁאֵין שְׁחִיטָה אֶלָּא בְּיִשְׂרָאֵל. אֲבָל נָכְרִי כֵּיוָן דְּלָאו בְּתוֹרַת שְׁחִיטָה הוּא, כְּשֶׁשְּׁחָטָהּ הַוְיָא לָהּ כְּמֵתָה מֵאֵלֶיהָ. הָרַ"ן. וּבְבֵית יוֹסֵף בְּשֵׁם רִ"י, מִשּׁוּם דִּכְתִיב וְזָבַחְתָּ וְאָכַלְתָּ אוֹתוֹ, שֶׁהוּא בַּר זְבִיחָה אֱכֹל מִזִּבְחוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ו) כְּלוֹמַר כְּתִקְרֹבֶת עֲבוֹדָה זָרָה דַּאֲסוּרָה בַהֲנָאָה, כְּדִתְנַן בְּמִשְׁנָה ג'‏ פֶּרֶק ב'‏ דַּעֲבוֹדָה זָרָה:
(ז) וְזֶה לְשׁוֹן הָרַמְבַּ"ם, לְפִי שֶׁעוֹבְדֵי עֲבוֹדָה זָרָה חֲלוּקִים לִשְׁנֵי חֲלָקִים, הָאֶחָד מֵהֶן הַיּוֹדְעִין לַעֲשׂוֹת אוֹתָהּ, רְצוֹנוֹ לוֹמַר לְהַבִּיט אֶל הַמַּזָּלוֹת הַצּוֹמְחִים לְצֹרֶךְ מְלַאכְתָּם, וּלְהוֹרִיד הָרוּחָנִיּוּת בָּהֶן וּשְׁאָר אוֹתָן שְׁטֻיּוֹת הַמְטַנְּפִים הַשֵּׂכֶל כוּ'. וְהַחֵלֶק הַשֵּׁנִי, הֵן הָעוֹבְדִים לְאוֹתָן הַצּוּרוֹת הַיְדוּעוֹת כְּפִי מַה שֶּׁלָּמְדוּ מֵחַכְמֵיהֶם בִּלְבָד. וְכֵן הֵם רֹב עוֹבְדֵי עֲבוֹדָה זָרָה. וְהַחֵלֶק הַזֶּה הָאַחֲרוֹן עָלָיו אָמְרוּ חֲכָמִים זַ"ל בַּלָּשׁוֹן הַזֶּה, נָכְרִים שֶׁבְּחוּץ לָאָרֶץ לָאו עוֹבְדֵי עֲבוֹדָה זָרָה אֶלָּא מַעֲשֵׂה אֲבוֹתֵיהֶם בִּידֵיהֶם. הָרַמְבַּ"ם:
(ח) וּכְרַבִּי יְהוּדָה בֶּן בְּתֵירָא דְּתַנְיָא כוּ'‏ מִנַיִן לְתִקְרֹבֶת עֲבוֹדָה זָרָה שֶׁהִיא מְטַמְּאָה בְאֹהֶל, שֶׁנֶאֱמַר [תְּהִלִּים קו] וַיִּצָּמְדוּ לְבַעַל פְּעוֹר וַיֹּאכְלוּ זִבְחֵי מֵתִים, מַה מֵּת מְטַמֵּא בְאֹהֶל אַף תִּקְרֹבֶת עֲבוֹדָה זָרָה מְטַמְּאָה בְאֹהֶל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ט) דְּאִלּוּ בְּמֵזִיד, לֹא יֵאָכֵל עוֹלָמִית. כִּדְאַשְׁכְּחָן בִּמְבַשֵּׁל בְּשַׁבָּת בְּמֵזִיד בְּמִשְׁנָה ג'‏ פֶּרֶק ב'‏ דִּתְרוּמוֹת, וּבְפֵרוּשׁ הָרַ"ב שָׁם, דְּלַאֲחֵרִים מֻתָּר לְמוֹצָאֵי שַׁבָּת. וְכֵן כָּתְבוּ הָרַמְבַּ"ם וְהַטּוּר. וְלִכְאוֹרָה בִּשְׁחִיטָה נַמִּי דִּינָא הָכִי. וּמִכָּל מָקוֹם מַתְנִיתִין לֹא מִתּוֹקְמָא בְּמֵזִיד, דִּשְׁחִיטָתוֹ כְשֵׁרָה קָתָנֵי לֹא שְׁנָא לוֹ וְלֹא שְׁנָא לַאֲחֵרִים כוּ'. וְכִי תֵימָא אַמַּאי כָּשֵׁר לַאֲחֵרִים הָא הֲוֵי מְחַלֵּל שַׁבָּת, דְּפֵרֵשׁ הָרַ"ב בַּמִּשְׁנָה דִלְקַמָּן דִּשְׁחִיטָתוֹ פְסוּלָה, יֵשׁ לוֹמַר דִּלְקַמָּן מַיְרֵי בְּפַרְהֶסְיָא וְהָכָא בְּצִנְעָא. וְעוֹד יֵשׁ לוֹמַר דְּמִשּׁוּם פַּעַם אַחַת לָא חֲשִׁיב מוּמָר. תּוֹסָפוֹת. אֲבָל בְּהָרַמְבַּ"ם וְהַטּוּר מַשְׁמַע דִּבְמֵזִיד אָסוּר אַף לַאֲחֵרִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(י) גְּמָרָא, מַאי טַעֲמָא, כֵּיוָן דְּאִי אֶפְשָׁר לְכַזַּיִת בָּשָׂר בְּלֹא שְׁחִיטָה, אַדַּעְתָּא דְחוֹלֶה קָא שָׁחַט. פֵּרֵשׁ רַשִׁ"י, וְלֵיכָּא לְמִגְזַר שֶׁמָּא יַרְבֶּה בַּשְּׁחִיטָה בִּשְׁבִיל בָּרִיא, דְּהָא מִשּׁוּם הַהוּא זַיִת דְּחוֹלֶה בָּעֵי לְמִשְׁחָט כֻּלָּהּ:
(יא) בְּנַפְשׁוֹ. בְּשַׁבָּת סְקִילָה. וּבְיוֹם הַכִּפּוּרִים כָּרֵת. הָרַ"ן:

(א) דִּלְכַתְּחִלָּה הִצְרִיכוּהוּ רַבָּנָן תַּרְוַיְהוּ מִשּׁוּם הַרְחָקָה לַעֲבֵרָה דִּלְמָא לֹא אָתֵי לְמֶעְבַּד רֻבָּא דְחַד. רַשִׁ"י:
(ב) דְּלָאו בְּנֵי שְׁחִיטָה כְּלָל נִינְהוּ, מִשּׁוּם דִּכְתִיב אֲסֵפָה דִּידְהוּ בִּמְקוֹם שְׁחִיטָה דְּאַחֲרִינֵי, כְּדִכְתִיב [בְּמִדְבָּר יא] הֲצֹאן וּבָקָר יִשָּׁחֵט לָהֶם וּמָצָא לָהֶם אִם אֶת כָּל דְּגֵי הַיָּם יֵאָסֵף לָהֶם, וּמִדְּשָׁנֵי קְרָא בְּדִבּוּרֵיהּ וְכָתַב בְּהוּ אֲסֵפָה, שְׁמַע מִנַּהּ דַּוְקָא הוּא. גְּמָרָא:
(ג) שֶׁל אֶחָד. רֻבּוֹ שֶׁל כָּל אֶחָד וְאֶחָד. רַשִׁ"י:
(ד) לָאו מִקְּרָא יָלֵיף. אֶלָּא כְּלוֹמַר עַל הִלְכוֹת שְׁחִיטָה נִצְטַוָּה. רַשִׁ"י. וְהַתּוֹסָפוֹת כָּתְבוּ דְּיֵשׁ מְפָרְשִׁים בְּגִימַטְרִיָּא דְּכַאֲשֶׁר, א'‏ אֶחָד בָּעוֹף, ש'‏ שְׁנַיִם בַּבְּהֵמָה, ר'‏ רֻבּוֹ שֶׁל אֶחָד כָּמוֹהוּ, רָאשֵׁי תֵבוֹת דְּכַאֲשֶׁר לְמַפְרֵעַ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) חֲצִי אֶחָד כוּ'. הוּא הַדִּין חֲצִי כָל אֶחָד וְאֶחָד. וּמֵרֵישָׁא נַמִּי שְׁמִיעַ לֵיהּ דְּדַוְקָא רֹב בָּעִינַן. וְלֹא נָקְטֵיהּ אֶלָּא אַגַּב דְּבָעֵי לְמִתְנֵי אֶחָד וָחֵצִי בַּבְּהֵמָה, אַף עַל גַּב דַּהֲווּ טְפֵי מֵרֹב הַשְּׁנַיִם בֵּין הַכֹּל, שְׁחִיטָה פְסוּלָה. תּוֹסָפוֹת:

(א) אֵלּוּ טְרֵפוֹת. טְרֵפָה הָאֲמוּרָה בַּתּוֹרָה, זוֹ שֶׁטָּרְפָה אוֹתָהּ חַיַּת הַיַּעַר, כְּגוֹן אֲרִי. וְאֵין אַתָּה יָכוֹל לוֹמַר שֶׁטָּרְפָה אוֹתָהּ וְהֵמִיתָה אוֹתָהּ, שֶׁאִם מֵתָה הֲרֵי הִיא נְבֵלָה כוּ', הָא אֵינוֹ מְדַבֵּר אֶלָּא בְּשֶׁנִּטְרְפָה וְלֹא מֵתָה. יָכוֹל אִם בָּא זְאֵב וְגָרַר גְּדִי בְּרַגְלוֹ אוֹ בְּאָזְנוֹ וְרָדַף אָדָם וְהִצִּילוֹ מִפִּיו יִהְיֶה אָסוּר שֶׁהֲרֵי נִטְרַף, תַּלְמוּד לוֹמַר וּבָשָׂר בַּשָּׂדֶה טְרֵפָה וְגוֹ', עַד שֶׁיַּעֲשֶׂה אוֹתוֹ בָּשָׂר הָרָאוּי לַכֶּלֶב, הָא לָמַדְתָּ שֶׁטְּרֵפָה הָאֲמוּרָה בַּתּוֹרָה הִיא שֶׁטָּרְפָה אוֹתָהּ חַיַּת הַיַּעַר וְשָׁבְרָה אוֹתָהּ וְנָטָה לָמוּת וַעֲדַיִן לֹא מֵתָה. אַף עַל פִּי שֶׁקָּדַם וּשְׁחָטָהּ קֹדֶם שֶׁתָּמוּת הֲרֵי זוֹ אֲסוּרָה מִשּׁוּם טְרֵפָה, הוֹאִיל וְאִי אֶפְשָׁר שֶׁתִּחְיֶה מִמַּכָּה זוֹ הַבָּא עָלֶיהָ. נִמְצֵאתָ לָמֵד שֶׁהַתּוֹרָה אָסְרָה הַמֵּתָה וְהִיא הַנְּבֵלָה, וְאָסְרָה הַנּוֹטָה לָמוּת וְהִיא הַטְּרֵפָה. וּכְשֵׁם שֶׁלֹּא תַחֲלֹק בְּמֵתָה בֵּין מֵחֲמַת עַצְמָהּ עַל יְדֵי חֹלִי בֵּין עַל יְדֵי אֵיזֶה סִבָּה, כָּךְ לֹא תַחֲלֹק בְּנוֹטָה לָמוּת כוּ'‏ בֵּין שֶׁהָיָה הַגּוֹרֵם בִּידֵי בָשָׂר וָדָם בֵּין בִּידֵי שָׁמַיִם. אִם כֵּן לָמָּה נֶאֱמַר טְרֵפָה, דִּבֵּר הַכָּתוּב בְּהֹוֶה כוּ'. וְעִנְיַן הַכָּתוּב שֶׁהַנּוֹטָה לָמוּת מֵחֲמַת מַכּוֹתֶיהָ וְאִי אֶפְשָׁר לָהּ לִחְיוֹת מֵחֲמַת מַכָּה זוֹ אֲסוּרָה. הָרַמְבַּ"ם:
(ב) נְקוּבַת הַוֶּשֶׁט וּפְסוּקַת הַגַּרְגֶּרֶת. מַסִּיק בַּגְּמָרָא דְּמִשְׁנָה זוֹ שְׁנָאָהּ רַבִּי עֲקִיבָא קֹדֶם שֶׁהוֹדָה לְרַבִּי יְשֵׁבָב:
(ג) הִיא הָרְצוּעָה הֶעָבָה הַמְחַבֶּרֶת אֶת הַטַּבָּעוֹת:
(ד) חֲלָלוֹ. בִּשְׁאָר נְקוּבֵי לֹא אִצְטְרִיךְ לְמִתְנֵי לְבֵית חֲלָלוֹ, שֶׁהֵן דַּקִּין וּפְשִׁיטָא שֶׁאֵין נֶקֶב חָשׁוּב בָּהֶן כְּלָל אִם לֹא נִקַּב לְבֵית חֲלָלוֹ. אֲבָל לֵב שֶׁהוּא עָב סָלְקָא דַעְתִּין שֶׁאִם נִקַּב בְּעֹמֶק שֶׁהוּא חָשׁוּב נֶקֶב כוּ'. וּבְכָל הָנֵי נְקוּבֵי דִּתְנַן, פְּשִׁיטָא דְּכָל שֶׁכֵּן נִטַּל כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ה) שֶׁחָסְרָה. בַּגְּמָרָא פָּרֵיךְ, כֵּיוָן דְּתָנָא נִקְּבָה לָמָּה לִי לְמִתְנֵי חָסְרָה. וּמְשָׁנֵי, דִּלְרַבִּי שִׁמְעוֹן אִצְטְרִיךְ לְאַשְׁמְעִינַן, דַּאֲפִלּוּ רַבִּי שִׁמְעוֹן מוֹדֶה בְּחָסְרָה דְּלֹא בָעֵי שֶׁתִּנָּקֵב לְבֵית הַסִּמְפּוֹנוֹת:
(ו) הָרַמְבַּ"ם. וְטַעֲמַיְהוּ, שֶׁהַמָּרָה שֵׁם לַלֵּחָה שֶׁבַּכִּיס. וַהֲוָה לֵיהּ לְפָרֵשׁ כֵּן גַּם כֵּן בַּקֵּבָה:
(ז) שֶׁנִּקַּב בְּאוֹתוֹ צַד שֶׁבֵּינוֹ לַחֲבֵרוֹ, שֶׁלֹּא נִקַּב חִיצוֹן שֶׁבְּכֻלָּם. וְהַדְרָא דְכַנְתָּא, דַּקִּים הַסּוֹבְבִים אוֹתוֹ סָבִיב כַּעֲגֻלָּה:
(ח) הַמְסֵס. לְפִי שֶׁמְּעַכֶּלֶת הַמַּאֲכָל נִקְרָא הַמְסֵס. כְּמוֹ וְהָיָה כִּמְסוֹס נוֹסֵס [יְשַׁעְיָה י]. עָרוּךְ, וְהַהֵ"א שִׁמּוּשִׁית וּבְפַתַּ"ח:
(ט) וּבֵית הַכּוֹסוֹת. צָרִיךְ עִיּוּן, דִּלְפִי סִדְרָן הֲוָה לֵיהּ לְמִתְנֵי בֵּית הַכּוֹסוֹת וְאַחַר כָּךְ הַמְסֵס:
(י) מַשְׁמַע דְּאִלּוּ נִקְּבוּ מֵחֲלָלוֹ שֶׁל זֶה לַחֲלָלוֹ שֶׁל זֶה אֲפִלּוּ בִּמְקוֹם חִבּוּרָן טְרֵפָה. וְדִבְרֵי תֵימַהּ הֵן, דְּהָא לְקַמָּן תְּנַן בְּהֶדְיָא דְּבִכְהַאי גַוְנָא כָּשֵׁר כוּ'. וּלְשׁוֹן רַשִׁ"י, לַאֲפוֹקֵי שֶׁנִּקְּבוּ בִּמְקוֹם חִבּוּרָן בִּשְׁתֵּי דָפְנוֹתֵיהֶן וְהַנֶּקֶב הוֹלֵךְ מֵחֲלָלוֹ שֶׁל זֶה לַחֲלָלוֹ שֶׁל זֶה:
(יא) שֶׁרִסּוּק אֵבָרִים מִכְּלַל הַטְּרֵפוֹת, עַל מְנָת שֶׁאֵרַע מֵחֲמַת הַנְּפִילָה. הָרַמְבַּ"ם. וְכָתַב רַשִׁ"י, וְאֵין טַעַם בַּטְּרֵפוֹת, שֶׁהֲלָכָה לְמֹשֶׁה מִסִּינַי הֵן:
(יב) וְסוֹפָהּ לָמוּת, לֹא דְסוֹפָהּ לִנָּקֵב, דְּאִם כֵּן בִּכְלַל נְקוּבָה הִיא. תּוֹסָפוֹת:
(יג) הַנֵּץ. גַּבֵּי טְרֵפוֹת עוֹף הֲוָה שַׁיָּךְ לְמִתְנְיֵהּ, אֶלָּא מִשּׁוּם רַבִּי יְהוּדָה תָּנָא לֵיהּ הָכָא. תּוֹסָפוֹת:
(יד) וְאֵיזֶהוּ לִקּוּי, כָּל שֶׁנִּתְמַסְמֵס הַבָּשָׂר עַד שֶׁאִם אָחַז אָדָם בְּמִקְצָתוֹ מִתְמַסְמֵס וְנוֹפֵל. טוּר:
(טו) שֶׁהַגִּידִים מְעוּרִים שָׁם, וְהוּא אֶמְצָעִית הַכֻּלְיָא. רַשִׁ"י. וּמְפֹרָשׁ בָּרַשְׁבָּ"א יוֹתֵר, שֶׁכָּתַב וּמְקוֹם חָרִיץ הֵיכָן, לֹבֶן שֶׁתַּחַת הַמָּתְנַיִם שֶׁנִּכְנַס בְּתוֹךְ הַכֻּלְיָא:
(טז) וְהַצֶּלַע שֶׁכְּנֶגְדָּהּ מְחֻבֶּרֶת יָפֶה בַּחֲצִי חֻלְיָא הַקַּיֶּמֶת. רַשִׁ"י:
(יז) אֶלָּא מָקוֹם שֶׁאֵין בּוֹ מִילַת. גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י, מָקוֹם יֵשׁ בַּכֶּרֶס חָלָק שֶׁאֵין שָׁם אוֹתוֹ צֶמֶר שֶׁגּוֹרְרִין בְּרוֹתְחִין מִן הַכֶּרֶס:
(יח) בְּלֹא צֶלַע. וּמַשְׁכַּחַת לָהּ בְּשִׁלְהֵי כִּפְלֵי יֵשׁ חֻלְיוֹת הַרְבֵּה בְּלֹא צֶלַע. דְּאִי בִּדְיֵשׁ בָּהּ צֶלַע תֵּפּוּק לֵיהּ מִשּׁוּם נֶעְקְרָה צֶלַע. גְּמָרָא:
(יט) כָּמוֹהָ. בִּבְהֵמָה שֶׁשְּׁחָטָהּ אֲנַן קַיְמִין. לְהָכִי לֹא תְנַן שֶׁאֵינָהּ חַיָּה. תּוֹסָפוֹת:
(כ) חַיָּה. דִּכְתִיב זֹאת הַחַיָּה אֲשֶׁר תֹּאכֵלוּ, חַיָּה אֱכֹל [זֹאת מִעוּט הוּא], שֶׁאֵינָהּ חַיָּה לֹא תֵיכוֹל. גְּמָרָא:

(א) הַמַּקְשָׁה כוּ'. אוֹרְחָא דְמִלְּתָא נָקַט, שֶׁעַל יְדֵי לֵידַת קִשּׁוּי רָגִיל לְהוֹצִיא כוּ'. וְנִקְרָא קִשּׁוּי לְפִי שֶׁהוּא חֹלִי קָשֶׁה וּמַכְאוֹבָיו חֲזָקִים. הָרַמְבַּ"ם:
(ב) וְלֹא קָרִינַן בֵּיהּ בְּהֵמָה בַּבְּהֵמָה, דְּלָאו בְּתוֹכָהּ הוּא. רַשִׁ"י:
(ג) דְּקָרִינַן בֵּיהּ בְּהֵמָה בַּבְּהֵמָה תֹּאכֵלוּ, דִּכְתִיב וְכָל בְּהֵמָה מַפְרֶסֶת וְגוֹ'‏ בַּבְּהֵמָה תֹּאכֵלוּ, פֵּרֵשׁ רַשִׁ"י שָׁדֵי בְּהֵמָה דְרֵישֵׁיהּ אַבְּהֵמָה דְסֵיפֵיהּ וְדָרֵישׁ הָכִי, בְּהֵמָה הַנִּמְצֵאת בַּבְּהֵמָה תֹּאכֵלוּ:
(ד) אָסוּר. דִּכְתִיב אוֹתָהּ תֹּאכֵלוּ, אוֹתָהּ כְּשֶׁהִיא שְׁלֵמָה וְלֹא חֲסֵרָה. גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י, כְּשֶׁהִיא שְׁלֵמָה אֱכֹל כָּל הַנִּמְצָא בְתוֹכָהּ, אֲבָל כְּשֶׁהִיא חֲסֵרָה, שֶׁהַנִּמְצָא בְתוֹכָהּ מִגּוּפָהּ הוּא שֶׁנֶּחְסַר מִמֶּנָּה, לֹא תֹאכַל הַנִּמְצָא:
(ה) וּתְמֵהַנִי, דִּבְמִשְׁנָה ב'‏ פֶּרֶק ג'‏ כָּתַב הָרַ"ב דְּנִקַּב הַטְּחוֹל טְרֵפָה, וּמִי גָרַע נֶחְתַּךְ מִנִּקַּב. וּבַגְּמָרָא מְשָׁנֵי, דְּהוּא הַדִּין בְּהֵמָה נַמִּי אֲסוּרָה. וְאַיְדֵי דְתָנָא רֵישָׁא בַּחֲתִיכָה עַצְמָהּ, תָּנָא אִסּוּרָא דְסֵיפָא נַמִּי בַּחֲתִיכָה. תּוֹסְפוֹת יוֹם טוֹב. וְהַתּוֹסָפוֹת חֲדָשִׁים מְיַשֵּׁב, חֲדָא דְנֶחְתַּךְ קִיל לְדַעַת כַּמָּה פּוֹסְקִים. וְעוֹד יֵשׁ לוֹמַר דְּמַיְרֵי בְּנֶחְתַּךְ בְּמָקוֹם הַדַּק:
(ו) הַכְּלָל. לַאֲתוֹיֵי קָלוּט בִּמְעֵי פָרָה לְרַבִּי שִׁמְעוֹן. דְּאַף עַל גַּב דְּאָמַר רַבִּי שִׁמְעוֹן קָלוּט בִּמְעֵי פָרָה אָסוּר, הָנֵי מִילֵי הֵיכָא דְיָצָא לַאֲוִיר הָעוֹלָם, אֲבָל בִּמְעֵי אִמּוֹ שָׁרֵי. גְּמָרָא:

(א) וּמִיהוּ סְתָם מַתְנִיתִין דְּסוֹף פֶּרֶק ז'‏ דִּבְכוֹרוֹת סְבִירָא לֵיהּ כְּמַאן דְּאָמַר בַּגְּמָרָא נוֹהֵג אַף בִּזְכָרִים, דְּאוֹתוֹ מַשְׁמַע זָכָר, וּבְנוֹ הַיְנוּ מִי שֶׁכָּרוּךְ אַחֲרָיו וְהַיְנוּ נְקֵבָה, הִלְכָּךְ נוֹהֵג בְּתַרְוַיְהוּ כוּ'. הִלְכָּךְ אִי וַדַּאי לָן שֶׁהוּא אָבִיו אָסוּר לְשָׁחֲטוֹ בְּיוֹם אֶחָד. הָרֹא"שׁ:

(ג) כֵּיוָן דְּאִיכָּא דוּכְתָּא דַהֲוֵי לְצֹרֶךְ, כְּגוֹן בְּרֵאשִׁית הַגֵּז. תּוֹסָפוֹת:

(ט) רַשִׁ"י. וְלִישְׁנָא דְסוּגְיָא דְּרֵישׁ פֶּרֶק ב'‏ נָקְטֵי, וְשָׁם פֵּרֵשׁ רַשִׁ"י שֶׁלֹּא יַחְתֹּךְ כָּל הַמִּפְרֶקֶת לִשְׁנַיִם. וְעַיֵּן בְּמַסֶּכֶת כֵּלִים פֶּרֶק ד'‏ מִשְׁנָה ב':
(י) כְּלוֹמַר בָּאֱלָל דְּרֵישָׁא קָאָמְרִינַן דִּבְכֻלְּהוּ אֱלָל קָאָמַר דְּמִצְטָרְפִין, וּבְאַפֵּי נַפְשַׁיְהוּ לֹא מְטַמְּאוּ. וְאָתָא רַבִּי יְהוּדָה לְמֵימַר בְּאֱלָל גִּיד מוֹדִינָא כוּ', אֲבָל בָּשָׂר שֶׁפְּלַטְתּוֹ כוּ'. רַשִׁ"י:
(יא) וּבַגְּמָרָא, וְהוּא שֶׁכִּנְּסוֹ. וּפֵרֵשׁ רַשִׁ"י דְּגַלֵּי דַעְתֵּיהּ דְּלֹא בָטְלֵי מֵעִקָּרָא. אֲבָל נִתְכַּנֵּס מֵאֵלָיו, כְּגוֹן שֶׁכִּנְּסוּהוּ תִּינוֹקוֹת אַחַר הֶפְשֵׁט, לֹא, הוֹאִיל וּבָטֵל מִתְּחִלָּה וְנַעֲשָׂה עוֹר, תּוּ לֹא הָדַר הֲוֵי בָּשָׂר:
(יב) רַשִׁ"י. דְּאִי אַאֲכִילָה מַאי בָּעֵי מְכֻנָּס דְּהַיְנוּ שֶׁכִּנְּסוֹ, הָא כֵּיוָן דַּאֲכָלוֹ אֵין כִּנּוּס גְּדוֹלָה מִזּוֹ. וְהָא דְלֹא נָקַט טָמֵא, דְּלֹא תֵימָא מִדְּרַבָּנָן, לְהָכִי קָאָמַר חַיָּבִים עָלָיו. תּוֹסָפוֹת:

(א) תַּלְמוּד לוֹמַר כוּ'. וְאִי לָאו קַל וָחֹמֶר לֹא הֲוָה צָרִיךְ קְרָא, אַף עַל גַּב דִּסְתָמָא כְּתִיב דְּמַשְׁמַע בֵּין חֻלִּין בֵּין מֻקְדָּשִׁין דַּהֲוָה אֲמֵינָא דָּבָר הַלָּמֵד מֵעִנְיָנוֹ, בַּמֶּה הַכָּתוּב מְדַבֵּר, בְּחֻלִּין, דִּכְתִיב לְעֵיל מֵהַאי קְרָא גֵּז צֹאנְךָ, וְרֵאשִׁית הַגֵּז אֵינוֹ נוֹהֵג בְּמֻקְדָּשִׁין. תּוֹסָפוֹת:

=== 11 | chapter 11 ===
(א) שֶׁכֵּן רַבִּי אִלָּעִאי פּוֹטֵר נַמִּי בְּהוּ מֵהַאי טַעֲמָא דְּיָלֵיף לֵיהּ נְתִינָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
(ב) מִשּׁוּם דְּאַשְׁכְּחָן אֲמוֹרָאֵי טוּבָא דְּנָהֲגוּ:
(ג) בַּגְּמָרָא פָּרְכִינַן עֲלָהּ מֵהָא דְאָמְרִינַן דְּקָדָשִׁים אֲפִלּוּ שֶׁל בֶּדֶק הַבַּיִת אֲסוּרִים בְּגִיזָה כְּמוֹ שֶׁכָּתוּב פֶּרֶק י'‏ מִשְׁנָה ב'. וּמַסִּיק, בַּמַּקְדִּישׁ חוּץ מִגִּיזָה, וּכְחָשָׁה דְגִיזָה. וּפֵרוּשׁ כְּחָשָׁה דְגִיזָה, שֶׁהַגִּיזָה הַנִּטֶּלֶת מַכְחֶשֶׁת הַבָּשָׂר, וְאִי לָאו דְּאָמַר חוּץ הָיָה אָסוּר לְגָזְזָהּ [שֶׁלֹּא לְהַכְחִישָׁהּ]. וְהָנֵי חוּץ מִשּׁוּם קָדְשֵׁי בֶדֶק הַבַּיִת נָקְטֵיהּ, דְּאִי קָדְשֵׁי מִזְבֵּחַ אֲפִלּוּ כִּי אָמַר חוּץ לֹא מַהֲנֵי, דַּאֲפִלּוּ לֹא קַדִּישׁ אֶלָּא אֵבָר מִמֶּנָּה פָּשְׁטָה בְּכֻלָּהּ:
(ד) בִּרְחֵלוֹת. וְלֹא בָעִזִּים. אַף עַל גַּב דְּעִזִּים אִקְּרֵי נַמִּי צֹאן דִּכְתִיב [בְּרֵאשִׁית כז] לֶךְ נָּא אֶל הַצֹּאן וְקַח לִי מִשָּׁם שְׁנֵי גְדָיֵי עִזִּים. תּוֹסָפוֹת:
(ה) בַּגְּמָרָא, וְנֵילַף גִּיזָה גִּיזָה מִבְּכוֹר כוּ'. וּמַסְקִינַן, אֶלָּא כִּדְאָמַר רַבִּי יְהוֹשֻׁעַ בֶּן לֵוִי, לַעֲמֹד לְשָׁרֵת [דְּסָמוּךְ לְגֵז כְּתִיב], דָּבָר הָרָאוּי לְשָׁרֵת, הָכִי נַמִּי דָּבָר הָרָאוּי לְשָׁרֵת. פֵּרֵשׁ רַשִׁ"י לִתְכֵלֶת דְּבִגְדֵי שְׂרָד, וְהָתָם צֶמֶר בָּעִינַן, וְאֵין צֶמֶר אֶלָּא שֶׁל רְחֵלִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(א) לֹא בְמֻקְדָּשִׁים. דְּאָמַר קְרָא שַׁלֵּחַ וְגוֹ', יָצָא זֶה שֶׁאִי אַתָּה מְצֻוֶּה לְשַׁלְּחוֹ אֶלָּא לַהֲבִיאוֹ לַגִּזְבָּר, דְּאַף עַל גַּב דְּמָרְדוּ לֹא פָקְעוּ קְדֻשָּׁתַיְהוּ דְּכָל הֵיכָא דְאִיתֵיהּ בֵּי גַּזָּא דְרַחֲמָנָא אִיתָא דִּכְתִיב לַה'‏ הָאָרֶץ וּמְלוֹאָהּ [תְּהִלִּים כד]. גְּמָרָא:
